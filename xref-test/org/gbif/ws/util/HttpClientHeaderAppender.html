11  /**
12   * A very specific logback appender that looks for http headers as logevents coming apache http client.
13   * The headers are then aggregated into a map and can be used for further debugging.
14   * Useful when debugging http header traffic for example when dealing with Authentication headers.
15   * A simple logback file activating this appender would look like this:
16   * <pre>
17   * {@code
18   *   <appender name="HEADER" class="org.gbif.ws.util.HttpClientHeaderAppender" />
19   *   <logger name="org.apache.http.headers" level="DEBUG">
20   *     <appender-ref ref="HEADER"/>
21   *   </logger>
22   * }
23   * </pre>
24   */
25  public class HttpClientHeaderAppender extends UnsynchronizedAppenderBase<LoggingEvent> {
26  
27    private static Map<String, String> headers = Maps.newHashMap();
28    private static Pattern splitHeader = Pattern.compile(" (.+):(.+)");
29  
30    public static String getFirst(String key) {
31      return headers.get(key);
32    }
33  
34    public static boolean isEmpty() {
35      return headers.isEmpty();
36    }
37  
38    @Override
39    protected void append(LoggingEvent event) {
40      final String msg = event.getMessage();
41      if (msg.startsWith(">>")) {
42        Matcher m = splitHeader.matcher(msg);
43        if (m.find()) {
44          headers.put(m.group(1).trim(), m.group(2).trim());
45        }
46      }
47    }
48  
49  }
