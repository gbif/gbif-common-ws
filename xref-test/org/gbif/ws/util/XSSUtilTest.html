65    @ParameterizedTest

67    public void testContainsXSS(String value, boolean containsXSS, String stripXSSValue) {
68      assertEquals(containsXSS, XSSUtil.containsXSS(value));
69  
70      if (containsXSS) {
71        assertEquals(stripXSSValue, XSSUtil.stripXSS(value));
