44    @Test
45    public void testGetInjectable() {
46      ComponentContext mockComponentContext = Mockito.mock(ComponentContext.class);
47      Context mockContext = Mockito.mock(Context.class);
48  
49      Injectable<FacetedSearchRequest> injectable =
50        provider.getInjectable(mockComponentContext, mockContext, FacetedSearchRequest.class);
51      assertNotNull(injectable);
52  
53      injectable = provider.getInjectable(mockComponentContext, mockContext, String.class);
54      assertNull(injectable);
55    }

57    @Test
58    public void testGetValue() {
59      // mock context and request
60      HttpContext ctx = Mockito.mock(HttpContext.class);
61      HttpRequestContext req = Mockito.mock(HttpRequestContext.class);
62      Mockito.when(ctx.getRequest()).thenReturn(req);
63  
64      MultivaluedMap<String, String> params = new MultivaluedMapImpl();
65      Mockito.when(req.getQueryParameters()).thenReturn(params);
66  
67      // test with real facets
68      params.clear();
69      params.add(PARAM_FACET, TestSearchParameter.FIRSTNAME.name());
70      params.add(PARAM_FACET, TestSearchParameter.LASTNAME.name());
71      params.add(PARAM_FACET_MULTISELECT, Boolean.TRUE.toString());
72      FacetedSearchRequest<TestSearchParameter> x = (FacetedSearchRequest<TestSearchParameter>) provider.getValue(ctx);
73      assertTrue(x.getFacets().contains(TestSearchParameter.FIRSTNAME));
74      assertTrue(x.getFacets().contains(TestSearchParameter.LASTNAME));
75      assertTrue(x.isMultiSelectFacets());
76  
77      // test with real facets, case insensitive
78      params.clear();
79      params.add(PARAM_FACET, "firstname");
80      params.add(PARAM_FACET, "LastName");
81      x = (FacetedSearchRequest<TestSearchParameter>) provider.getValue(ctx);
82      assertTrue(x.getFacets().contains(TestSearchParameter.FIRSTNAME));
83      assertTrue(x.getFacets().contains(TestSearchParameter.LASTNAME));
84  
85      // test with wrong facet names
86      params.clear();
87      params.add(PARAM_FACET, "noname");
88      params.add(PARAM_FACET, "bisrtname");
89      x = (FacetedSearchRequest<TestSearchParameter>) provider.getValue(ctx);
90      assertTrue(x.getFacets().isEmpty());
91  
92    }
