32    @Test
33    public void testGetInjectable() {
34      ComponentContext mockComponentContext = Mockito.mock(ComponentContext.class);
35      Context mockContext = Mockito.mock(Context.class);
36  
37      Injectable<NameUsageSearchRequest> injectable = provider.getInjectable(mockComponentContext, mockContext, NameUsageSearchRequest.class);
38      assertNotNull(injectable);
39  
40      injectable = provider.getInjectable(mockComponentContext, mockContext, String.class);
41      assertNull(injectable);
42    }
43  
44    @Test
45    public void testGetValue() {
46      // mock context and request
47      HttpContext ctx = Mockito.mock(HttpContext.class);
48      HttpRequestContext req = Mockito.mock(HttpRequestContext.class);
49      Mockito.when(ctx.getRequest()).thenReturn(req);
50  
51      MultivaluedMap<String, String> params = new MultivaluedMapImpl();
52      Mockito.when(req.getQueryParameters()).thenReturn(params);
53  
54      // test with real facets
55      params.clear();
56      params.add(PARAM_FACET, NameUsageSearchParameter.HABITAT.name());
57      params.add(PARAM_FACET, NameUsageSearchParameter.DATASET_KEY.name());
58      params.add(PARAM_HIGHLIGHT, Boolean.TRUE.toString());
59      params.add(PARAM_HIGHLIGHT_CONTEXT, String.valueOf(250));
60      params.add(PARAM_HIGHLIGHT_FIELD, NameUsageSearchRequest.QueryField.DESCRIPTION.name());
61      params.add(PARAM_HIGHLIGHT_FIELD, NameUsageSearchRequest.QueryField.VERNACULAR.name());
62      params.add(PARAM_QUERY_FIELD, NameUsageSearchRequest.QueryField.VERNACULAR.name());
63      params.add(PARAM_QUERY_FIELD, NameUsageSearchRequest.QueryField.SCIENTIFIC.name());
64  
65      NameUsageSearchRequest x = provider.getValue(ctx);
66      assertTrue(x.getFacets().contains(NameUsageSearchParameter.HABITAT));
67      assertTrue(x.getFacets().contains(NameUsageSearchParameter.DATASET_KEY));
68      assertEquals(2, x.getFacets().size());
69  
70      assertTrue(x.isHighlight());
71      assertEquals((Integer)250, x.getHighlightContext());
72  
73      assertEquals(2, x.getHighlightFields().size());
74      assertTrue(x.getHighlightFields().contains(NameUsageSearchRequest.QueryField.DESCRIPTION));
75      assertTrue(x.getHighlightFields().contains(NameUsageSearchRequest.QueryField.VERNACULAR));
76  
77      assertEquals(2, x.getQueryFields().size());
78      assertTrue(x.getQueryFields().contains(NameUsageSearchRequest.QueryField.VERNACULAR));
79      assertTrue(x.getQueryFields().contains(NameUsageSearchRequest.QueryField.SCIENTIFIC));
80    }
81

85    }
