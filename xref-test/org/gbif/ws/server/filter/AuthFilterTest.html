120   private void assertPrincipalName(String username) {
121     ContainerRequest req = filter.filter(mockRequest);
122     if (username == null) {
123       verify(req, atLeastOnce()).setSecurityContext(secCtxCaptor.capture());
124       assertNull(secCtxCaptor.getValue().getUserPrincipal());
125       assertFalse(secCtxCaptor.getValue().isUserInRole("ADMIN"));
126     } else {
127       verify(req, atLeastOnce()).setSecurityContext(secCtxCaptor.capture());
128       assertEquals(username, secCtxCaptor.getValue().getUserPrincipal().getName());
129       assertTrue(secCtxCaptor.getValue().isUserInRole("ADMIN") || secCtxCaptor.getValue().isUserInRole("USER"));
130     }
131   }
132 
133   /**
134    * @param expectedType if null expect to be authorized
135    */
136   private void assertResponseStatus(Response.Status expectedType) {
137     boolean webexcept = false;
138     try {
139       ContainerRequest req = filter.filter(mockRequest);
140     } catch (WebApplicationException e) {
141       webexcept = true;
142       assertWebApplicationExceptionOfType(e, expectedType);
143     }
144     if (!webexcept && expectedType != null) {
145       fail("Expected WebApplicationException of type " + expectedType);
146     }
147   }
148 
149   private void assertWebApplicationExceptionOfType(WebApplicationException e, Response.Status expectedType) {
150     assertTrue(expectedType.getStatusCode() == e.getResponse().getStatus());
151   }
152 
153   private void setMockAuth(String authentication) {
154     when(mockRequest.getHeaderValue(ArgumentMatchers.<String>eq(ContainerRequest.AUTHORIZATION))).thenReturn(authentication);
155   }
