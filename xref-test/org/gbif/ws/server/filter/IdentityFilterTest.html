85    @Before
86    public void setUp() {
87      authService = GbifAuthService.multiKeyAuthService(GbifAuthServiceTest.buildAppKeyMap());
88      identityFilter = new IdentityFilter(new IdentityAccessService() {
89  
90        @Nullable
91        @Override
92        public GbifUser get(String username) {
93          return TEST_USERS.get(username);
94        }
95  
96        @Nullable
97        @Override
98        public GbifUser authenticate(String username, String password) {
99          return Optional.ofNullable(get(username))
100                 .filter( u -> u.getPasswordHash().equals(password))
101                 .orElse(null);
102       }
103     }, authService);
104   }

106   private static String toAuthorizationString(GbifUser user) {

110   private static String toAuthorizationString(String user, String password) {

137   /**
138    * Test a user authenticated by appkey.
139    * @throws Exception
140    */
141   @Test
142   public void testUserByAppKey() throws Exception {
143     ContainerRequest req = GbifAuthServiceTest.createMockContainerRequestFor(heinz.getUserName(),
144             GbifAuthService.singleKeyAuthService(APPKEY, APPSECRET));
145     ContainerRequest filteredReq = identityFilter.filter(req);
146     assertNotNull(filteredReq.getUserPrincipal());
147     assertEquals(heinz.getUserName(), filteredReq.getUserPrincipal().getName());
148     assertFalse(filteredReq.isUserInRole(UserRole.REGISTRY_ADMIN.name()));
149   }
150 
151   private void assertPrincipalName(String expectedUsername, ContainerRequest request) {
152     ContainerRequest req = identityFilter.filter(request);
153     SecurityContext securityContext = req.getSecurityContext();
154 
155     if (expectedUsername == null) {
156       assertNull(securityContext.getUserPrincipal());
157       assertFalse(securityContext.isUserInRole("ADMIN"));
158     } else {
159       assertEquals(expectedUsername, securityContext.getUserPrincipal().getName());
160       assertTrue(securityContext.isUserInRole("ADMIN") || securityContext.isUserInRole("USER"));
161     }
162   }

164   /**
165    * Creates a mock {@link ContainerRequest} that will return the provided authorization string in the
166    * "Authorization" header.
167    *
168    * @param authorization
169    * @return
170    */
171   private ContainerRequest getMockRequestAuthorization(String authorization) {
172     ContainerRequest cr = mock(ContainerRequest.class);
173     when(cr.getHeaderValue(ArgumentMatchers.eq(ContainerRequest.AUTHORIZATION))).thenReturn(authorization);
174     when(cr.getSecurityContext()).thenCallRealMethod();
175     Mockito.doCallRealMethod().when(cr).setSecurityContext(any(SecurityContext.class));
176     return cr;
177   }
