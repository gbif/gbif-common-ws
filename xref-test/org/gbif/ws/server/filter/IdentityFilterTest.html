90    @Before
91    public void setUp() {
92      authService = GbifAuthService.multiKeyAuthService(GbifAuthServiceTest.buildAppKeyMap());
93      identityFilter = new IdentityFilter(new IdentityAccessService() {
94  
95        @Nullable
96        @Override
97        public GbifUser get(String username) {
98          return TEST_USERS.get(username);
99        }
100 
101       @Nullable
102       @Override
103       public GbifUser authenticate(String username, String password) {
104         return Optional.ofNullable(get(username))
105                 .filter( u -> u.getPasswordHash().equals(password))
106                 .orElse(null);
107       }
108     }, authService);

111   private static String toAuthorizationString(GbifUser user) {

115   private static String toAuthorizationString(String user, String password) {
116     return new String(Base64.encode((user + ":" + password).getBytes(StandardCharsets.UTF_8)), StandardCharsets.UTF_8);
117   }

148    * Test a user authenticated by appkey.
149    * @throws Exception
150    */
151   @Test
152   public void testUserByAppKey() throws Exception {
153     ContainerRequest req = GbifAuthServiceTest.createMockContainerRequestFor(heinz.getUserName(),
154             GbifAuthService.singleKeyAuthService(APPKEY, APPSECRET));
155     ContainerRequest filteredReq = identityFilter.filter(req);
156     assertNotNull(filteredReq.getUserPrincipal());
157     assertEquals(heinz.getUserName(), filteredReq.getUserPrincipal().getName());
158     assertFalse(filteredReq.isUserInRole(UserRole.REGISTRY_ADMIN.name()));
159   }
160 
161   private void assertPrincipalName(String expectedUsername, ContainerRequest request) {
162     ContainerRequest req = identityFilter.filter(request);
163     SecurityContext securityContext = req.getSecurityContext();
164 
165     if (expectedUsername == null) {
166       assertNull(securityContext.getUserPrincipal());
167       assertFalse(securityContext.isUserInRole("ADMIN"));
168     } else {
169       assertEquals(expectedUsername, securityContext.getUserPrincipal().getName());
170       assertTrue(securityContext.isUserInRole("ADMIN") || securityContext.isUserInRole("USER"));
171     }
172   }

174   /**
175    * Creates a mock {@link ContainerRequest} that will return the provided authorization string in the
176    * "Authorization" header.
177    *
178    * @param authorization
179    * @return
180    */
181   private ContainerRequest getMockRequestAuthorization(String authorization) {
182     ContainerRequest cr = mock(ContainerRequest.class);
183     when(cr.getHeaderValue(ArgumentMatchers.eq(ContainerRequest.AUTHORIZATION))).thenReturn(authorization);
184     when(cr.getSecurityContext()).thenCallRealMethod();
185     Mockito.doCallRealMethod().when(cr).setSecurityContext(any(SecurityContext.class));
186     return cr;
187   }
