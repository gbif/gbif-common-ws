89     * @param defaultPort default port to use should no properties be provided
90     * @return The port, or the default
91     * @throws TestContainerException Should a property be provided but be an illegal argument

93    protected static int getTestPort(int defaultPort) throws TestContainerException {
94      for (String property : SYSTEM_PROPERTY_TEST_PORT) {
95        String port = System.getProperty(property);
96        if (port != null) {
97          try {
98            return Integer.parseInt(port);
99          } catch (NumberFormatException e) {
100           throw new TestContainerException(
101             "illegal value[" + port +"] supplied for property[" + property + "] which should defined a port number to use.", e);
102         }
103       }
104     }
