031/**
032 * The GBIF authentication scheme is modelled after the Amazon scheme on how to sign REST HTTP requests
033 * using a private key. It uses the standard HTTP Authorization header to transport the following information:
034 * Authorization: GBIF applicationKey:signature
035 *
036 * <br/>
037 * The header starts with the authentication scheme (GBIF), followed by the plain applicationKey (the public key)
038 * and a unique signature for the very request which is generated using a fixed set of request attributes
039 * which are then encrypted by a standard HMAC-SHA1 algorithm.
040 *
041 * <br/>
042 * A POST request with a GBIF header would look like this:
043 *
044 * <pre>
045 * POST /dataset HTTP/1.1
046 * Host: api.gbif.org
047 * Date: Mon, 26 Mar 2007 19:37:58 +0000
048 * x-gbif-user: trobertson
049 * Content-MD5: LiFThEP4Pj2TODQXa/oFPg==
050 * Authorization: GBIF gbif.portal:frJIUN8DYpKDtOLCwo//yllqDzg=
051 * </pre>
052 *
053 * When signing an HTTP request in addition to the Authorization header some additional custom headers are added
054 * which are used to sign and digest the message.
055 * <br/>
056 * x-gbif-user is added to transport a proxied user in which the application is acting.
057 * <br/>
058 * Content-MD5 is added if a body entity exists.
059 * See Content-MD5 header specs: http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.15
060 */
061@Singleton
062public class GbifAuthService {
063
064  private static final Logger LOG = LoggerFactory.getLogger(GbifAuthService.class);
065
066  private static final String ALGORITHM = "HmacSHA1";
067  private static final String CHAR_ENCODING = "UTF8";
068  public static final String HEADER_AUTHORIZATION = "Authorization";
069  public static final String HEADER_CONTENT_TYPE = "Content-Type";
070  public static final String HEADER_CONTENT_MD5 = "Content-MD5";
071  public static final String GBIF_SCHEME = "GBIF";
072  public static final String HEADER_GBIF_USER = "x-gbif-user";
073  public static final String HEADER_ORIGINAL_REQUEST_URL = "x-url";
074  private static final char NEWLINE = '\n';
075  private static final Pattern COLON_PATTERN = Pattern.compile(":");
076
077  private final ImmutableMap<String, String> keyStore;
078  private final String appKey;
079  private final ObjectMapper mapper = new JacksonJsonContextResolver().getContext(null);
080
081  private GbifAuthService(Map<String, String> appKeys, String appKey) {
082    keyStore = ImmutableMap.copyOf(appKeys);
083    this.appKey = appKey;
084    LOG.info("Initialised appkey store with {} keys", keyStore.size());
085  }
086
087  /**
088   * Creates a new GBIF authentication service for applications that need to validate requests
089   * for various application keys. Used by the GBIF webservice apps that require authentication.
090   */
091  public static GbifAuthService multiKeyAuthService(Map<String, String> appKeys) {
092    return new GbifAuthService(appKeys, null);
093  }
094
095  /**
096   * Creates a new GBIF authentication service for clients that want to sign requests always using a single
097   * application key. Used by the GBIF portal and other trusted applications that need to proxy a user.
098   */
099  public static GbifAuthService singleKeyAuthService(String appKey, String secret) {
100    LOG.info("Initialising auth service with key {}", appKey);
101    return new GbifAuthService(ImmutableMap.of(appKey, secret), appKey);
102  }
103
104  /**
105   * Extracts the information to be encrypted from a request and concatenates them into a single String.
106   * When the server receives an authenticated request, it compares the computed request signature
107   * with the signature provided in the request in StringToSign.
108   * For that reason this string may only contain information also available in the exact same form to the server.
109   *
110   * @return unique string for a request
111   *
112   * @see <a href="http://docs.amazonwebservices.com/AmazonS3/latest/dev/RESTAuthentication.html">AWS Docs</a>
113   */
114  private static String buildStringToSign(ContainerRequest request) {
115    StringBuilder sb = new StringBuilder();
116
117    sb.append(request.getMethod());
118    sb.append(NEWLINE);
119    // custom header set by varnish overrides real URI
120    // see http://dev.gbif.org/issues/browse/GBIFCOM-137
121    if (request.getRequestHeaders().containsKey(HEADER_ORIGINAL_REQUEST_URL)) {
122      sb.append(request.getRequestHeaders().getFirst(HEADER_ORIGINAL_REQUEST_URL));
123    } else {
124      sb.append(getCanonicalizedPath(request.getRequestUri()));
125    }
126
127    appendHeader(sb, request.getRequestHeaders(), HEADER_CONTENT_TYPE, false);
128    appendHeader(sb, request.getRequestHeaders(), HEADER_CONTENT_MD5, true);
129    appendHeader(sb, request.getRequestHeaders(), HEADER_GBIF_USER, true);
130
131    return sb.toString();
132  }
133
134  /**
135   * Build the string to be signed for a client request by extracting header information from the request.
136   * For PUT/POST requests that contain a body content it is required that the Content-MD5 header
137   * is already present on the request instance!
138   */
139  private static String buildStringToSign(ClientRequest request) {
140    StringBuilder sb = new StringBuilder();
141
142    sb.append(request.getMethod());
143    sb.append(NEWLINE);
144    sb.append(getCanonicalizedPath(request.getURI()));
145
146    appendHeader(sb, request.getHeaders(), HEADER_CONTENT_TYPE, false);
147    appendHeader(sb, request.getHeaders(), HEADER_CONTENT_MD5, true);
148    appendHeader(sb, request.getHeaders(), HEADER_GBIF_USER, true);
149
150    return sb.toString();
151  }
152
153  private static void appendHeader(StringBuilder sb, MultivaluedMap<String, ?> request, String header, boolean caseSensitive) {
154    if (request.containsKey(header)) {
155      sb.append(NEWLINE);
156      if (caseSensitive) {
157        sb.append(request.getFirst(header));
158      } else {
159        sb.append(request.getFirst(header).toString().toLowerCase());
160      }
161    }
162  }
163
164  /**
165   * @return an absolute uri of the resource path alone, excluding host, scheme and query parameters
166   */
167  private static String getCanonicalizedPath(URI uri) {
168    return uri.normalize().getPath();
169  }
170
171  private static String buildAuthHeader(String applicationKey, String signature) {
172    return GBIF_SCHEME + " " + applicationKey + ':' + signature;
173  }
174
175  /**
176   * Generates a Base64 encoded HMAC-SHA1 signature of the passed in string with the given secret key.
177   * See Message Authentication Code specs http://tools.ietf.org/html/rfc2104
178   * @param stringToSign the string to be signed
179   * @param secretKey the secret key to use in the
180   */
181  private String buildSignature(String stringToSign, String secretKey) {
182    try {
183      Mac mac = Mac.getInstance(ALGORITHM);
184      SecretKeySpec secret = new SecretKeySpec(secretKey.getBytes(Charset.forName("UTF8")), ALGORITHM);
185      mac.init(secret);
186      byte[] digest = mac.doFinal(stringToSign.getBytes());
187
188      return new String(Base64.encode(digest), "ASCII");
189
190    } catch (NoSuchAlgorithmException e) {
191      throw new RuntimeException("Cant find " + ALGORITHM + " message digester", e);
192    } catch (UnsupportedEncodingException e) {
193      throw new RuntimeException("Unsupported character encoding " + CHAR_ENCODING, e);
194    } catch (InvalidKeyException e) {
195      throw new RuntimeException("Invalid secret key " + secretKey, e);
196    }
197  }
198
199  /**
200   * Signs a request by adding a Content-MD5 and Authorization header.
201   * For PUT/POST requests that contain a body entity the Content-MD5 header is created using the same
202   * JSON mapper for serialization as the clients use.
203   *
204   * Other format than JSON are not supported currently !!!
205   */
206  public void signRequest(String username, ClientRequest request) {
207    Preconditions.checkNotNull(appKey, "To sign request a single application key is required");
208    // first add custom GBIF headers so we can use them to build the string to sign
209
210    // the proxied username
211    request.getHeaders().putSingle(HEADER_GBIF_USER, username);
212
213    // the canonical path header
214    request.getHeaders().putSingle(HEADER_ORIGINAL_REQUEST_URL, getCanonicalizedPath(request.getURI()));
215    // adds content md5
216    if (request.getEntity() != null) {
217      request.getHeaders().putSingle(HEADER_CONTENT_MD5, buildContentMD5(request.getEntity()));
218    }
219
220    // build the unique string to sign
221    final String stringToSign = buildStringToSign(request);
222    // find private key for this app
223    final String secretKey = getPrivateKey(appKey);
224    if (secretKey == null) {
225      LOG.warn("Skip signing request with unknown application key: {}", appKey);
226      return;
227    }
228    // sign
229    String signature = buildSignature(stringToSign, secretKey);
230
231    // build authorization header string
232    String header = buildAuthHeader(appKey, signature);
233    // add authorization header
234    LOG.debug("Adding authentication header to request {} for proxied user {} : {}", request.getURI(), username, header);
235    request.getHeaders().putSingle(HEADER_AUTHORIZATION, header);
236  }
237
238  /**
239   * Generates the Base64 encoded 128 bit MD5 digest of the entire content string suitable for the
240   * Content-MD5 header value.
241   * See http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.15
242   */
243  private String buildContentMD5(Object entity) {
244    try {
245      byte[] content = mapper.writeValueAsBytes(entity);
246      return new String(Base64.encode(DigestUtils.md5(content)), "ASCII");
247
248    } catch (IOException e) {
249      LOG.error("Failed to serialize http entity [{}]", entity);
250      throw Throwables.propagate(e);
251    }
252  }

254  /**
255   * Tries to get the appkey from the request header.
256   * @param requestHeaderAccessor lambda expression to access the headers of a request.
257   * @return the appkey found or null
258   */
259  public static String getAppKeyFromRequest(RequestHeaderAccessor requestHeaderAccessor) {
260    if(StringUtils.startsWith(requestHeaderAccessor.getHeader(HEADER_AUTHORIZATION), GBIF_SCHEME + " ")) {
261      String[] values = COLON_PATTERN.split(requestHeaderAccessor.getHeader(HEADER_AUTHORIZATION).substring(5), 2);
262      if (values.length == 2) {
263        return values[0];
264      }
265    }
266    return null;
267  }

269  public boolean isValidRequest(ContainerRequest request) {
270    // parse auth header
271    final String authHeader = request.getHeaderValue(HEADER_AUTHORIZATION);
272    if (Strings.isNullOrEmpty(authHeader) || !authHeader.startsWith(GBIF_SCHEME + " ")) {
273      LOG.info(HEADER_AUTHORIZATION + " header is no GBIF scheme");
274      return false;
275    }
276
277    String[] values = COLON_PATTERN.split(authHeader.substring(5), 2);
278    if (values.length < 2) {
279      LOG.warn("Invalid syntax for application key and signature: {}", authHeader);
280      return false;
281    }
282
283    final String appKey = values[0];
284    final String signatureFound = values[1];
285    if (appKey == null || signatureFound == null) {
286      LOG.warn("Authentication header missing applicationKey or signature: {}", authHeader);
287      return false;
288    }
289
290    String secretKey = getPrivateKey(appKey);
291    if (secretKey == null) {
292      LOG.warn("Unknown application key: {}", appKey);
293      return false;
294    }
295    //
296    String stringToSign = buildStringToSign(request);
297    // sign
298    String signature = buildSignature(stringToSign, secretKey);
299    // compare signatures
300    if (signatureFound.equals(signature)) {
301      LOG.debug("Trusted application with matching signatures");
302      return true;
303
304    }
305    LOG.info("Invalid signature: {}", authHeader);
306    LOG.debug("StringToSign: {}", stringToSign);
307    return false;
308  }
309
310  private String getPrivateKey(String applicationKey) {
311    return keyStore.get(applicationKey);
312  }
