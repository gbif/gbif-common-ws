028/**
029 * The GBIF authentication scheme is modelled after the Amazon scheme on how to sign REST http requests
030 * using a private key. It uses the standard Http Authorization Header to transport the following information:
031 * Authorization: GBIF applicationKey:Signature
032 * The header starts with the authentication scheme (GBIF), followed by the plain applicationKey and a unique signature
033 * for the very request which is generated using a fixed set of request attributes which are then encrypted by
034 * a standard HMAC-SHA1 algorithm.
035 * A POST request with a GBIF header would look like this:
036 * <pre>
037 * POST /dataset HTTP/1.1
038 * Host: johnsmith.s3.amazonaws.com
039 * Date: Mon, 26 Mar 2007 19:37:58 +0000
040 * x-gbif-user: trobertson
041 * Content-MD5: 12NnTPXtoDX8k2g6QQ/2hw==
042 * Authorization: GBIF admin:frJIUN8DYpKDtOLCwo//yllqDzg=
043 * </pre>
044 *
045 * When signing an http request in addition to the Authentication header some additional custom headers are added
046 * which are used to sign and digest the message.
047 * <br/>
048 * x-gbif-user is added to transport a proxied user in which the application is acting.
049 * <br/>
050 * Content-MD5 is added if a body entity exists.
051 */
052@Singleton
053public class GbifAuthService {
054
055  private static final Logger LOG = LoggerFactory.getLogger(GbifAuthService.class);
056
057  private static final String ALGORITHM = "HmacSHA1";
058  private static final String CHAR_ENCODING = "UTF8";
059  public static final String HEADER_AUTHORIZATION = "Authorization";
060  public static final String HEADER_CONTENT_TYPE = "Content-Type";
061  public static final String HEADER_CONTENT_MD5 = "Content-MD5";
062  public static final String GBIF_SCHEME = "GBIF";
063  public static final String HEADER_GBIF_USER = "x-gbif-user";
064  public static final String HEADER_ORIGINAL_REQUEST_URL = "x-url";
065  private static final char NEWLINE = '\n';
066  private static final Pattern COLON_PATTERN = Pattern.compile(":");
067
068  private final ImmutableMap<String, String> keyStore;
069  private final String appKey;
070  private final ObjectMapper mapper = new JacksonJsonContextResolver().getContext(null);
071
072  private GbifAuthService(Map<String, String> appKeys, String appKey) {
073    keyStore = ImmutableMap.copyOf(appKeys);
074    this.appKey = appKey;
075    LOG.info("Initialised appkey store with {} keys", keyStore.size());
076  }
077
078  /**
079   * Creates a new GBIF authentication service for applications that need to validate requests
080   * for various application keys. Used by the GBIF webservice apps that require authentication.
081   */
082  public static GbifAuthService multiKeyAuthService(Map<String, String> appKeys) {
083    return new GbifAuthService(appKeys, null);
084  }
085
086  /**
087   * Creates a new GBIF authentication service for clients that want to sign requests always using a single
088   * application key. Used by the GBIF portal and other trusted applications that need to proxy a user.
089   */
090  public static GbifAuthService singleKeyAuthService(String appKey, String secret) {
091    LOG.info("Initialising auth service with key {}", appKey);
092    return new GbifAuthService(ImmutableMap.of(appKey, secret), appKey);
093  }
094
095  /**
096   * Extracts the information to be encrypted from a request and concatenates them into a single String.
097   * When the server receives an authenticated request, it compares the computed request signature
098   * with the signature provided in the request in StringToSign.
099   * For that reason this string may only contain information also available in the exact same form to the server.
100   *
101   * @return unique string for a request
102   *
103   * @see <a href="http://docs.amazonwebservices.com/AmazonS3/latest/dev/RESTAuthentication.html">AWS Docs</a>
104   */
105  private String buildStringToSign(ContainerRequest request) {
106    StringBuilder sb = new StringBuilder();
107    sb.append(request.getMethod());
108
109    sb.append(NEWLINE);
110    // custom header set by varnish overrides real URI
111    // see http://dev.gbif.org/issues/browse/GBIFCOM-137
112    if (request.getRequestHeaders().containsKey(HEADER_ORIGINAL_REQUEST_URL)) {
113      sb.append(request.getRequestHeaders().getFirst(HEADER_ORIGINAL_REQUEST_URL));
114    } else {
115      sb.append(getCanonicalizedPath(request.getRequestUri()));
116    }
117
118    sb.append(NEWLINE);
119    sb.append(request.getHeaderValue(HEADER_CONTENT_TYPE));
120
121    // include md5 hash if existing (only for PUT/POST with content)
122    if (request.getRequestHeaders().containsKey(HEADER_CONTENT_MD5)) {
123      sb.append(NEWLINE);
124      sb.append(request.getHeaderValue(HEADER_CONTENT_MD5));
125    }
126
127    sb.append(NEWLINE);
128    sb.append(request.getHeaderValue(HEADER_GBIF_USER));
129
130    return sb.toString();
131  }
132
133  /**
134   * Build the string to be signed for a client request by extracting header information from the request.
135   * For PUT/POST requests that contain a body content it is required that the Content-MD5 header
136   * is already present on the request instance!
137   */
138  private String buildStringToSign(ClientRequest request) {
139    StringBuilder sb = new StringBuilder();
140    sb.append(request.getMethod());
141
142    sb.append(NEWLINE);
143    sb.append(getCanonicalizedPath(request.getURI()));
144
145    sb.append(NEWLINE);
146    sb.append(request.getHeaders().getFirst(HEADER_CONTENT_TYPE));
147
148    // include md5 hash if existing (only for PUT/POST with content)
149    if (request.getHeaders().containsKey(HEADER_CONTENT_MD5)) {
150      sb.append(NEWLINE);
151      sb.append(request.getHeaders().getFirst(HEADER_CONTENT_MD5));
152    }
153
154    sb.append(NEWLINE);
155    sb.append(request.getHeaders().getFirst(HEADER_GBIF_USER));
156
157    return sb.toString();
158  }
159
160  /**
161   * @return an absolute uri of the resource path alone, excluding host, scheme and query parameters
162   */
163  private String getCanonicalizedPath(URI uri) {
164    return uri.normalize().getPath();
165  }
166
167  private String buildAuthHeader(String applicationKey, String signature) {
168    return GBIF_SCHEME + " " + applicationKey + ":" + signature;
169  }
170
171  private String buildSignature(String stringToSign, String secretKey) {
172    try {
173      Mac mac = Mac.getInstance(ALGORITHM);
174      SecretKeySpec secret = new SecretKeySpec(secretKey.getBytes(Charset.forName("UTF8")), ALGORITHM);
175      mac.init(secret);
176      byte[] digest = mac.doFinal(stringToSign.getBytes());
177
178      return new String(Base64.encode(digest), "ASCII");
179
180    } catch (NoSuchAlgorithmException e) {
181      throw new RuntimeException("Cant find " + ALGORITHM + " message digester", e);
182    } catch (UnsupportedEncodingException e) {
183      throw new RuntimeException("Unsupported character encoding " + CHAR_ENCODING, e);
184    } catch (InvalidKeyException e) {
185      throw new RuntimeException("Invalid secret key " + secretKey, e);
186    }
187  }
188
189  /**
190   * Signs a request by adding a Content-MD5 and Authorization header.
191   * For PUT/POST requests that contain a body entity the Content-MD5 header is created using the same
192   * JSON mapper for serialization as the clients use.
193   *
194   * Other format than JSON are not supported currently !!!
195   */
196  public void signRequest(String username, ClientRequest request) {
197    Preconditions.checkNotNull(appKey, "To sign request a single application key is required");
198    // first add custom GBIF headers so we can use them to build the string to sign
199
200    // the proxied username
201    request.getHeaders().putSingle(HEADER_GBIF_USER, username);
202    // the canonical path header
203    request.getHeaders().putSingle(HEADER_ORIGINAL_REQUEST_URL, getCanonicalizedPath(request.getURI()));
204    // adds content md5
205    if (request.getEntity() != null) {
206      request.getHeaders().putSingle(HEADER_CONTENT_MD5, buildContentMD5(request.getEntity()));
207    }
208
209    // build the unique string to sign
210    final String stringToSign = buildStringToSign(request);
211    // find private key for this app
212    final String secretKey = getPrivateKey(appKey);
213    if (secretKey == null) {
214      LOG.warn("Skip signing request with unknown application key: {}", appKey);
215      return;
216    }
217    // sign
218    String signature = buildSignature(stringToSign, secretKey);
219    // build authorization header string
220    String header = buildAuthHeader(appKey, signature);
221    // add authorization header
222    LOG.debug("Adding authentication header to request {} for proxied user {} : {}", request.getURI(), username, header);
223    request.getHeaders().putSingle(HEADER_AUTHORIZATION, header);
224  }
225
226  private String buildContentMD5(Object entity) {
227    try {
228      MessageDigest instance = MessageDigest.getInstance("MD5");
229      return new String(Base64.encode(instance.digest(mapper.writeValueAsBytes(entity))), "ASCII");
230
231    } catch (IOException e) {
232      LOG.error("Failed to serialize http entity [{}]", entity);
233      throw new RuntimeException(e);
234
235    } catch (NoSuchAlgorithmException e) {
236      LOG.error("MD5 algorithm not found");
237      throw new RuntimeException(e);
238    }
239  }

241  public boolean isValidRequest(ContainerRequest request) {
242    // parse auth header
243    final String authHeader = request.getHeaderValue(HEADER_AUTHORIZATION);
244    if (Strings.isNullOrEmpty(authHeader) || !authHeader.startsWith(GBIF_SCHEME + " ")) {
245      LOG.info(HEADER_AUTHORIZATION + " header is no GBIF scheme");
246      return false;
247    }
248
249    String[] values = COLON_PATTERN.split(authHeader.substring(5), 2);
250    if (values.length < 2) {
251      LOG.warn("Invalid syntax for application key and signature: {}", authHeader);
252      return false;
253    }
254
255    final String appKey = values[0];
256    final String signatureFound = values[1];
257    if (appKey == null || signatureFound == null) {
258      LOG.warn("Authentication header missing applicationKey or signature: {}", authHeader);
259      return false;
260    }
261
262    String secretKey = getPrivateKey(appKey);
263    if (secretKey == null) {
264      LOG.warn("Unknown application key: {}", appKey);
265      return false;
266    }
267    //
268    String stringToSign = buildStringToSign(request);
269    // sign
270    String signature = buildSignature(stringToSign, secretKey);
271    // compare signatures
272    if (signatureFound.equals(signature)) {
273      LOG.debug("Trusted application with matching signatures");
274      return true;
275
276    }
277    LOG.info("Invalid signature: {}", authHeader);
278    return false;
279  }
280
281  private String getPrivateKey(String appKey) {
282    return keyStore.get(appKey);
283  }
