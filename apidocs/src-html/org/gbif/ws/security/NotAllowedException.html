001package org.gbif.ws.security;
002
003import org.gbif.api.model.common.User;
004
005import javax.ws.rs.WebApplicationException;
006import javax.ws.rs.core.Response;
007
008/**
009 * HTTP 403 jersey exception when a request has a valid authentication, but the authenticated user is not allowed
010 * to execute the resource.
011 */
012public class NotAllowedException extends WebApplicationException {
013
014     public NotAllowedException() {
015         super(Response.status(Response.Status.FORBIDDEN).build());
016     }
017
018     /**
019      * @param message the String that is the entity of the 403 response.
020      */
021     public NotAllowedException(String message) {
022         super(Response.status(Response.Status.FORBIDDEN).entity(message).type("text/plain").build());
023     }
024
025  /**
026   * @param user that is authenticated
027   */
028  public NotAllowedException(User user) {
029    this (user.getUserName() + " is not allowed to access this resource.");
030  }
031
032}
