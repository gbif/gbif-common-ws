001package org.gbif.ws.security;
002
003import java.nio.charset.StandardCharsets;
004import java.security.InvalidKeyException;
005import java.security.NoSuchAlgorithmException;
006import java.util.Base64;
007import javax.crypto.Mac;
008import javax.crypto.spec.SecretKeySpec;
009
010public abstract class BaseSigningService implements SigningService {
011
012  private static final String ALGORITHM = "HmacSHA1";
013
014  /**
015   * Generates a Base64 encoded HMAC-SHA1 signature of the passed request data with the secret key.
016   * See Message Authentication Code specs http://tools.ietf.org/html/rfc2104
017   *
018   * @param requestDataToSign the request data to be signed
019   * @param secretKey         the secret key
020   */
021  @Override
022  public String buildSignature(RequestDataToSign requestDataToSign, String secretKey) {
023    try {
024      Mac mac = Mac.getInstance(ALGORITHM);
025      SecretKeySpec secret = new SecretKeySpec(secretKey.getBytes(StandardCharsets.UTF_8),
026          ALGORITHM);
027      mac.init(secret);
028      byte[] digest = mac.doFinal(requestDataToSign.stringToSign().getBytes());
029
030      return new String(Base64.getEncoder().encode(digest), StandardCharsets.US_ASCII);
031    } catch (NoSuchAlgorithmException e) {
032      throw new RuntimeException("Cant find " + ALGORITHM + " message digester", e);
033    } catch (InvalidKeyException e) {
034      throw new RuntimeException("Invalid secret key " + secretKey, e);
035    }
036  }
037}
