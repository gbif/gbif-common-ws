001package org.gbif.ws.security;
002
003import com.google.common.collect.ImmutableMap;
004import com.google.common.collect.Maps;
005import java.io.IOException;
006import java.util.Properties;
007import org.gbif.utils.file.properties.PropertiesUtil;
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010import org.springframework.stereotype.Component;
011
012@Component
013public class FileSystemKeyStore implements KeyStore {
014
015  private static final Logger LOG = LoggerFactory.getLogger(FileSystemKeyStore.class);
016
017  private final ImmutableMap<String, String> store;
018
019  public FileSystemKeyStore(AppkeysConfigurationProperties appkeysConfiguration) {
020    try {
021      Properties props = PropertiesUtil.loadProperties(appkeysConfiguration.getFile());
022      store = Maps.fromProperties(props);
023    } catch (IOException e) {
024      throw new IllegalArgumentException(
025          "Property file path to application keys does not exist: "
026              + appkeysConfiguration.getFile(),
027          e);
028    }
029    LOG.info("Initialised appkey store with {} keys", store.size());
030  }
031
032  public String getPrivateKey(final String applicationKey) {
033    return store.get(applicationKey);
034  }
035}
