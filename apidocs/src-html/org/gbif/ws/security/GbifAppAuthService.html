027/**
028 * The GBIF authentication scheme is modelled after the Amazon scheme on how to sign REST http requests
029 * using a private key. It uses the standard Http Authorization Header to transport the following information:
030 * Authorization: GBIF applicationKey:Signature
031 * The header starts with the authentication scheme (GBIF), followed by the plain applicationKey and a unique signature
032 * for the very request which is generated using a fixed set of request attributes which are then encrypted by
033 * a standard HMAC-SHA1 algorithm.
034 * A simple GET request with a GBIF header would look like this:
035 * <pre>
036 * GET /name_usage/536123 HTTP/1.1
037 * Host: johnsmith.s3.amazonaws.com
038 * Date: Mon, 26 Mar 2007 19:37:58 +0000
039 * x-gbif-user: trobertson
040 * x-gbif-date: 16 Mar 2007 19:37:58 GMT
041 * Authorization: GBIF admin:frJIUN8DYpKDtOLCwo//yllqDzg=
042 * </pre>
043 * When signing an http request in addition to the Authentication header some additional custom headers are added
044 * which are used to sign and digest the message.
045 * <br/>
046 * x-gbif-user is added to transport a proxied user in which the application is acting.
047 * <br/>
048 * x-gbif-date is added because there is no way for a jersey client to access the automatically generated date header
049 * before the message is actually send.
050 * <br/>
051 * x-gbif-content-hash is added similar to the date above hashing the message body if it exists.
052 * This will only be for POST/PUT requests instead of using the regular Content-MD5
053 * which cannot be accessed in the jersey client.
054 */
055@Singleton
056public class GbifAppAuthService {
057
058  private static final Logger LOG = LoggerFactory.getLogger(GbifAppAuthService.class);
059
060  public static final String ALGORITHM = "HmacSHA1";
061  public static final String CHAR_ENCODING = "UTF8";
062  public static final String HEADER_AUTHORIZATION = "Authorization";
063  public static final String HEADER_CONTENT_TYPE = "Content-Type";
064  public static final String GBIF_SCHEME = "GBIF";
065  public static final String HEADER_GBIF_USER = "x-gbif-user";
066  public static final String HEADER_GBIF_DATE = "x-gbif-date";
067  public static final String HEADER_GBIF_CONTENT_HASH = "x-gbif-content-hash";
068  public static final String HEADER_ORIGINAL_REQUEST_URL = "x-url";
069  private static final Pattern COLON_PATTERN = Pattern.compile(":");
070
071  private final ImmutableMap<String, String> keyStore;
072
073  public GbifAppAuthService(Map<String, String> appKeys) {
074    keyStore = ImmutableMap.copyOf(appKeys);
075    LOG.info("Initialised appkey store with {} keys", keyStore.size());
076  }
077
078  public GbifAppAuthService(String appKey, String secret) {
079    keyStore = ImmutableMap.of(appKey, secret);
080    LOG.info("Initialised appkey store with key {}", appKey);
081  }
082
083  /**
084   * Extracts the information to be encrypted from a request and concatenates them into a single String.
085   * When the server receives an authenticated request, it compares the computed request signature
086   * with the signature provided in the request in StringToSign.
087   * For that reason this string may only contain information also available in the exact same form to the server.
088   *
089   * @return unique string for a request
090   *
091   * @see <a href="http://docs.amazonwebservices.com/AmazonS3/latest/dev/RESTAuthentication.html">AWS Docs</a>
092   */
093  private String buildStringToSign(ContainerRequest request) {
094    StringBuilder sb = new StringBuilder();
095    sb.append(request.getMethod());
096    sb.append("/n");
097    sb.append(request.getHeaderValue(HEADER_CONTENT_TYPE));
098    sb.append("/n");
099    sb.append(getCanonicalizedGbifHeaders(request.getRequestHeaders()));
100    sb.append("/n");
101    // custom header set by varnish overrides real URI
102    // see http://dev.gbif.org/issues/browse/GBIFCOM-137
103    if (request.getRequestHeaders().containsKey(HEADER_ORIGINAL_REQUEST_URL)) {
104      sb.append(request.getRequestHeaders().getFirst(HEADER_ORIGINAL_REQUEST_URL));
105    } else {
106      sb.append(getCanonicalizedPath(request.getRequestUri()));
107    }
108
109    return sb.toString();
110  }
111
112  /**
113   * Build the string to be signed for a client request
114   */
115  private String buildStringToSign(ClientRequest request) {
116    StringBuilder sb = new StringBuilder();
117    sb.append(request.getMethod());
118    sb.append("/n");
119    sb.append(request.getHeaders().getFirst(HEADER_CONTENT_TYPE));
120    sb.append("/n");
121    sb.append(getCanonicalizedGbifHeaders(request.getHeaders()));
122    sb.append("/n");
123    sb.append(getCanonicalizedPath(request.getURI()));
124    return sb.toString();
125  }
126
127  /**
128   * Grabs all gbif specific headers starting with x-gbif.
129   */
130  private String getCanonicalizedGbifHeaders(MultivaluedMap<String, ?> headers) {
131    StringBuilder sb = new StringBuilder();
132    List<String> gbifHeaders = Lists.newArrayList();
133    for (String hKey : headers.keySet()) {
134      if (hKey.startsWith("x-gbif")) {
135        for (Object val : headers.get(hKey)) {
136          gbifHeaders.add(hKey + ":" + val.toString());
137        }
138      }
139    }
140    Collections.sort(gbifHeaders);
141    for (String h : gbifHeaders) {
142      sb.append(h);
143      sb.append("/n");
144    }
145    return sb.toString();
146  }
147
148  /**
149   * @return an absolute uri of the resource path alone, excluding host, scheme and query parameters
150   */
151  private String getCanonicalizedPath(URI uri) {
152    return uri.normalize().getPath();
153  }
154
155  private String buildAuthHeader(String applicationKey, String signature) {
156    return GBIF_SCHEME + " " + applicationKey + ":" + signature;
157  }
158
159  private String buildSignature(String stringToSign, String secretKey) {
160    try {
161      Mac mac = Mac.getInstance(ALGORITHM);
162      SecretKeySpec secret = new SecretKeySpec(secretKey.getBytes(Charset.forName("UTF8")), ALGORITHM);
163      mac.init(secret);
164      byte[] digest = mac.doFinal(stringToSign.getBytes());
165
166      return new String(Base64.encode(digest), "ASCII");
167
168    } catch (NoSuchAlgorithmException e) {
169      throw new RuntimeException("Cant find " + ALGORITHM + " message digester", e);
170    } catch (UnsupportedEncodingException e) {
171      throw new RuntimeException("Unsupported character encoding " + CHAR_ENCODING, e);
172    } catch (InvalidKeyException e) {
173      throw new RuntimeException("Invalid secret key " + secretKey, e);
174    }
175  }
176
177  private void addGbifHeaders(String username, ClientRequest request) {
178    // adds the datestamp
179    request.getHeaders().putSingle(HEADER_GBIF_DATE, new Date().toGMTString());
180    // adds the proxied user name to the headers
181    request.getHeaders().putSingle(HEADER_GBIF_USER, username);
182    // adds the entities java hash code
183    if (request.getEntity() != null) {
184      request.getHeaders().putSingle(HEADER_GBIF_CONTENT_HASH, request.getEntity().hashCode());
185    }
186    // adds the canonical path header used for signing
187    request.getHeaders().putSingle(HEADER_ORIGINAL_REQUEST_URL, getCanonicalizedPath(request.getURI()));
188  }
189
190  public void signRequest(String appKey, String username, ClientRequest request) {
191    // first add custom GBIF headers so we can use them to build the string to sign
192    addGbifHeaders(username, request);
193    // build the unique string to sign
194    String stringToSign = buildStringToSign(request);
195    // find private key for this app
196    String secretKey = getPrivateKey(appKey);
197    if (secretKey == null) {
198      LOG.warn("Skip signing request with unknown application key: {}", appKey);
199      return;
200    }
201    // sign
202    String signature = buildSignature(stringToSign, secretKey);
203    // build authorization header string
204    String header = buildAuthHeader(appKey, signature);
205    // add authorization header
206    LOG.debug("Adding authentication header to request {} for proxied user {} : {}", request.getURI(), username, header);
207    request.getHeaders().putSingle(HEADER_AUTHORIZATION, header);
208  }

210  public boolean isValidRequest(ContainerRequest request) {
211    // parse auth header
212    final String authHeader = request.getHeaderValue(HEADER_AUTHORIZATION);
213    if (Strings.isNullOrEmpty(authHeader) || !authHeader.startsWith(GBIF_SCHEME + " ")) {
214      LOG.info(HEADER_AUTHORIZATION + " header is no GBIF scheme");
215      return false;
216    }
217
218    String[] values = COLON_PATTERN.split(authHeader.substring(5), 2);
219    if (values.length < 2) {
220      LOG.warn("Invalid syntax for application key and signature: {}", authHeader);
221      return false;
222    }
223
224    final String appKey = values[0];
225    final String signatureFound = values[1];
226    if (appKey == null || signatureFound == null) {
227      LOG.warn("Authentication header missing applicationKey or signature: {}", authHeader);
228      return false;
229    }
230
231    String secretKey = getPrivateKey(appKey);
232    if (secretKey == null) {
233      LOG.warn("Unknown application key: {}", appKey);
234      return false;
235    }
236    //
237    String stringToSign = buildStringToSign(request);
238    // sign
239    String signature = buildSignature(stringToSign, secretKey);
240    // compare signatures
241    if (signatureFound.equals(signature)) {
242      LOG.debug("Trusted application with matching signatures");
243      return true;
244
245    }
246    LOG.info("Invalid signature: {}", authHeader);
247    return false;
248  }
249
250  private String getPrivateKey(String appKey) {
251    return keyStore.get(appKey);
252  }
