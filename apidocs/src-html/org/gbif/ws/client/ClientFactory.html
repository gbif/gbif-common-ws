001package org.gbif.ws.client;
002
003import com.fasterxml.jackson.databind.ObjectMapper;
004import feign.Contract;
005import feign.Feign;
006import feign.InvocationHandlerFactory;
007import feign.RequestInterceptor;
008import feign.codec.Decoder;
009import feign.codec.Encoder;
010import feign.codec.ErrorDecoder;
011import org.gbif.ws.json.JacksonJsonObjectMapperProvider;
012import org.gbif.ws.security.Md5EncodeService;
013import org.gbif.ws.security.Md5EncodeServiceImpl;
014import org.gbif.ws.security.SecretKeySigningService;
015import org.gbif.ws.security.SigningService;
016
017@SuppressWarnings("unused")
018public class ClientFactory {
019
020  private String url;
021  private RequestInterceptor requestInterceptor;
022  private Decoder decoder;
023  private Encoder encoder;
024  private ErrorDecoder errorDecoder;
025  private Contract contract;
026  private InvocationHandlerFactory invocationHandlerFactory;
027
028  /**
029   * Read-only clients factory.
030   */
031  public ClientFactory(String url) {
032    this.url = url;
033    ObjectMapper objectMapper = JacksonJsonObjectMapperProvider.getObjectMapper();
034    this.requestInterceptor = null;
035    this.encoder = new ClientEncoder(objectMapper);
036    this.decoder = new ClientDecoder(objectMapper);
037    this.errorDecoder = new ClientErrorDecoder();
038    this.contract = new ClientContract();
039    this.invocationHandlerFactory = new ClientInvocationHandlerFactory();
040  }
041
042  /**
043   * Read-write clients factory using simple user basic authentication.
044   */
045  public ClientFactory(String username, String password, String url) {
046    this.url = url;
047    ObjectMapper objectMapper = JacksonJsonObjectMapperProvider.getObjectMapper();
048    this.requestInterceptor = new SimpleUserAuthRequestInterceptor(username, password);
049    this.encoder = new ClientEncoder(objectMapper);
050    this.decoder = new ClientDecoder(objectMapper);
051    this.errorDecoder = new ClientErrorDecoder();
052    this.contract = new ClientContract();
053    this.invocationHandlerFactory = new ClientInvocationHandlerFactory();
054  }
055
056  /**
057   * Read-write client factory using GBIF authentication by application key.
058   */
059  public ClientFactory(String username, String url, String appKey, String secretKey) {
060    this.url = url;
061    ObjectMapper objectMapper = JacksonJsonObjectMapperProvider.getObjectMapper();
062    this.requestInterceptor =
063        new GbifAuthRequestInterceptor(username, appKey, secretKey, new SecretKeySigningService(),
064            new Md5EncodeServiceImpl(objectMapper));
065    this.encoder = new ClientEncoder(objectMapper);
066    this.decoder = new ClientDecoder(objectMapper);
067    this.errorDecoder = new ClientErrorDecoder();
068    this.contract = new ClientContract();
069    this.invocationHandlerFactory = new ClientInvocationHandlerFactory();
070  }
071
072  /**
073   * Read-write client factory using GBIF authentication by application key. Uses custom services.
074   */
075  public ClientFactory(String username, String url, String appKey, String secretKey,
076      SigningService signingService,
077      Md5EncodeService md5EncodeService, ObjectMapper objectMapper) {
078    this.url = url;
079    this.requestInterceptor =
080        new GbifAuthRequestInterceptor(username, appKey, secretKey, signingService, md5EncodeService);
081    this.encoder = new ClientEncoder(objectMapper);
082    this.decoder = new ClientDecoder(objectMapper);
083    this.errorDecoder = new ClientErrorDecoder();
084    this.contract = new ClientContract();
085    this.invocationHandlerFactory = new ClientInvocationHandlerFactory();
086  }
087
088  public <T> T newInstance(Class<T> clazz) {
089    Feign.Builder builder = Feign.builder()
090        .encoder(encoder)
091        .decoder(decoder)
092        .errorDecoder(errorDecoder)
093        .contract(contract)
094        .invocationHandlerFactory(invocationHandlerFactory);
095
096    if (requestInterceptor != null) {
097      builder.requestInterceptor(requestInterceptor);
098    }
099
100    return builder.target(clazz, url);
101  }
102}
