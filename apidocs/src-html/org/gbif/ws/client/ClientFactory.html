001package org.gbif.ws.client;
002
003import com.fasterxml.jackson.databind.ObjectMapper;
004import feign.Contract;
005import feign.Feign;
006import feign.InvocationHandlerFactory;
007import feign.codec.Decoder;
008import feign.codec.Encoder;
009import feign.codec.ErrorDecoder;
010import org.gbif.ws.json.JacksonJsonObjectMapperProvider;
011import org.gbif.ws.security.Md5EncodeService;
012import org.gbif.ws.security.Md5EncodeServiceImpl;
013import org.gbif.ws.security.SecretKeySigningService;
014import org.gbif.ws.security.SigningService;
015
016@SuppressWarnings("unused")
017public class ClientFactory {
018
019  private String url;
020  private GbifAuthRequestInterceptor requestInterceptor;
021  private Decoder decoder;
022  private Encoder encoder;
023  private ErrorDecoder errorDecoder;
024  private Contract contract;
025  private InvocationHandlerFactory invocationHandlerFactory;
026
027  public ClientFactory(String url) {
028    this.url = url;
029    ObjectMapper objectMapper = JacksonJsonObjectMapperProvider.getObjectMapper();
030    this.requestInterceptor = null;
031    this.encoder = new ClientEncoder(objectMapper);
032    this.decoder = new ClientDecoder(objectMapper);
033    this.errorDecoder = new ClientErrorDecoder();
034    this.contract = new ClientContract();
035    this.invocationHandlerFactory = new ClientInvocationHandlerFactory();
036  }
037
038  public ClientFactory(String username, String url, String appKey, String secretKey) {
039    this.url = url;
040    ObjectMapper objectMapper = JacksonJsonObjectMapperProvider.getObjectMapper();
041    this.requestInterceptor =
042        new GbifAuthRequestInterceptor(username, appKey, secretKey, new SecretKeySigningService(),
043            new Md5EncodeServiceImpl(objectMapper));
044    this.encoder = new ClientEncoder(objectMapper);
045    this.decoder = new ClientDecoder(objectMapper);
046    this.errorDecoder = new ClientErrorDecoder();
047    this.contract = new ClientContract();
048    this.invocationHandlerFactory = new ClientInvocationHandlerFactory();
049  }
050
051  public ClientFactory(String username, String url, String appKey, String secretKey,
052      SigningService signingService,
053      Md5EncodeService md5EncodeService, ObjectMapper objectMapper) {
054    this.url = url;
055    this.requestInterceptor =
056        new GbifAuthRequestInterceptor(username, appKey, secretKey, signingService, md5EncodeService);
057    this.encoder = new ClientEncoder(objectMapper);
058    this.decoder = new ClientDecoder(objectMapper);
059    this.errorDecoder = new ClientErrorDecoder();
060    this.contract = new ClientContract();
061    this.invocationHandlerFactory = new ClientInvocationHandlerFactory();
062  }
063
064  public <T> T newInstance(Class<T> clazz) {
065    Feign.Builder builder = Feign.builder()
066        .encoder(encoder)
067        .decoder(decoder)
068        .errorDecoder(errorDecoder)
069        .contract(contract)
070        .invocationHandlerFactory(invocationHandlerFactory);
071
072    if (requestInterceptor != null) {
073      builder.requestInterceptor(requestInterceptor);
074    }
075
076    return builder.target(clazz, url);
077  }
078}
