001package org.gbif.ws.client.filter;
002
003import org.gbif.api.model.common.User;
004import org.gbif.api.model.common.UserPrincipal;
005
006import java.security.Principal;
007
008import com.google.common.base.Strings;
009import com.google.inject.Provider;
010
011/**
012 * A Principal provider providing the same principal every time.
013 * The principal to be provided can be changed at any time.
014 *
015 * Useful for testing ws-clients.
016 */
017public class SimplePrincipalProvider implements Provider<Principal> {
018
019  private UserPrincipal current;
020
021  public void setPrincipal(String username) {
022    if (Strings.isNullOrEmpty(username)) {
023      current = null;
024    } else {
025      User user = new User();
026      user.setUserName(username);
027      current = new UserPrincipal(user);
028    }
029  }
030
031  @Override
032  public Principal get() {
033    return current;
034  }
035}
