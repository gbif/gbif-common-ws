040package org.gbif.ws.client.filter;
041
042import org.gbif.ws.security.GbifAppAuthService;
043
044import java.security.Principal;
045
046import com.google.inject.Provider;
047import com.sun.jersey.api.client.ClientHandlerException;
048import com.sun.jersey.api.client.ClientRequest;
049import com.sun.jersey.api.client.ClientResponse;
050import com.sun.jersey.api.client.filter.ClientFilter;
051
052/**
053 * Thread safe client filter adding an HTTP Authentication header to the HTTP request using the custom GBIF schema
054 * for trusted applications.
055 * In addition to the Authentication this filter will add these headers to the request:
056 * <ul>
057 * <li>Content-MD5: the MD5 hash for the request body</li>
058 * <li>x-gbif-user: the username of the proxied user</li>
059 * <li>x-gbif-date: the current datetime stamp</li>
060 * </ul>
061 * The custom date header is added in addition to the regular date header because it is not possible
062 * to access the generated date header in a jersey ClientFilter. The regular date header is generated by
063 * the RequestDate HttpRequestInterceptor of HttpClient and is therefore not available to jersey client.
064 * A current timestamp is important for signing the request though.
065 */
066public final class HttpGbifAuthFilter extends ClientFilter {
067
068  private final String applicationId;
069  private final Provider<Principal> principalProvider;
070  private final GbifAppAuthService authService;
071
072  /**
073   * Creates a new HTTP GBIF Authentication filter using provided application ID and trusted principal provider.
074   */
075  public HttpGbifAuthFilter(final String applicationId, GbifAppAuthService authService,
076    Provider<Principal> principalProvider) {
077    this.applicationId = applicationId;
078    this.principalProvider = principalProvider;
079    this.authService = authService;
080  }
081
082  @Override
083  public ClientResponse handle(final ClientRequest cr) throws ClientHandlerException {
084
085    if (principalProvider != null) {
086      Principal prince = principalProvider.get();
087      if (prince != null) {
088        authService.signRequest(applicationId, prince.getName(), cr);
089      }
090    }
091    return getNext().handle(cr);
092  }
093
094}
