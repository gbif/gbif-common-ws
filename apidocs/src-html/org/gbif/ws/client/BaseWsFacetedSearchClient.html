001package org.gbif.ws.client;
002
003
004import org.gbif.api.model.common.search.FacetedSearchRequest;
005import org.gbif.api.model.common.search.SearchParameter;
006import org.gbif.api.model.common.search.SearchResponse;
007import org.gbif.api.service.common.SearchService;
008
009import javax.annotation.Nullable;
010import javax.ws.rs.core.MultivaluedMap;
011
012import com.sun.jersey.api.client.GenericType;
013import com.sun.jersey.api.client.WebResource;
014
015import static org.gbif.ws.util.WebserviceParameter.PARAM_FACET;
016import static org.gbif.ws.util.WebserviceParameter.PARAM_FACET_MINCOUNT;
017import static org.gbif.ws.util.WebserviceParameter.PARAM_FACET_MULTISELECT;
018
019
020/**
021 * Base web service search client supporting a {@link FacetedSearchRequest}.
022 * 
023 * @param <T> type of the response content
024 * @param <P> search parameter type
025 * @param <R> search request type
026 */
027public abstract class BaseWsFacetedSearchClient<T, P extends Enum<?> & SearchParameter, R extends FacetedSearchRequest<P>>
028  extends BaseWsSearchClient<T, P, R> implements SearchService<T, P, R> {
029
030  protected BaseWsFacetedSearchClient(WebResource resource, GenericType<SearchResponse<T, P>> gType) {
031    super(resource, gType);
032  }
033
034  /**
035   * Adds support for faceted search request parameters.
036   */
037  @Override
038  protected MultivaluedMap<String, String> getParameterFromRequest(@Nullable R searchRequest) {
039    // The searchRequest is transformed in a parameter map
040    MultivaluedMap<String, String> parameters = super.getParameterFromRequest(searchRequest);
041
042    if (searchRequest != null) {
043      parameters.putSingle(PARAM_FACET_MULTISELECT, Boolean.toString(searchRequest.isMultiSelectFacets()));
044      if (searchRequest.getFacetMinCount() != null) {
045        parameters.putSingle(PARAM_FACET_MINCOUNT, Integer.toString(searchRequest.getFacetMinCount()));
046      }
047      if (searchRequest.getFacets() != null) {
048        for (P facet : searchRequest.getFacets()) {
049          parameters.add(PARAM_FACET, facet.name());
050        }
051      }
052    }
053
054    return parameters;
055  }
056}
