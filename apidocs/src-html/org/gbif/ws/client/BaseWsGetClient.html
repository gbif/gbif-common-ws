001package org.gbif.ws.client;
002
003import java.util.Locale;
004import javax.annotation.Nullable;
005import javax.ws.rs.core.MediaType;
006
007import com.sun.jersey.api.client.WebResource;
008import com.sun.jersey.api.client.filter.ClientFilter;
009
010/**
011 * Base client providing a get by key method for a specific entity type T.
012 * The client applies a json media type and provides various convenience methods to execute CRUD requests
013 * and can be configured to work with authentication.
014 * The client can return other types than the main T class via generic types passed in.
015 *
016 * @param <T> the main entity class
017 * @param <K> the key class of the main entity
018 */
019public abstract class BaseWsGetClient<T, K> extends BaseWsClient {
020
021  protected final Class<T> resourceClass;
022
023  /**
024   * @param resourceClass the main entity class
025   * @param resource      the base url to the underlying webservice
026   * @param authFilter    optional authentication filter, can be null
027   */
028  protected BaseWsGetClient(Class<T> resourceClass, WebResource resource, @Nullable ClientFilter authFilter) {
029    super(resource);
030    this.resourceClass = resourceClass;
031    if (authFilter != null) {
032      this.resource.addFilter(authFilter);
033    }
034  }
035
036  /**
037   * Gets a resource by its key by executing a json type http GET request.
038   * The clients resourceClass is used to convert the response into an object.
039   *
040   * @return requested resource or {@code null} if it couldn't be found
041   */
042  @Nullable
043  public T get(K key) {
044    if (key == null) {
045      throw new IllegalArgumentException("Key cannot be null");
046    }
047    return resource.path(key.toString()).type(MediaType.APPLICATION_JSON).get(resourceClass);
048  }
049
050  /**
051   * Executes a json type http GET request with paths added to base resource.
052   * The clients resourceClass is used to convert the response into an object.
053   *
054   * @param path optional paths to be appended to the base resource
055   *
056   * @return requested resource or null
057   */
058  @Nullable
059  protected T get(String... path) {
060    return getResource(path).type(MediaType.APPLICATION_JSON).get(resourceClass);
061  }
062
063  /**
064   * Gets a resource by its key by executing a json type http GET request.
065   * Also applies a language http header.
066   * The clients resourceClass is used to convert the response into an object.
067   *
068   * @param key    primary key for base resource class
069   * @param locale identifier for a region
070   *
071   * @return requested resource or null
072   */
073  @Nullable
074  protected T get(K key, Locale locale) {
075    if (key == null) {
076      throw new IllegalArgumentException("Key cannot be null");
077    }
078    return get(locale, key.toString());
079  }
080
081  /**
082   * Gets a resource by a path by executing a json type http GET request.
083   * Also applies a language http header.
084   * The clients resourceClass is used to convert the response into an object.
085   *
086   * @param locale identifier for a region.
087   * @param path   optional paths to be appended to the base resource.
088   *
089   * @return requested resource or null
090   */
091  @Nullable
092  protected T get(Locale locale, String... path) {
093    WebResource.Builder res = getResource(path).type(MediaType.APPLICATION_JSON);
094    if (locale != null) {
095      res = res.acceptLanguage(locale);
096    }
097    return res.get(resourceClass);
098  }
099
100}
