001package org.gbif.ws.client;
002
003import com.google.common.io.CharStreams;
004import feign.Response;
005import feign.codec.ErrorDecoder;
006import org.gbif.api.exception.ServiceUnavailableException;
007import org.gbif.ws.MethodNotAllowedException;
008import org.gbif.ws.NotFoundException;
009import org.slf4j.Logger;
010import org.slf4j.LoggerFactory;
011import org.springframework.security.access.AccessDeniedException;
012
013import javax.validation.ValidationException;
014import java.io.IOException;
015import java.io.Reader;
016import java.net.URI;
017
018@SuppressWarnings("UnstableApiUsage")
019public class ClientErrorDecoder implements ErrorDecoder {
020
021  private static final Logger LOG = LoggerFactory.getLogger(ClientErrorDecoder.class);
022
023  @Override
024  public Exception decode(String methodKey, Response response) {
025    String message = null;
026
027    if (response.body() != null) {
028      try (Reader reader = response.body().asReader()) {
029        //Easy way to read the stream and get a String object
030        message = CharStreams.toString(reader);
031        LOG.error("Client exception: {}", message);
032      } catch (IOException e) {
033        LOG.error("Exception during reading client error response", e);
034      }
035    }
036
037    switch (response.status()) {
038      case 400:
039        return new IllegalArgumentException("A bad request received");
040      case 401:
041        return new AccessDeniedException("Unauthorized request received");
042      case 403:
043        return new AccessDeniedException("Forbidden request received");
044      case 404:
045        return new NotFoundException("Resource not found", URI.create(response.request().url()));
046      case 405:
047        return new MethodNotAllowedException("Method not allowed to user");
048      case 422:
049        return message != null
050            ? new ValidationException(extractValidationErrorMessage(message))
051            : new ValidationException();
052      case 500:
053        return new ServiceUnavailableException("An internal server error occurred, please try again later");
054      case 501:
055        return new UnsupportedOperationException(message != null ? message : "Method not implement yet");
056      default:
057        return new RuntimeException(message != null ? message : "Unexpected exception");
058    }
059  }
060
061  private String extractValidationErrorMessage(String message) {
062    return message.replace("<ul><li>", "")
063        .replace("</li></ul>", "")
064        .replace("<li></li>", ", ");
065  }
066}
