016package org.gbif.ws.client;
017
018import org.gbif.api.exception.ServiceUnavailableException;
019import org.gbif.ws.MethodNotAllowedException;
020import org.gbif.ws.NotFoundException;
021
022import java.io.IOException;
023import java.io.Reader;
024import java.net.URI;
025
026import javax.validation.ValidationException;
027
028import org.slf4j.Logger;
029import org.slf4j.LoggerFactory;
030import org.springframework.security.access.AccessDeniedException;
031
032import com.google.common.io.CharStreams;
033
034import feign.Response;
035import feign.codec.ErrorDecoder;
036
037@SuppressWarnings("UnstableApiUsage")
038public class ClientErrorDecoder implements ErrorDecoder {
039
040  private static final Logger LOG = LoggerFactory.getLogger(ClientErrorDecoder.class);
041
042  @Override
043  public Exception decode(String methodKey, Response response) {
044    String message = null;
045
046    if (response.body() != null) {
047      try (Reader reader = response.body().asReader()) {
048        // Easy way to read the stream and get a String object
049        message = CharStreams.toString(reader);
050        LOG.error("Client exception: {}", message);
051      } catch (IOException e) {
052        LOG.error("Exception during reading client error response", e);
053      }
054    }
055
056    switch (response.status()) {
057      case 400:
058        return new IllegalArgumentException("A bad request received");
059      case 401:
060        return new AccessDeniedException("Unauthorized request received");
061      case 403:
062        return new AccessDeniedException("Forbidden request received");
063      case 404:
064        return new NotFoundException("Resource not found", URI.create(response.request().url()));
065      case 405:
066        return new MethodNotAllowedException("Method not allowed to user");
067      case 422:
068        return message != null
069            ? new ValidationException(extractValidationErrorMessage(message))
070            : new ValidationException();
071      case 500:
072        return new ServiceUnavailableException(
073            "An internal server error occurred, please try again later");
074      case 501:
075        return new UnsupportedOperationException(
076            message != null ? message : "Method not implement yet");
077      default:
078        return new RuntimeException(message != null ? message : "Unexpected exception");
079    }
080  }
081
082  private String extractValidationErrorMessage(String message) {
083    return message.replace("<ul><li>", "").replace("</li></ul>", "").replace("<li></li>", ", ");
084  }
085}
