001package org.gbif.ws.client;
002
003import com.google.common.io.CharStreams;
004import feign.Response;
005import feign.codec.ErrorDecoder;
006import org.gbif.api.exception.ServiceUnavailableException;
007import org.gbif.ws.NotFoundException;
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010import org.springframework.security.access.AccessDeniedException;
011
012import javax.validation.ValidationException;
013import java.io.IOException;
014import java.io.Reader;
015import java.net.URI;
016
017@SuppressWarnings("UnstableApiUsage")
018public class ClientErrorDecoder implements ErrorDecoder {
019
020  private static final Logger LOG = LoggerFactory.getLogger(ClientErrorDecoder.class);
021
022  @Override
023  public Exception decode(String methodKey, Response response) {
024    String message = null;
025
026    if (response.body() != null) {
027      try (Reader reader = response.body().asReader()) {
028        //Easy way to read the stream and get a String object
029        message = CharStreams.toString(reader);
030        LOG.error("Client exception: {}", message);
031      } catch (IOException e) {
032        LOG.error("Exception during reading client error response", e);
033      }
034    }
035
036    switch (response.status()) {
037      case 400:
038        return new IllegalArgumentException("A bad request received");
039      case 401:
040        return new AccessDeniedException("Unauthorized request received");
041      case 403:
042        return new AccessDeniedException("Forbidden request received");
043      case 404:
044        return new NotFoundException("Resource not found", URI.create(response.request().url()));
045      case 422:
046        return message != null
047            ? new ValidationException(extractValidationErrorMessage(message))
048            : new ValidationException();
049      case 500:
050        return new ServiceUnavailableException("An internal server error occurred, please try again later");
051      case 501:
052        return new UnsupportedOperationException(message != null ? message : "Method not implement yet");
053      default:
054        return new RuntimeException(message != null ? message : "Unexpected exception");
055    }
056  }
057
058  private String extractValidationErrorMessage(String message) {
059    return message.replace("<ul><li>", "")
060        .replace("</li></ul>", "")
061        .replace("<li></li>", ", ");
062  }
063}
