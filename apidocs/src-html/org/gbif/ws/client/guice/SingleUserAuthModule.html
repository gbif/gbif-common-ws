001package org.gbif.ws.client.guice;
002
003import com.google.inject.AbstractModule;
004import com.sun.jersey.api.client.filter.ClientFilter;
005import com.sun.jersey.api.client.filter.HTTPBasicAuthFilter;
006
007/**
008 * A small module that installs a {@link com.sun.jersey.api.client.filter.HTTPBasicAuthFilter}
009 * for a single, fixed user.
010 *
011 * If application need to write to webservices, e.g the registry, you need to make sure the user
012 * has admin rights in drupal.
013 *
014 * @see <a href="http://staging.gbif.org/drupal/user/register">register a new user in drupal</a>
015 */
016public class SingleUserAuthModule extends AbstractModule {
017
018  private final HTTPBasicAuthFilter filter;
019
020  public SingleUserAuthModule(String user, String password) {
021    filter = new HTTPBasicAuthFilter(user, password);
022  }
023
024  @Override
025  protected void configure() {
026    bind(ClientFilter.class).toInstance(filter);
027  }
028}
