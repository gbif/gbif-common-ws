030/**
031 * Base web service client module.
032 * Allows to specify HTTP connection parameters: httpTimeout, maxHttpConnections and maxHttpConnectionsPerRoute.
033 * The default values for those parameters are: httpTimeout = 10 seconds, maxHttpConnections = 100 and
034 * maxHttpConnectionsPerRoute = 100.
035 */
036public abstract class GbifWsClientModule extends PrivateModule {
037
038
039  /**
040   * Default names/keys to create an HTTP client connection.
041   * The values listed in this class are the expected key in a properties file.
042   */
043  public static class HttpClientConnParams {
044
045    // Used for each of i) getting a connection from the pool, establishing a connection and the
046    // socket timeout
047    public static final String HTTP_TIMEOUT = "httpTimeout";
048    public static final String MAX_HTTP_CONNECTIONS = "maxHttpConnections";
049    public static final String MAX_HTTP_CONNECTIONS_PER_ROUTE = "maxHttpConnectionsPerRoute";
050  }
051
052  private final Properties properties;
053  private final Set<Package> clientPackages;
054
055  // Default values to establish a client connection
056  protected static final int DEFAULT_HTTP_TIMEOUT_MSECS = 10000;
057  protected static final int DEFAULT_MAX_HTTP_CONNECTIONS = 100;
058  protected static final int DEFAULT_MAX_HTTP_CONNECTIONS_PER_ROUTE = 100;
059
060  protected GbifWsClientModule(Properties properties, Package... clientPackages) {
061    this.properties = properties;
062    this.clientPackages = Sets.newHashSet(clientPackages);
063    this.clientPackages.add(BaseWsGetClient.class.getPackage());
064  }
065
066  @Override
067  protected final void configure() {
068    Names.bindProperties(binder(), properties);
069
070    // configure jsonMixIns
071    JacksonJsonContextResolver.addMixIns(getMixIns());
072
073    configureClient();
074
075    // bind common interceptors
076    for (Package clientPackage : clientPackages) {
077      // order is important!
078      bindInterceptor(Matchers.inPackage(clientPackage), new PublicMethodMatcher(), new HttpErrorResponseInterceptor());
079    }
080  }
081
082  /**
083   * Implement this method to configure the clients guice module.
084   * You can install modules, bind classes.
085   * Dont forget to expose any classes that should be visible to the outside world!
086   */
087  protected abstract void configureClient();
088
089
090  /**
091   * Override this method to pass a map of mixIn classes into the Jackson context resolver.
092   * For example, the Jackson JSON configuration may need to know about how to (de)serialize polymorphic classes.
093   *
094   * @return the mixIn class map. Defaults to an empty map.
095   */
096  protected Map<Class<?>, Class<?>> getMixIns() {
097    return Maps.newHashMap();
098  }
099
100  protected Properties getProperties() {
101    return properties;
102  }
103
104  @Provides
105  @Singleton
106  public HttpClient provideHttpClient() {
107    int httpTimeout = DEFAULT_HTTP_TIMEOUT_MSECS;
108    int maxHttpConnections = DEFAULT_MAX_HTTP_CONNECTIONS;
109    int maxHttpConnectionsPerRoute = DEFAULT_MAX_HTTP_CONNECTIONS_PER_ROUTE;
110
111    if (properties.containsKey(HttpClientConnParams.HTTP_TIMEOUT)) {
112      httpTimeout = Integer.parseInt(properties.getProperty(HttpClientConnParams.HTTP_TIMEOUT));
113    }
114    if (properties.containsKey(HttpClientConnParams.MAX_HTTP_CONNECTIONS)) {
115      maxHttpConnections = Integer.parseInt(properties.getProperty(HttpClientConnParams.MAX_HTTP_CONNECTIONS));
116    }
117    if (properties.containsKey(HttpClientConnParams.MAX_HTTP_CONNECTIONS_PER_ROUTE)) {
118      maxHttpConnectionsPerRoute =
119        Integer.parseInt(properties.getProperty(HttpClientConnParams.MAX_HTTP_CONNECTIONS_PER_ROUTE));
120    }
121    return HttpUtil.newMultithreadedClient(httpTimeout, maxHttpConnections, maxHttpConnectionsPerRoute);
122  }
123
124  @Provides
125  @Singleton
126  @Inject
127  public Client providesJerseyClient(HttpClient client) {
128    return buildJerseyClient(client);
129  }
130
131  public static Client buildJerseyClient(HttpClient client) {
132    ApacheHttpClient4Handler hch = new ApacheHttpClient4Handler(client, null, false);
133
134    ClientConfig clientConfig = new DefaultClientConfig();
135    clientConfig.getClasses().add(JacksonJsonContextResolver.class);
136    // this line is critical! Note that this is the jersey version of this class name!
137    clientConfig.getClasses().add(JacksonJsonProvider.class);
138    clientConfig.getFeatures().put(JSONConfiguration.FEATURE_POJO_MAPPING, Boolean.TRUE);
139    return new ApacheHttpClient4(hch, clientConfig);
140  }
