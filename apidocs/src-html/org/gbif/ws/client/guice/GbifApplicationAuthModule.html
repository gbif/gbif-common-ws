001package org.gbif.ws.client.guice;
002
003import org.gbif.ws.client.filter.HttpGbifAuthFilter;
004import org.gbif.ws.client.filter.SimplePrincipalProvider;
005import org.gbif.ws.security.GbifAppAuthService;
006
007import java.util.Properties;
008
009import com.google.inject.AbstractModule;
010import com.sun.jersey.api.client.filter.ClientFilter;
011
012/**
013 * An authentication module for trusted GBIF applications.
014 * It requires an application key and can then act on behalf of any username.
015 *
016 * To set the proxied username you can either set it directly on the module instance
017 * or inject the SimplePrincipalProvider into your code anywhere to set the current user:
018 *
019 * <pre>
020 * {@code
021 * GbifApplicationAuthModule authMod = new GbifApplicationAuthModule();
022 * authMod.setPrincipal("admin");
023 *
024 * // set user via SimplePrincipalProvider:
025 *
026 * {@literal @}Inject
027 * SimplePrincipalProvider pp;
028 * pp.setPrincipal("admin");
029 * }
030 * </pre>
031 *
032 * Note that the users role still depends on the roles defined in the user account in Drupal and will not default
033 * to ADMIN. The user also has to exist in Drupal for the webservice authorization to work.
034 *
035 */
036// TODO: Make the application key configurable (Requires significant changes in the webservice AuthModule)
037public class GbifApplicationAuthModule extends AbstractModule {
038
039  public static final String PROPERTY_APP_KEY = "application.key";
040  public static final String PROPERTY_APP_SECRET = "application.secret";
041
042  private final SimplePrincipalProvider pp = new SimplePrincipalProvider();
043  private final GbifAppAuthService authService;
044  private final String applicationID;
045
046  /**
047   * Creates a new authentication guice module for a trusted application with its id and secret key.
048   * The proxied user will be set to match the application ID.
049   *
050   * @param appId        the trusted application id
051   * @param appSecretKey the secret key issued for this application
052   */
053  public GbifApplicationAuthModule(String appId, String appSecretKey) {
054    applicationID = appId;
055    authService = new GbifAppAuthService(appId, appSecretKey);
056    pp.setPrincipal(appId);
057  }
058
059  /**
060   * Creates a new authentication guice module for a trusted application with its id and secret key given as
061   * properties.
062   * The proxied user will be set to match the application.key.
063   *
064   * @param properties that holds an entry for application.key and application.secret
065   */
066  public GbifApplicationAuthModule(Properties properties) {
067    this(properties.getProperty(PROPERTY_APP_KEY), properties.getProperty(PROPERTY_APP_SECRET));
068  }
069
070  @Override
071  protected void configure() {
072    bind(SimplePrincipalProvider.class).toInstance(pp);
073    bind(GbifAppAuthService.class).toInstance(authService);
074    HttpGbifAuthFilter authFilter = new HttpGbifAuthFilter(applicationID, authService, pp);
075    bind(ClientFilter.class).toInstance(authFilter);
076  }
077
078  public void setPrincipal(String username) {
079    pp.setPrincipal(username);
080  }
081}
