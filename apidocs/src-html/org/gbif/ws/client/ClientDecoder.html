001package org.gbif.ws.client;
002
003import com.fasterxml.jackson.databind.ObjectMapper;
004import feign.FeignException;
005import feign.Response;
006import feign.Util;
007import feign.codec.DecodeException;
008import feign.codec.Decoder;
009import feign.jackson.JacksonDecoder;
010import org.springframework.http.HttpHeaders;
011import org.springframework.http.HttpStatus;
012import org.springframework.http.MediaType;
013import org.springframework.util.StringUtils;
014
015import java.io.IOException;
016import java.lang.reflect.Type;
017
018public class ClientDecoder implements Decoder {
019
020  private final JacksonDecoder jacksonDecoder;
021
022  public ClientDecoder(ObjectMapper objectMapper) {
023    this.jacksonDecoder = new JacksonDecoder(objectMapper);
024  }
025
026  @Override
027  public Object decode(Response response, Type type)
028      throws IOException, FeignException {
029    HttpStatus responseStatus = HttpStatus.resolve(response.status());
030
031    if (HttpStatus.NOT_FOUND == responseStatus) {
032      return null;
033    } else if (responseStatus != null && responseStatus.isError()) {
034      throw new DecodeException(response.status(), response.toString(), response.request());
035    }
036
037    MediaType contentType = getContentType(response);
038    if (MediaType.APPLICATION_JSON.equalsTypeAndSubtype(contentType)) {
039      return jacksonDecoder.decode(response, type);
040    } else if (MediaType.TEXT_PLAIN.equalsTypeAndSubtype(contentType)) {
041      return Util.toString(response.body().asReader());
042    } else if (MediaType.APPLICATION_OCTET_STREAM.equalsTypeAndSubtype(contentType)) {
043      return Util.toByteArray(response.body().asInputStream());
044    } else if (byte[].class.equals(type)) {
045      return Util.toByteArray(response.body().asInputStream());
046    } else {
047      throw new DecodeException(response.status(), "Unsupported response type", response.request());
048    }
049  }
050
051  /**
052   * Gets the first MediaType listed in the Content-Type header.
053   */
054  private static MediaType getContentType(Response response) {
055    return response.headers().get(HttpHeaders.CONTENT_TYPE).stream()
056        .findFirst()
057        .filter(StringUtils::hasLength)
058        .map(MediaType::parseMediaType).orElse(null);
059  }
060}
