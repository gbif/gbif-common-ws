001package org.gbif.ws.client;
002
003import org.gbif.api.vocabulary.Country;
004import org.gbif.api.vocabulary.Language;
005
006import java.util.Map;
007import javax.ws.rs.core.MultivaluedMap;
008
009import com.google.common.base.Strings;
010import com.sun.jersey.core.util.MultivaluedMapImpl;
011
012/**
013 * A simple builder class to generate query parameter maps.
014 * It converts enum values to their name string, but uses ISO2 letter codes for the country and language enum.
015 */
016public class QueryParamBuilder {
017
018  private final MultivaluedMap<String, String> params = new MultivaluedMapImpl();
019
020  /**
021   * Creates a new, empty query parameter builder.
022   */
023  public static QueryParamBuilder create() {
024    return new QueryParamBuilder();
025  }
026
027  /**
028   * Creates a query parameter builder from a list of key value pairs.
029   *
030   * @param kvp list of key value pairs with key first. For example K1,V1,K2,V2.
031   *
032   * @throws IllegalArgumentException if kvp list is not pairs with an even number
033   */
034  public static QueryParamBuilder create(Object... kvp) {
035    if (kvp.length % 2 != 0) {
036      throw new IllegalArgumentException("Equal number of kvp parameter required to build key value pairs.");
037    }
038
039    QueryParamBuilder builder = new QueryParamBuilder();
040    for (int x = kvp.length - 1; x > 0; x -= 2) {
041      Object k = kvp[x - 1];
042      Object v = kvp[x];
043      builder.queryParam(k, v);
044    }
045    return builder;
046  }
047
048  /**
049   * Creates a query parameter builder with a list of key value pairs having the same key.
050   *
051   * @param key    parameter key
052   * @param values list of values to be used to construct single key value pairs
053   *
054   * @throws IllegalArgumentException if kvp list is not pairs with an even number
055   */
056  public static QueryParamBuilder create(String key, Object[] values) {
057    return new QueryParamBuilder().queryParam(key, values);
058  }
059
060  /**
061   * Creates a query parameter builder from an existing parameter map.
062   *
063   * @param params map of query parameters.
064   */
065  public static QueryParamBuilder create(Map<String, String> params) {
066    QueryParamBuilder builder = new QueryParamBuilder();
067    for (Map.Entry<String, String> kvp : params.entrySet()) {
068      builder.queryParam(kvp.getKey(), kvp.getValue());
069    }
070    return builder;
071  }
072
073  /**
074   * Adds an additional query parameter the parameter map.
075   *
076   * @param key   the query parameter name
077   * @param value the query parameter value or a list if the parameter should be added several times
078   *
079   * @return the current builder to chain calls.
080   */
081  public QueryParamBuilder queryParam(Object key, Object... value) {
082    if (key != null && value != null) {
083      String k = key.toString();
084      if (!Strings.isNullOrEmpty(k)) {
085        for (Object val : value) {
086          if (val != null) {
087            // use iso codes for country and language enums, otherwise just toString
088            String v;
089            if (val instanceof Country){
090              v = ((Country) val).getIso2LetterCode();
091            } else if (val instanceof Language){
092              v = ((Language) val).getIso2LetterCode();
093            } else {
094              v = val.toString();
095            }
096            if (!Strings.isNullOrEmpty(v)) {
097              params.add(k, v);
098            }
099          }
100        }
101      }
102    }
103    return this;
104  }
105
106  public MultivaluedMap<String, String> build() {
107    return params;
108  }
109}
