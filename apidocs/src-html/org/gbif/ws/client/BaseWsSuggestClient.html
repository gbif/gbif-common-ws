001package org.gbif.ws.client;
002
003import org.gbif.api.model.common.search.FacetedSearchRequest;
004import org.gbif.api.model.common.search.SearchParameter;
005import org.gbif.api.model.common.search.SearchRequest;
006import org.gbif.api.model.common.search.SearchResponse;
007import org.gbif.api.service.common.SuggestService;
008
009import java.util.List;
010
011import com.sun.jersey.api.client.GenericType;
012import com.sun.jersey.api.client.WebResource;
013
014
015/**
016 * Base web service suggest & faceted search client.
017 */
018public abstract class BaseWsSuggestClient<T, P extends Enum<?> & SearchParameter, R extends FacetedSearchRequest<P>, ST, RSUG extends SearchRequest<P>>
019  extends BaseWsFacetedSearchClient<T, P, R> implements SuggestService<ST, P, RSUG> {
020
021  private final GenericType<List<ST>> suggestType;
022
023  /**
024   * @param resource the base web resource being the parent of /search and /suggest
025   * @param searchType the generic type for the search response.
026   * @param suggestType the generic type for the suggest response.
027   */
028  protected BaseWsSuggestClient(WebResource resource, GenericType<SearchResponse<T, P>> searchType,
029    GenericType<List<ST>> suggestType) {
030    super(resource, searchType);
031    this.suggestType = suggestType;
032  }
033
034  /**
035   * Generic suggest operation retrieving search entities T.
036   * 
037   * @param suggestRequest to issue the search/suggest operation.
038   * @return a List of NameUsageSearchSuggestResult objects
039   */
040  public List<ST> suggest(RSUG suggestRequest) {
041    if (suggestRequest == null) {
042      throw new IllegalArgumentException("searchSuggestRequest cannot be null");
043    }
044    return get(suggestType, null, getParameterFromSearchRequest(suggestRequest), suggestRequest, "suggest");
045  }
046
047}
