016package org.gbif.ws.client;
017
018import java.lang.reflect.Method;
019import java.lang.reflect.Modifier;
020import java.util.ArrayList;
021import java.util.Arrays;
022import java.util.LinkedHashMap;
023import java.util.List;
024import java.util.Map;
025
026import org.apache.commons.lang3.StringUtils;
027import org.springframework.cloud.openfeign.annotation.PathVariableParameterProcessor;
028import org.springframework.cloud.openfeign.annotation.QueryMapParameterProcessor;
029import org.springframework.cloud.openfeign.annotation.RequestHeaderParameterProcessor;
030import org.springframework.cloud.openfeign.annotation.RequestParamParameterProcessor;
031import org.springframework.cloud.openfeign.support.SpringMvcContract;
032import org.springframework.web.bind.annotation.RequestMapping;
033
034import feign.MethodMetadata;
035import feign.Util;
036
037import static feign.Util.checkState;
038import static feign.Util.emptyToNull;
039import static org.springframework.core.annotation.AnnotatedElementUtils.findMergedAnnotation;
040
041public class ClientContract extends SpringMvcContract {
042
043  public ClientContract() {
044    super(
045        Arrays.asList(
046            new PartialDateParameterProcessor(),
047            new PathVariableParameterProcessor(),
048            new RequestParamParameterProcessor(),
049            new RequestHeaderParameterProcessor(),
050            new QueryMapParameterProcessor()));
051  }
052
053  @Override
054  public List<MethodMetadata> parseAndValidatateMetadata(final Class<?> targetType) {
055    checkState(
056        targetType.getTypeParameters().length == 0,
057        "Parameterized types unsupported: %s",
058        targetType.getSimpleName());
059    final Map<String, MethodMetadata> result = new LinkedHashMap<>();
060
061    for (final Method method : targetType.getMethods()) {
062      if (method.getDeclaringClass() == Object.class
063          || (method.getModifiers() & Modifier.STATIC) != 0
064          || Util.isDefault(method)
065          // skip default methods which related to generic inheritance
066          // also default methods are considered as "unsupported operations"
067          || method.toString().startsWith("public default")
068          // skip not annotated methods (consider as "not implemented")
069          || method.getAnnotations().length == 0) {
070        continue;
071      }
072      final MethodMetadata metadata = this.parseAndValidateMetadata(targetType, method);
073      checkState(
074          !result.containsKey(metadata.configKey()),
075          "Overrides unsupported: %s",
076          metadata.configKey());
077      result.put(metadata.configKey(), metadata);
078    }
079
080    return new ArrayList<>(result.values());
081  }
082
083  @Override
084  protected void processAnnotationOnClass(MethodMetadata data, Class<?> clz) {
085    RequestMapping classAnnotation = findMergedAnnotation(clz, RequestMapping.class);
086    if (classAnnotation != null) {
087      // Prepend path from class annotation if specified
088      if (classAnnotation.value().length > 0) {
089        String pathValue = emptyToNull(classAnnotation.value()[0]);
090        data.template().uri(StringUtils.prependIfMissing(pathValue, "/"));
091      }
092    }
093  }
094}
