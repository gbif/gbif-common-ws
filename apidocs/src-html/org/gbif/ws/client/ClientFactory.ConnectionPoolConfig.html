001package org.gbif.ws.client;
002
003import com.fasterxml.jackson.databind.ObjectMapper;
004import feign.Contract;
005import feign.Feign;
006import feign.InvocationHandlerFactory;
007import feign.RequestInterceptor;
008import feign.codec.Decoder;
009import feign.codec.Encoder;
010import feign.codec.ErrorDecoder;
011import feign.httpclient.ApacheHttpClient;
012import lombok.Builder;
013import lombok.Data;
014import org.apache.http.client.HttpClient;
015import org.apache.http.client.config.RequestConfig;
016import org.apache.http.config.ConnectionConfig;
017import org.apache.http.config.SocketConfig;
018import org.apache.http.impl.client.HttpClients;
019
020import org.gbif.ws.json.JacksonJsonObjectMapperProvider;
021import org.gbif.ws.security.Md5EncodeService;
022import org.gbif.ws.security.Md5EncodeServiceImpl;
023import org.gbif.ws.security.SecretKeySigningService;
024import org.gbif.ws.security.SigningService;
025
026import java.nio.charset.Charset;
027import java.nio.charset.StandardCharsets;
028
029@SuppressWarnings("unused")
030public class ClientFactory {
031
032  private static final String HTTP_PROTOCOL = "http";
033  private static final String HTTPS_PROTOCOL = "https";
034
035  private String url;
036  private RequestInterceptor requestInterceptor;
037  private Decoder decoder;
038  private Encoder encoder;
039  private ErrorDecoder errorDecoder;
040  private Contract contract;
041  private InvocationHandlerFactory invocationHandlerFactory;
042  private ConnectionPoolConfig connectionPoolConfig;
043
044  /**
045   * Read-only clients factory.
046   */
047  public ClientFactory(String url) {
048    this.url = url;
049    ObjectMapper objectMapper = JacksonJsonObjectMapperProvider.getObjectMapper();
050    this.requestInterceptor = null;
051    this.encoder = new ClientEncoder(objectMapper);
052    this.decoder = new ClientDecoder(objectMapper);
053    this.errorDecoder = new ClientErrorDecoder();
054    this.contract = new ClientContract();
055    this.invocationHandlerFactory = new ClientInvocationHandlerFactory();
056  }
057
058  /**
059   * Read-write clients factory using simple user basic authentication.
060   */
061  public ClientFactory(String username, String password, String url) {
062    this.url = url;
063    ObjectMapper objectMapper = JacksonJsonObjectMapperProvider.getObjectMapper();
064    this.requestInterceptor = new SimpleUserAuthRequestInterceptor(username, password);
065    this.encoder = new ClientEncoder(objectMapper);
066    this.decoder = new ClientDecoder(objectMapper);
067    this.errorDecoder = new ClientErrorDecoder();
068    this.contract = new ClientContract();
069    this.invocationHandlerFactory = new ClientInvocationHandlerFactory();
070  }
071
072  /**
073   * Read-write client factory using GBIF authentication by application key.
074   */
075  public ClientFactory(String username, String url, String appKey, String secretKey) {
076    this.url = url;
077    ObjectMapper objectMapper = JacksonJsonObjectMapperProvider.getObjectMapper();
078    this.requestInterceptor =
079        new GbifAuthRequestInterceptor(username, appKey, secretKey, new SecretKeySigningService(),
080            new Md5EncodeServiceImpl(objectMapper));
081    this.encoder = new ClientEncoder(objectMapper);
082    this.decoder = new ClientDecoder(objectMapper);
083    this.errorDecoder = new ClientErrorDecoder();
084    this.contract = new ClientContract();
085    this.invocationHandlerFactory = new ClientInvocationHandlerFactory();
086  }
087
088  /**
089   * Read-write client factory using GBIF authentication by application key. Uses custom services.
090   */
091  public ClientFactory(String username, String url, String appKey, String secretKey,
092      SigningService signingService,
093      Md5EncodeService md5EncodeService, ObjectMapper objectMapper) {
094    this.url = url;
095    this.requestInterceptor =
096        new GbifAuthRequestInterceptor(username, appKey, secretKey, signingService, md5EncodeService);
097    this.encoder = new ClientEncoder(objectMapper);
098    this.decoder = new ClientDecoder(objectMapper);
099    this.errorDecoder = new ClientErrorDecoder();
100    this.contract = new ClientContract();
101    this.invocationHandlerFactory = new ClientInvocationHandlerFactory();
102  }
103
104  private static HttpClient newMultithreadedClient(ConnectionPoolConfig connectionPoolConfig) {
105
106    return HttpClients.custom()
107            .setMaxConnTotal(connectionPoolConfig.getMaxConnections())
108            .setMaxConnPerRoute(connectionPoolConfig.getMaxPerRoute())
109            .setDefaultSocketConfig(SocketConfig.custom().setSoTimeout(connectionPoolConfig.getTimeout()).build())
110            .setDefaultConnectionConfig(ConnectionConfig.custom().setCharset(Charset.forName(StandardCharsets.UTF_8.name())).build())
111            .setDefaultRequestConfig(RequestConfig.custom().setConnectTimeout(connectionPoolConfig.getTimeout())
112                                       .setConnectionRequestTimeout(connectionPoolConfig.getTimeout()).build())
113            .build();
114  }
115
116  public <T> T newInstance(Class<T> clazz) {
117    return newInstance(clazz, null);
118  }
119
120  public <T> T newInstance(Class<T> clazz, ConnectionPoolConfig connectionPoolConfig) {
121    Feign.Builder builder = Feign.builder()
122      .encoder(encoder)
123      .decoder(decoder)
124      .errorDecoder(errorDecoder)
125      .contract(contract)
126      .decode404()
127      .invocationHandlerFactory(invocationHandlerFactory);
128
129    if (requestInterceptor != null) {
130      builder.requestInterceptor(requestInterceptor);
131    }
132    if (connectionPoolConfig != null) {
133      builder.client(new ApacheHttpClient(newMultithreadedClient(connectionPoolConfig)));
134    }
135    return builder.target(clazz, url);
136  }
137
138  @Data
139  @Builder
140  public static class ConnectionPoolConfig {
141    private final Integer timeout;
142    private final Integer maxConnections;
143    private final Integer maxPerRoute;
144  }
145
146}
