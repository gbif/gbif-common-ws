001package org.gbif.ws.client.interceptor;
002
003import org.gbif.api.exception.ServiceUnavailableException;
004
005import java.security.AccessControlException;
006
007import javax.validation.ValidationException;
008
009import com.sun.jersey.api.client.ClientResponse;
010import com.sun.jersey.api.client.UniformInterfaceException;
011import org.aopalliance.intercept.MethodInterceptor;
012import org.aopalliance.intercept.MethodInvocation;
013
014/**
015 * A client method interceptor that translates http response errors into actual java exceptions.
016 *
017 * @see <a href="http://dev.gbif.org/wiki/display/POR/Service+Exceptions">GBIF Dev wiki</a> for exception mappings.
018 *      400 => IllegalArgumentException,
019 *      401 => AccessControlException,
020 *      403 => AccessControlException,
021 *      404 => NULL,
022 *      422 => ValidationException,
023 *      500 => ServiceUnavailableException,
024 *      501 => UnsupportedOperationException,
025 *      All other types of unexpected responses the exception passes through the interceptor unchanged.
026 */
027public class HttpErrorResponseInterceptor implements MethodInterceptor {
028
029  /**
030   * @return a short description of the http status followed by the entire http response body
031   */
032  private static String readBody(UniformInterfaceException e) {
033    String status = "HTTP " + e.getResponse().getStatus() + ": ";
034    try {
035      return status + e.getResponse().getEntity(String.class);
036    } catch (Exception e1) {
037      return status + "Failed to read http body";
038    }
039  }
040
041  @Override
042  public Object invoke(MethodInvocation invocation) throws Throwable {
043    Object result = null;
044    // exceptions can surface as the result or be thrown - catch both in this var:
045    UniformInterfaceException e = null;
046    try {
047      result = invocation.proceed();
048    } catch (UniformInterfaceException e2) {
049      e = e2;
050    }
051
052    if (result instanceof UniformInterfaceException) {
053      e = (UniformInterfaceException) result;
054    }
055
056    // now check the UniformInterfaceException if it exists
057    if (e != null) {
058      ClientResponse response = e.getResponse();
059      UniformInterfaceException eX = new UniformInterfaceException(readBody(e), e.getResponse());
060
061      switch (response.getStatus()) {
062        case 204:
063          // no content, return null;
064          return null;
065        case 400:
066          throw new IllegalArgumentException("A bad request received.", eX);
067        case 401:
068          throw new AccessControlException("Unauthorized request received.");
069        case 403:
070          throw new AccessControlException("Forbidden request received.");
071        case 404:
072          return null;
073        case 422:
074          throw new ValidationException(eX); // TODO: consider passing fields in headers
075        case 500:
076          throw new ServiceUnavailableException("An internal server error occurred, please try again later.", eX);
077        case 501:
078          throw new UnsupportedOperationException("Method not implement yet.", eX);
079        default:
080          throw eX;
081      }
082    }
083
084    return result;
085  }
086}
