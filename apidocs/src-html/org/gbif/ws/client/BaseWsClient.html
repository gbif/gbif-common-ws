001package org.gbif.ws.client;
002
003import org.gbif.api.model.common.paging.Pageable;
004import org.gbif.ws.json.JacksonJsonContextResolver;
005
006import java.io.IOException;
007import java.util.Locale;
008import javax.annotation.Nullable;
009import javax.ws.rs.core.MediaType;
010import javax.ws.rs.core.MultivaluedMap;
011
012import com.google.common.base.Joiner;
013import com.sun.jersey.api.client.GenericType;
014import com.sun.jersey.api.client.WebResource;
015import org.codehaus.jackson.map.ObjectMapper;
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018
019import static org.gbif.api.model.common.paging.PagingConstants.PARAM_LIMIT;
020import static org.gbif.api.model.common.paging.PagingConstants.PARAM_OFFSET;
021
022/**
023 * The base webservice client for all GBIF clients not tight to a specific entity class but containing various
024 * convenience methods to deal with a web resource, paging and parameters.
025 */
026public abstract class BaseWsClient {
027
028  private static final Joiner PATH_JOINER = Joiner.on("/");
029  // Jersey client resource used for the remote communication.
030  protected final WebResource resource;
031  protected final Logger log = LoggerFactory.getLogger(getClass());
032  private final ObjectMapper mapper = new JacksonJsonContextResolver().getContext(null);
033
034  public BaseWsClient(WebResource resource) {
035    this.resource = resource;
036    log.info("Creating new {} using webservices at {}", getClass().getSimpleName(), resource.toString());
037  }
038
039  protected WebResource getResource() {
040    return resource;
041  }
042
043  /**
044   * Gets the base web resource with added paging parameters.
045   *
046   * @param page Which may be null
047   *
048   * @return The resource with appropriate paging parameters set
049   */
050  protected WebResource getResource(@Nullable Pageable page) {
051    return applyPage(resource, page);
052  }
053
054  protected WebResource applyPage(WebResource resource, @Nullable Pageable page) {
055    if (page == null) {
056      return resource;
057    } else {
058      return resource.queryParam(PARAM_LIMIT, String.valueOf(page.getLimit()))
059        .queryParam(PARAM_OFFSET, String.valueOf(page.getOffset()));
060    }
061  }
062
063  /**
064   * Gets the base web resource with added paths.
065   *
066   * @param path optional paths to be appended to the base resource
067   *
068   * @return The resource with appropriate path and parameters set.
069   */
070  protected WebResource getResource(String... path) {
071    return getResource(null, path);
072  }
073
074  /**
075   * Gets the base web resource with added paths and paging parameters.
076   *
077   * @param page Which may be null
078   * @param path optional paths to be appended to the base resource
079   *
080   * @return The resource with appropriate path and parameters set.
081   */
082  protected WebResource getResource(Pageable page, String... path) {
083    return getResource(page).path(PATH_JOINER.join(path));
084  }
085
086  /**
087   * Gets the base web resource with added paging parameters.
088   *
089   * @param page   optional paging parameters
090   * @param params optional additional query parameters
091   * @param path   optional paths to be appended to the base resource
092   *
093   * @return The resource with appropriate path and parameters set.
094   */
095  protected WebResource getResource(@Nullable Pageable page, @Nullable MultivaluedMap<String, String> params,
096    String... path) {
097    WebResource res = getResource(page, path);
098    if (params != null) {
099      return res.queryParams(params);
100    }
101    return res;
102  }
103
104  @Override
105  public String toString() {
106    return getClass().getSimpleName() + "{" + resource + '}';
107  }
108
109  /**
110   * Executes a json type http GET request with paths added to base resource.
111   * The passed generic type for the requested resource is used to convert the response into an object.
112   *
113   * @param gt   the generic type of the returned resource.
114   * @param path optional paths to be appended to the base resource.
115   *
116   * @return requested resource
117   */
118  protected <T> T get(GenericType<T> gt, String... path) {
119    return get(gt, null, null, null, path);
120  }
121
122  /**
123   * Executes a json type http GET request with paths added to base resource.
124   * The passed generic type for the requested resource is used to convert the response into an object.
125   *
126   * @param gt   the generic type of the returned resource.
127   * @param page paging parameters to be applied to request.
128   * @param path optional paths to be appended to the base resource.
129   *
130   * @return requested resource
131   */
132  protected <T> T get(GenericType<T> gt, Pageable page, String... path) {
133    return get(gt, null, null, page, path);
134  }
135
136  /**
137   * Executes a json type http GET request with paths added to base resource.
138   * The passed generic type for the requested resource is used to convert the response into an object.
139   *
140   * @param gt     the generic type of the returned resource.
141   * @param params extra query parameters to be added to request
142   * @param path   optional paths to be appended to the base resource.
143   *
144   * @return requested resource
145   */
146  protected <T> T get(GenericType<T> gt, MultivaluedMap<String, String> params, String... path) {
147    return get(gt, null, params, null, path);
148  }
149
150  /**
151   * Executes a json type http GET request with paths added to base resource.
152   * The passed generic type for the requested resource is used to convert the response into an object.
153   *
154   * @param gt     the generic type of the returned resource.
155   * @param locale identifier for a region to set http language header.
156   * @param page   paging parameters to be applied to request.
157   * @param params extra query parameters to be added to request
158   * @param path   optional paths to be appended to the base resource.
159   *
160   * @return requested resource
161   */
162  protected <T> T get(GenericType<T> gt, @Nullable Locale locale, @Nullable MultivaluedMap<String, String> params,
163    @Nullable Pageable page, String... path) {
164
165    WebResource res = getResource(page, params, path);
166
167    if (locale == null) {
168      return res.type(MediaType.APPLICATION_JSON).get(gt);
169    }
170    return res.type(MediaType.APPLICATION_JSON).acceptLanguage(locale).get(gt);
171  }
172
173  /**
174   * Executes a json type http GET request to any given resource.
175   * The passed generic type for the requested resource is used to convert the response into an object.
176   * Note that the given web resource is used to send the http request and not the internal base resource.
177   *
178   * @param gt       the generic type of the returned resource.
179   * @param page     paging parameters to be applied to request.
180   * @param resource web resource to send request to.
181   *
182   * @return requested resource
183   */
184  protected <T> T get(GenericType<T> gt, Pageable page, WebResource resource) {
185    return applyPage(resource, page).type(MediaType.APPLICATION_JSON).get(gt);
186  }
187
188  /**
189   * Convert the http entity into a byte array to avoid jackson creating a chunked encoding request!
190   */
191  protected byte[] toBytes(Object entity) {
192    try {
193      return mapper.writeValueAsBytes(entity);
194    } catch (IOException e) {
195      log.error("Failed to serialize http entity [{}]", entity);
196      throw new IllegalStateException(e);
197    }
198  }
199
200  /**
201   * Executes an http POST passing on a json encoded entity.
202   *
203   * @param entity to POST
204   * @param path   to POST to
205   */
206  protected void post(Object entity, String... path) {
207    getResource(path).type(MediaType.APPLICATION_JSON).post(toBytes(entity));
208  }
209
210  /**
211   * Executes an http POST passing on a json encoded entity.
212   *
213   * @param gt     the generic type of the returned resource.
214   * @param entity to POST
215   * @param path   to POST to
216   */
217  protected <T> T post(GenericType<T> gt, Object entity, String... path) {
218    return getResource(path).type(MediaType.APPLICATION_JSON).post(gt, toBytes(entity));
219  }
220
221  /**
222   * Executes an http POST passing on a json encoded entity.
223   *
224   * @param cl     the class of the returned resource.
225   * @param entity to POST
226   * @param path   to POST to
227   */
228  protected <T> T post(Class<T> cl, Object entity, String... path) {
229    return getResource(path).type(MediaType.APPLICATION_JSON).post(cl, toBytes(entity));
230  }
231
232  /**
233   * Executes an http PUT passing on a json encoded entity.
234   *
235   * @param entity to PUT
236   * @param path   to PUT to
237   */
238  protected void put(Object entity, String... path) {
239    getResource(path).type(MediaType.APPLICATION_JSON).put(toBytes(entity));
240  }
241
242  /**
243   * Executes an http PUT passing on a json encoded entity.
244   *
245   * @param gt     the generic type of the returned resource.
246   * @param entity to PUT
247   * @param path   to PUT to
248   */
249  protected <T> T put(GenericType<T> gt, Object entity, String... path) {
250    return getResource(path).type(MediaType.APPLICATION_JSON).put(gt, toBytes(entity));
251  }
252
253  /**
254   * Executes an http PUT passing on a json encoded entity.
255   *
256   * @param cl     the class of the returned resource.
257   * @param entity to PUT
258   * @param path   to PUT to
259   */
260  protected <T> T put(Class<T> cl, Object entity, String... path) {
261    return getResource(path).type(MediaType.APPLICATION_JSON).put(cl, toBytes(entity));
262  }
263
264  /**
265   * Executes an http DELETE.
266   *
267   * @param path to DELETE to
268   */
269  protected void delete(String... path) {
270    getResource(path).type(MediaType.APPLICATION_JSON).delete();
271  }
272}
