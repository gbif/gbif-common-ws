004import org.gbif.ws.security.Md5EncodeService;
005import org.gbif.ws.security.Md5EncodeServiceImpl;
006import org.gbif.ws.security.SecretKeySigningService;
007import org.gbif.ws.security.SigningService;
008
009import java.nio.charset.Charset;
010import java.nio.charset.StandardCharsets;
011import java.time.Duration;
012import java.util.Objects;
013
014import com.fasterxml.jackson.databind.ObjectMapper;
015import feign.Contract;
016import feign.Feign;
017import feign.InvocationHandlerFactory;
018import feign.RequestInterceptor;
019import feign.codec.Decoder;
020import feign.codec.Encoder;
021import feign.codec.ErrorDecoder;
022import feign.httpclient.ApacheHttpClient;
023import io.github.resilience4j.core.IntervalFunction;
024import io.github.resilience4j.feign.FeignDecorators;
025import io.github.resilience4j.feign.Resilience4jFeign;
026import io.github.resilience4j.retry.Retry;

039/**
040 * ClientBuilder used to create Feign Clients.
041 * This builders support retry using exponential backoff and multithreaded http client.
042 */
043public class ClientBuilder {
044
045  private static final String HTTP_PROTOCOL = "http";
046  private static final String HTTPS_PROTOCOL = "https";
047
048  private static final RetryRegistry RETRY_REGISTRY = RetryRegistry.ofDefaults();
049  private static final Logger LOG = LoggerFactory.getLogger(ClientBuilder.class);
050
051  private String url;
052  private RequestInterceptor requestInterceptor;
053  private Decoder decoder;
054  private Encoder encoder;
055  private ConnectionPoolConfig connectionPoolConfig;
056  private RetryConfig retryConfig;
057  private ObjectMapper objectMapper;
058
059  private final ErrorDecoder errorDecoder = new ClientErrorDecoder();
060  private final Contract contract = new ClientContract();
061  private final InvocationHandlerFactory invocationHandlerFactory = new ClientInvocationHandlerFactory();
062
063  /**
064   * Creates a builder instance, by default uses the GBIF Jackson ObjectMapper.
065   */
066  public ClientBuilder() {
067    withObjectMapper(JacksonJsonObjectMapperProvider.getObjectMapper());
068  }
069
070  /**
071   * Exponential backoff retry configuration.
072   */
073  public ClientBuilder withExponentialBackoffRetry(Duration initialInterval, double multiplier, int maxAttempts) {
074    retryConfig = RetryConfig.custom()
075                              .maxAttempts(maxAttempts)
076                              .intervalFunction(IntervalFunction.ofExponentialBackoff(initialInterval, multiplier))
077                              .build();
078    return this;
079  }
080
081  /**
082   * Target base url.
083   */
084  public ClientBuilder withUrl(String url) {
085    this.url = url;
086    return this;
087  }
088
089  /**
090   * Simple base credentials.
091   */
092  public ClientBuilder withCredentials(String username, String password) {
093    this.requestInterceptor = new SimpleUserAuthRequestInterceptor(username, password);
094    return this;
095  }
096
097  /**
098   * Custom AppKey credentials.
099   */
100  public ClientBuilder withAppKeyCredentials(String username, String appKey, String secretKey) {
101    this.requestInterceptor =
102      new GbifAuthRequestInterceptor(username, appKey, secretKey, new SecretKeySigningService(),
103                                     new Md5EncodeServiceImpl(objectMapper));
104    return this;
105  }
106
107  /**
108   * Connection pool configuration to create a multithreaded client.
109   */
110  public ClientBuilder withConnectionPoolConfig(ConnectionPoolConfig connectionPoolConfig) {
111    this.connectionPoolConfig = connectionPoolConfig;
112    return this;
113  }
114
115  /**
116   * Jakcson ObjectMapper used to serialize JSON data.
117   */
118  public ClientBuilder withObjectMapper(ObjectMapper objectMapper) {
119    this.objectMapper = objectMapper;
120    this.encoder = new ClientEncoder(objectMapper);
121    this.decoder = new ClientDecoder(objectMapper);
122    return this;
123  }
124
125  /**
126   * Custom GBIF authentication.
127   */
128  public ClientBuilder withCustomGbifAuth(String username,
129                                          String appKey,
130                                          String secretKey,
131                                          SigningService signingService,
132                                          Md5EncodeService md5EncodeService) {
133    this.requestInterceptor =
134      new GbifAuthRequestInterceptor(username, appKey, secretKey, signingService, md5EncodeService);
135    return this;
136  }
137
138  /**
139   * Creates a new client instance.
140   */
141  public <T> T build(Class<T> clazz) {
142
143    Feign.Builder builder = Feign.builder();
144
145    if (Objects.nonNull(retryConfig)) {
146      Retry retry = RETRY_REGISTRY.retry(clazz.getName(), retryConfig);
147      //logging
148      retry.getEventPublisher().onError(event -> LOG.error(event.toString()));
149
150      FeignDecorators decorators = FeignDecorators
151                                    .builder()
152                                    .withRetry(retry)
153                                    .build();
154
155      builder = Resilience4jFeign.builder(decorators);
156
157    } else {
158      //Feign builder do not support invocation handler
159      builder.invocationHandlerFactory(invocationHandlerFactory);
160    }
161
162    builder
163      .encoder(encoder)
164      .decoder(decoder)
165      .errorDecoder(errorDecoder)
166      .contract(contract)
167      .decode404();
168
169    if (requestInterceptor != null) {
170      builder.requestInterceptor(requestInterceptor);
171    }
172    if (connectionPoolConfig != null) {
173      builder.client(new ApacheHttpClient(newMultithreadedClient(connectionPoolConfig)));
174    }
175    return builder.target(clazz, url);
176  }
177
178  /**
179   * Creates a Http multithreaded client.
180   */
181  private static HttpClient newMultithreadedClient(ConnectionPoolConfig connectionPoolConfig) {
182
183    return HttpClients.custom()
184            .setMaxConnTotal(connectionPoolConfig.getMaxConnections())
185            .setMaxConnPerRoute(connectionPoolConfig.getMaxPerRoute())
186            .setDefaultSocketConfig(SocketConfig.custom().setSoTimeout(connectionPoolConfig.getTimeout()).build())
187            .setDefaultConnectionConfig(ConnectionConfig.custom().setCharset(Charset.forName(StandardCharsets.UTF_8.name())).build())
188            .setDefaultRequestConfig(RequestConfig.custom().setConnectTimeout(connectionPoolConfig.getTimeout())
189                                       .setConnectionRequestTimeout(connectionPoolConfig.getTimeout()).build())
190            .build();
191  }
192
193
194  @Data
195  @Builder
196  public static class ConnectionPoolConfig {
197    private final Integer timeout;
198    private final Integer maxConnections;
199    private final Integer maxPerRoute;
200  }
201
202}
