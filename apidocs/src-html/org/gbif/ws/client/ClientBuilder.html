016package org.gbif.ws.client;
017
018import org.gbif.ws.json.JacksonJsonObjectMapperProvider;
019import org.gbif.ws.security.Md5EncodeService;
020import org.gbif.ws.security.Md5EncodeServiceImpl;
021import org.gbif.ws.security.SecretKeySigningService;
022import org.gbif.ws.security.SigningService;
023
024import java.nio.charset.Charset;
025import java.nio.charset.StandardCharsets;
026import java.time.Duration;
027
028import org.apache.http.client.HttpClient;
029import org.apache.http.client.config.RequestConfig;
030import org.apache.http.config.ConnectionConfig;
031import org.apache.http.config.SocketConfig;
032import org.apache.http.impl.client.HttpClients;
033
034import com.fasterxml.jackson.databind.ObjectMapper;
035
036import feign.Contract;
037import feign.Feign;
038import feign.InvocationHandlerFactory;
039import feign.Request;
040import feign.RequestInterceptor;
041import feign.Retryer;
042import feign.codec.Decoder;
043import feign.codec.Encoder;
044import feign.codec.ErrorDecoder;
045import feign.httpclient.ApacheHttpClient;
046import lombok.Builder;
047import lombok.Data;
048
049/**
050 * ClientBuilder used to create Feign Clients.
051 * This builders support retry using exponential backoff and multithreaded http client.
052 */
053@SuppressWarnings("unused")
054public class ClientBuilder {
055
056  private static final String HTTP_PROTOCOL = "http";
057  private static final String HTTPS_PROTOCOL = "https";
058
059  private String url;
060  private int connectTimeoutMillis = 10_000;
061  private int readTimeoutMillis = 60_000;
062  private RequestInterceptor requestInterceptor;
063  private Decoder decoder;
064  private Encoder encoder;
065  private ConnectionPoolConfig connectionPoolConfig;
066  private ObjectMapper objectMapper;
067  private Retryer retryer;
068
069  private final ErrorDecoder errorDecoder = new ClientErrorDecoder();
070  private final Contract contract = new ClientContract();
071  private final InvocationHandlerFactory invocationHandlerFactory =
072      new ClientInvocationHandlerFactory();
073
074  /**
075   * Creates a builder instance, by default uses the GBIF Jackson ObjectMapper.
076   */
077  public ClientBuilder() {
078    withObjectMapper(JacksonJsonObjectMapperProvider.getObjectMapper());
079  }
080
081  /**
082   * Exponential backoff retryer.
083   */
084  public ClientBuilder withExponentialBackoffRetry(
085      Duration initialInterval, double multiplier, int maxAttempts) {
086    retryer = new ClientRetryer(initialInterval.toMillis(), maxAttempts, multiplier);
087    return this;
088  }
089
090  /**
091   * Target base url.
092   */
093  public ClientBuilder withUrl(String url) {
094    this.url = url;
095    return this;
096  }
097
098  /**
099   * Simple base credentials.
100   */
101  public ClientBuilder withCredentials(String username, String password) {
102    this.requestInterceptor = new SimpleUserAuthRequestInterceptor(username, password);
103    return this;
104  }
105
106  /**
107   * Custom AppKey credentials.
108   */
109  public ClientBuilder withAppKeyCredentials(String username, String appKey, String secretKey) {
110    this.requestInterceptor =
111        new GbifAuthRequestInterceptor(
112            username,
113            appKey,
114            secretKey,
115            new SecretKeySigningService(),
116            new Md5EncodeServiceImpl(objectMapper));
117    return this;
118  }
119
120  /**
121   * Connection pool configuration to create a multithreaded client.
122   */
123  public ClientBuilder withConnectionPoolConfig(ConnectionPoolConfig connectionPoolConfig) {
124    this.connectionPoolConfig = connectionPoolConfig;
125    return this;
126  }
127
128  /**
129   * Client connection timeout in milliseconds.
130   */
131  public ClientBuilder withConnectTimeout(int connectTimeoutMillis) {
132    this.connectTimeoutMillis = connectTimeoutMillis;
133    return this;
134  }
135
136  /**
137   * Client read timeout in milliseconds.
138   */
139  public ClientBuilder withReadTimeout(int readTimeoutMillis) {
140    this.readTimeoutMillis = readTimeoutMillis;
141    return this;
142  }
143
144  /**
145   * Jakcson ObjectMapper used to serialize JSON data.
146   */
147  public ClientBuilder withObjectMapper(ObjectMapper objectMapper) {
148    this.objectMapper = objectMapper;
149    this.encoder = new ClientEncoder(objectMapper);
150    this.decoder = new ClientDecoder(objectMapper);
151    return this;
152  }
153
154  /**
155   * Custom GBIF authentication.
156   */
157  public ClientBuilder withCustomGbifAuth(
158      String username,
159      String appKey,
160      String secretKey,
161      SigningService signingService,
162      Md5EncodeService md5EncodeService) {
163    this.requestInterceptor =
164        new GbifAuthRequestInterceptor(
165            username, appKey, secretKey, signingService, md5EncodeService);
166    return this;
167  }
168
169  /**
170   * Creates a new client instance.
171   */
172  public <T> T build(Class<T> clazz) {
173
174    Feign.Builder builder =
175        Feign.builder()
176            .encoder(encoder)
177            .decoder(decoder)
178            .errorDecoder(errorDecoder)
179            .contract(contract)
180            .options(new Request.Options(connectTimeoutMillis, readTimeoutMillis))
181            .decode404()
182            .invocationHandlerFactory(invocationHandlerFactory);
183
184    if (retryer != null) {
185      builder.retryer(retryer);
186    }
187
188    if (requestInterceptor != null) {
189      builder.requestInterceptor(requestInterceptor);
190    }
191
192    if (connectionPoolConfig != null) {
193      builder.client(new ApacheHttpClient(newMultithreadedClient(connectionPoolConfig)));
194    }
195
196    return builder.target(clazz, url);
197  }
198
199  /**
200   * Creates a Http multithreaded client.
201   */
202  private static HttpClient newMultithreadedClient(ConnectionPoolConfig connectionPoolConfig) {
203
204    return HttpClients.custom()
205        .setMaxConnTotal(connectionPoolConfig.getMaxConnections())
206        .setMaxConnPerRoute(connectionPoolConfig.getMaxPerRoute())
207        .setDefaultSocketConfig(
208            SocketConfig.custom().setSoTimeout(connectionPoolConfig.getTimeout()).build())
209        .setDefaultConnectionConfig(
210            ConnectionConfig.custom()
211                .setCharset(Charset.forName(StandardCharsets.UTF_8.name()))
212                .build())
213        .setDefaultRequestConfig(
214            RequestConfig.custom()
215                .setConnectTimeout(connectionPoolConfig.getTimeout())
216                .setConnectionRequestTimeout(connectionPoolConfig.getTimeout())
217                .build())
218        .build();
219  }
220
221  @Data
222  @Builder
223  public static class ConnectionPoolConfig {
224    private final Integer timeout;
225    private final Integer maxConnections;
226    private final Integer maxPerRoute;
227  }
228}
