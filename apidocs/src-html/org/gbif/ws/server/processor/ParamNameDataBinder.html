001package org.gbif.ws.server.processor;
002
003import java.lang.reflect.Method;
004import java.util.Map;
005import javax.servlet.ServletRequest;
006import org.gbif.api.annotation.ParamName;
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009import org.springframework.beans.MutablePropertyValues;
010import org.springframework.web.servlet.mvc.method.annotation.ExtendedServletRequestDataBinder;
011
012/** ServletRequestDataBinder which supports fields renaming using {@link ParamName} */
013@SuppressWarnings("NullableProblems")
014public class ParamNameDataBinder extends ExtendedServletRequestDataBinder {
015
016  private static final Logger LOG = LoggerFactory.getLogger(ParamNameDataBinder.class);
017
018  private final Map<String, String> paramMappings;
019  private final Map<String, String> methodMappings;
020
021  public ParamNameDataBinder(
022      Object target,
023      String objectName,
024      Map<String, String> paramMappings,
025      Map<String, String> methodMappings) {
026    super(target, objectName);
027    this.paramMappings = paramMappings;
028    this.methodMappings = methodMappings;
029  }
030
031  /**
032   * Using a mapping which was created by {@link ParamNameProcessor} this method binds actual values
033   * to the parameters.
034   */
035  @Override
036  protected void addBindValues(
037      MutablePropertyValues mutablePropertyValues, ServletRequest request) {
038    super.addBindValues(mutablePropertyValues, request);
039    for (Map.Entry<String, String> entry : paramMappings.entrySet()) {
040      String paramName = entry.getKey();
041      String fieldName = entry.getValue();
042      if (mutablePropertyValues.contains(paramName)) {
043        mutablePropertyValues.add(
044            fieldName, mutablePropertyValues.getPropertyValue(paramName).getValue());
045      }
046    }
047
048    for (Map.Entry<String, String> entry : methodMappings.entrySet()) {
049      String paramName = entry.getKey();
050      String methodName = entry.getValue();
051      if (mutablePropertyValues.contains(paramName)) {
052        try {
053          final Method declaredMethod =
054              getTarget().getClass().getDeclaredMethod(methodName, String.class);
055          declaredMethod.invoke(
056              getTarget(), mutablePropertyValues.getPropertyValue(paramName).getValue());
057          mutablePropertyValues.removePropertyValue(paramName);
058        } catch (Exception e) {
059          LOG.error("There was a problem to invoke a method {}", methodName);
060        }
061      }
062    }
063  }
064}
