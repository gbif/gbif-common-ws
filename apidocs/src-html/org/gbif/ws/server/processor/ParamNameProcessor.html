001package org.gbif.ws.server.processor;
002
003import java.lang.reflect.Field;
004import java.lang.reflect.Method;
005import java.util.Arrays;
006import java.util.HashMap;
007import java.util.Map;
008import java.util.concurrent.ConcurrentHashMap;
009import org.gbif.api.annotation.ParamName;
010import org.springframework.beans.BeanUtils;
011import org.springframework.beans.factory.annotation.Autowired;
012import org.springframework.core.MethodParameter;
013import org.springframework.web.bind.WebDataBinder;
014import org.springframework.web.bind.annotation.RequestParam;
015import org.springframework.web.context.request.NativeWebRequest;
016import org.springframework.web.servlet.mvc.method.annotation.RequestMappingHandlerAdapter;
017import org.springframework.web.servlet.mvc.method.annotation.ServletModelAttributeMethodProcessor;
018
019/** Process {@link ParamName}. */
020public class ParamNameProcessor extends ServletModelAttributeMethodProcessor {
021
022  @Autowired private RequestMappingHandlerAdapter requestMappingHandlerAdapter;
023
024  private static final Map<Class<?>, Map<String, String>> PARAM_MAPPINGS_CACHE =
025      new ConcurrentHashMap<>(256);
026  private static final Map<Class<?>, Map<String, String>> METHODS_MAPPINGS_CACHE =
027      new ConcurrentHashMap<>(256);
028
029  public ParamNameProcessor() {
030    super(false);
031  }
032
033  @Override
034  public boolean supportsParameter(MethodParameter parameter) {
035    return parameter.hasParameterAnnotation(RequestParam.class)
036        && !BeanUtils.isSimpleProperty(parameter.getParameterType())
037        && (Arrays.stream(parameter.getParameterType().getDeclaredMethods())
038                .anyMatch(method -> method.getAnnotation(ParamName.class) != null)
039            || Arrays.stream(parameter.getParameterType().getDeclaredFields())
040                .anyMatch(field -> field.getAnnotation(ParamName.class) != null));
041  }
042
043  @Override
044  protected void bindRequestParameters(WebDataBinder binder, NativeWebRequest nativeWebRequest) {
045    Object target = binder.getTarget();
046    Map<String, String> paramMappings = this.getParamMappings(target.getClass());
047    Map<String, String> methodMappings = this.getMethodMappings(target.getClass());
048    ParamNameDataBinder paramNameDataBinder =
049        new ParamNameDataBinder(target, binder.getObjectName(), paramMappings, methodMappings);
050    requestMappingHandlerAdapter.getWebBindingInitializer().initBinder(paramNameDataBinder);
051    super.bindRequestParameters(paramNameDataBinder, nativeWebRequest);
052  }
053
054  /**
055   * Get param mappings. It creates a simple mapping: parameter name -> field name. Cache param
056   * mappings in memory.
057   */
058  private Map<String, String> getParamMappings(Class<?> targetClass) {
059    // first check cache
060    if (PARAM_MAPPINGS_CACHE.containsKey(targetClass)) {
061      return PARAM_MAPPINGS_CACHE.get(targetClass);
062    }
063
064    Map<String, String> paramMappings = new HashMap<>(32);
065
066    // process fields
067    Field[] fields = targetClass.getDeclaredFields();
068    for (Field field : fields) {
069      ParamName paramName = field.getAnnotation(ParamName.class);
070      if (paramName != null && !paramName.value().isEmpty()) {
071        paramMappings.put(paramName.value(), field.getName());
072      }
073    }
074
075    // put them to cache
076    PARAM_MAPPINGS_CACHE.put(targetClass, paramMappings);
077    return paramMappings;
078  }
079
080  /**
081   * Get param mappings. It creates a simple mapping: parameter name -> method name. Cache param
082   * mappings in memory.
083   */
084  private Map<String, String> getMethodMappings(Class<?> targetClass) {
085    // first check cache
086    if (METHODS_MAPPINGS_CACHE.containsKey(targetClass)) {
087      return METHODS_MAPPINGS_CACHE.get(targetClass);
088    }
089
090    Map<String, String> methodMappings = new HashMap<>(32);
091
092    // process methods
093    final Method[] methods = targetClass.getDeclaredMethods();
094    for (Method method : methods) {
095      final ParamName paramName = method.getAnnotation(ParamName.class);
096      if (paramName != null && !paramName.value().isEmpty()) {
097        methodMappings.put(paramName.value(), method.getName());
098      }
099    }
100
101    // put them to cache
102    METHODS_MAPPINGS_CACHE.put(targetClass, methodMappings);
103    return methodMappings;
104  }
105}
