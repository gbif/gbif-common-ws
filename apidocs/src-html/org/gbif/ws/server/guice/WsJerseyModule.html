001package org.gbif.ws.server.guice;
002
003import org.gbif.ws.json.JacksonJsonContextResolver;
004import org.gbif.ws.server.filter.AuthFilter;
005import org.gbif.ws.server.filter.AuthResponseFilter;
006import org.gbif.ws.server.filter.CreatedResponseFilter;
007import org.gbif.ws.server.filter.CrossDomainResponseFilter;
008import org.gbif.ws.server.filter.JsonpResponseFilter;
009import org.gbif.ws.server.filter.RequestHeaderParamUpdateFilter;
010import org.gbif.ws.server.interceptor.NullToNotFound;
011import org.gbif.ws.server.interceptor.NullToNotFoundInterceptor;
012
013import java.util.HashMap;
014import java.util.LinkedList;
015import java.util.List;
016import java.util.Map;
017import javax.annotation.Nullable;
018import javax.ws.rs.core.MediaType;
019
020import com.google.common.base.Function;
021import com.google.common.base.Joiner;
022import com.google.common.collect.Lists;
023import com.google.common.collect.Maps;
024import com.google.inject.matcher.Matchers;
025import com.sun.jersey.api.container.filter.RolesAllowedResourceFilterFactory;
026import com.sun.jersey.api.core.PackagesResourceConfig;
027import com.sun.jersey.api.core.ResourceConfig;
028import com.sun.jersey.api.json.JSONConfiguration;
029import com.sun.jersey.guice.JerseyServletModule;
030import com.sun.jersey.guice.spi.container.servlet.GuiceContainer;
031import com.sun.jersey.spi.container.ContainerRequestFilter;
032import com.sun.jersey.spi.container.ContainerResponseFilter;
033import com.sun.jersey.spi.container.ResourceFilters;
034
035/**
036 * A basic jersey configuration for JSON, JSONP etc.
037 * All providers found in org.gbif.ws.server.provider will be automatically added.
038 */
039public class WsJerseyModule extends JerseyServletModule {
040
041  private final String resourcePackages;
042  private final boolean installAuth;
043  private final List<Class<? extends ContainerResponseFilter>> responseFilters = Lists.newArrayList();
044  private final LinkedList<Class<? extends ContainerRequestFilter>> requestFilters = Lists.newLinkedList();
045
046  private final Function<Class<?>, String> fnClassName = new Function<Class<?>, String>() {
047
048    @Override
049    public String apply(Class<?> input) {
050      return input.getName();
051    }
052  };
053
054  /**
055   * Creates the module using default configuration, optionally installing authentication filters and customising the
056   * responses.
057   * 
058   * @param resourcePackages To scan for the Jersey resources
059   * @param installAuth Controls whether the GBIF default AuthFilter should be configured
060   * @param responseFilters Optionally controlling the response
061   */
062  public WsJerseyModule(String resourcePackages, boolean installAuth,
063    @Nullable List<Class<? extends ContainerResponseFilter>> responseFilters) {
064    this(resourcePackages, installAuth, responseFilters, null);
065  }
066
067  /**
068   * Creates the module using default configuration, optionally installing authentication filters and customising the
069   * responses. Note that the request filters will be handled as follows if provided:
070   * <ol>
071   * <li>The default filters will always be first</li>
072   * <li>The authentication filter will be next if enabled (installAuth=true)</li>
073   * <li>The provided filters will be last in the given order</li>
074   * </ol>
075   * 
076   * @param resourcePackages To scan for the Jersey resources
077   * @param installAuth Controls whether the authentication filter should be configured
078   * @param responseFilters To customise the response if necessary
079   * @param requestFilters To apply in the chain of request filtering (see note above)
080   */
081  public WsJerseyModule(String resourcePackages, boolean installAuth,
082    @Nullable List<Class<? extends ContainerResponseFilter>> responseFilters,
083    List<Class<? extends ContainerRequestFilter>> requestFilters) {
084    // Let Jersey look for root resources automatically
085    this.resourcePackages = "org.gbif.ws.server.provider," + resourcePackages;
086    this.installAuth = installAuth;
087    if (responseFilters != null) {
088      this.responseFilters.addAll(responseFilters);
089    }
090    if (requestFilters != null) {
091      this.requestFilters.addAll(requestFilters);
092    }
093  }
094
095  @Override
096  protected void configureServlets() {
097    Map<String, String> params = new HashMap<String, String>(5);
098
099    bind(JacksonJsonContextResolver.class);
100    params.put(JSONConfiguration.FEATURE_POJO_MAPPING, "true");
101
102    // allow resources to use suffices to ask for media types
103    Map<String, String> mediaTypes = Maps.newHashMap();
104    mediaTypes.put("json", MediaType.APPLICATION_JSON);
105    mediaTypes.put("xml", MediaType.APPLICATION_XML);
106    mediaTypes.put("txt", MediaType.TEXT_PLAIN);
107    mediaTypes.put("zip", MediaType.APPLICATION_OCTET_STREAM);
108    params.put(ResourceConfig.PROPERTY_MEDIA_TYPE_MAPPINGS,
109      Joiner.on(", ").withKeyValueSeparator(" : ").join(mediaTypes));
110
111    // Let Jersey look for root resources automatically
112    params.put(PackagesResourceConfig.PROPERTY_PACKAGES, resourcePackages);
113
114    // secure resources via JSR 250
115    params.put(ResourceFilters.class.getName(), RolesAllowedResourceFilterFactory.class.getName());
116
117    // The default filters are always added to start of the chain followed by authentication
118    if (installAuth) {
119      requestFilters.addFirst(AuthFilter.class);
120    }
121    requestFilters.addFirst(RequestHeaderParamUpdateFilter.class);
122    params.put(ResourceConfig.PROPERTY_CONTAINER_REQUEST_FILTERS,
123      Joiner.on(";").join(Lists.transform(requestFilters, fnClassName)));
124
125    // response filters
126    List<Class<? extends ContainerResponseFilter>> respFilters = Lists.newArrayList();
127    respFilters.add(JsonpResponseFilter.class);
128    respFilters.add(CreatedResponseFilter.class);
129    respFilters.add(CrossDomainResponseFilter.class);
130    if (installAuth) {
131      respFilters.add(AuthResponseFilter.class);
132    }
133    respFilters.addAll(responseFilters);
134    params.put(ResourceConfig.PROPERTY_CONTAINER_RESPONSE_FILTERS,
135      Joiner.on(";").join(Lists.transform(respFilters, fnClassName)));
136
137    serve("/*").with(GuiceContainer.class, params);
138
139    bindInterceptor(Matchers.any(), Matchers.annotatedWith(NullToNotFound.class), new NullToNotFoundInterceptor());
140  }
141
142}
