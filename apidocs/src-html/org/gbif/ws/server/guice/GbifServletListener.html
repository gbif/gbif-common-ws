001package org.gbif.ws.server.guice;
002
003import org.gbif.ws.json.JacksonJsonContextResolver;
004import org.gbif.ws.util.PropertiesUtil;
005
006import java.io.IOException;
007import java.util.List;
008import java.util.Map;
009import java.util.Properties;
010
011import javax.annotation.Nullable;
012
013import com.google.common.collect.Lists;
014import com.google.common.collect.Maps;
015import com.google.inject.Guice;
016import com.google.inject.Injector;
017import com.google.inject.Module;
018import com.google.inject.servlet.GuiceServletContextListener;
019import com.sun.jersey.spi.container.ContainerRequestFilter;
020import com.sun.jersey.spi.container.ContainerResponseFilter;
021
022/**
023 * A basic servlet listener for all our webservices.
024 * Note that the request filters will be ordered as follows if provided:
025 * <ol>
026 * <li>The default filters will always be first</li>
027 * <li>The authentication filter will be next if enabled (installAuth=true)</li>
028 * <li>The provided filters will be last in the given order</li>
029 * </ol>
030 */
031public abstract class GbifServletListener extends GuiceServletContextListener {
032
033  private final Properties properties;
034  private final String resourcePackages;
035  private final boolean installAuth;
036  private final List<Class<? extends ContainerResponseFilter>> responseFilters = Lists.newArrayList();
037  private final List<Class<? extends ContainerRequestFilter>> requestFilters = Lists.newArrayList();
038
039  private Injector injector;
040
041  /**
042   * @param properties The properties to load
043   * @param resourcePackages packages to scan for jersey resources,
044   *        for example "org.gbif.registry.ws,org.gbif.registry.common.ws"
045   * @param installAuthenticationFilters if true installs support for authentication. Requires an implementation of
046   *        UserService to be installed by one of the additional modules.
047   */
048  protected GbifServletListener(Properties properties, String resourcePackages, boolean installAuthenticationFilters) {
049    this(properties, resourcePackages, installAuthenticationFilters, null);
050  }
051
052  /**
053   * @param properties The properties to load
054   * @param resourcePackages packages to scan for jersey resources,
055   *        for example "org.gbif.registry.ws,org.gbif.registry.common.ws"
056   * @param installAuthenticationFilters if true installs support for authentication. Requires an implementation of
057   *        UserService to be installed by one of the additional modules.
058   * @param responseFilters A list of response filters that are applied to the requests.
059   */
060  protected GbifServletListener(Properties properties, String resourcePackages, boolean installAuthenticationFilters,
061    @Nullable List<Class<? extends ContainerResponseFilter>> responseFilters) {
062    this(properties, resourcePackages, installAuthenticationFilters, responseFilters, null);
063  }
064
065  /**
066   * @param properties The properties to load
067   * @param resourcePackages packages to scan for jersey resources,
068   *        for example "org.gbif.registry.ws,org.gbif.registry.common.ws"
069   * @param installAuthenticationFilters if true installs support for authentication. Requires an implementation of
070   *        UserService to be installed by one of the additional modules.
071   * @param responseFilters A list of response filters that are applied to the requests.
072   * @param requestFilters A list of request filters that are applied to the requests. Note: read constructor
073   *        documentation on {@link WsJerseyModule}
074   */
075  protected GbifServletListener(Properties properties, String resourcePackages, boolean installAuthenticationFilters,
076    @Nullable List<Class<? extends ContainerResponseFilter>> responseFilters,
077    @Nullable List<Class<? extends ContainerRequestFilter>> requestFilters) {
078    this.properties = properties;
079    this.resourcePackages = resourcePackages;
080    this.installAuth = installAuthenticationFilters;
081    if (responseFilters != null) {
082      this.responseFilters.addAll(responseFilters);
083    }
084    if (requestFilters != null) {
085      this.requestFilters.addAll(requestFilters);
086    }
087  }
088
089  /**
090   * @param propertyFileName Fully qualified filename that must be on the classpath
091   * @param resourcePackages packages to scan for jersey resources,
092   *        for example "org.gbif.registry.ws,org.gbif.registry.common.ws"
093   * @param installAuthenticationFilters if true installs support for authentication. Requires an implementation of
094   *        UserService to be installed by one of the additional modules.
095   * @param responseFilters A list of response filters that are applied to the requests.
096   */
097  protected GbifServletListener(String propertyFileName, String resourcePackages, boolean installAuthenticationFilters,
098    @Nullable List<Class<? extends ContainerResponseFilter>> responseFilters) {
099    this(readProperties(propertyFileName), resourcePackages, installAuthenticationFilters,
100      responseFilters);
101  }
102
103  /**
104   * Hides the checked exception that can be thrown when reading a file.
105   */
106  private static Properties readProperties(String propertyFileName) {
107    Properties properties = null;
108    try {
109      properties = PropertiesUtil.readFromClasspath(propertyFileName);
110    } catch (IOException e) {
111      new IllegalArgumentException("Error reading properties file", e);
112    }
113    return properties;
114  }
115
116  /**
117   * @param propertyFileName Fully qualified filename that must be on the classpath
118   * @param resourcePackages packages to scan for jersey resources,
119   *        for example "org.gbif.registry.ws,org.gbif.registry.common.ws"
120   * @param installAuthenticationFilters if true installs support for authentication. Requires an implementation of
121   *        UserService to be installed by one of the additional modules.
122   */
123  protected GbifServletListener(String propertyFileName, String resourcePackages,
124    boolean installAuthenticationFilters) {
125    this(propertyFileName, resourcePackages, installAuthenticationFilters, null);
126  }
127
128
129  /**
130   * Implement this to return the list of all additional modules needed to be installed.
131   * 
132   * @return a list of modules, never null.
133   */
134  protected abstract List<Module> getModules(Properties properties);
135
136  /**
137   * The Jackson JSON configuration needs to know about how to (de)serialize polymorphic classes.
138   * Override this method to pass a map of mixIn classes into the Jackson context resolver.
139   * 
140   * @return the mixIn class map. Defaults to an empty map.
141   */
142  protected Map<Class<?>, Class<?>> getPolymorphicClassMap() {
143    return Maps.newHashMap();
144  }
145
146  @Override
147  protected synchronized Injector getInjector() {
148    if (injector == null) {
149
150      // configure jsonMixIns
151      JacksonJsonContextResolver.addMixIns(getPolymorphicClassMap());
152
153      List<Module> modules = getModules(properties);
154      modules.add(new WsJerseyModule(resourcePackages, installAuth, responseFilters, requestFilters));
155
156      injector = Guice.createInjector(modules);
157    }
158    return injector;
159  }
160
161}
