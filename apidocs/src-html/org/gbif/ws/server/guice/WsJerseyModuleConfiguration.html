001package org.gbif.ws.server.guice;
002
003import org.gbif.ws.server.filter.AuthFilter;
004import org.gbif.ws.server.filter.CreatedResponseFilter;
005import org.gbif.ws.server.filter.CrossDomainResponseFilter;
006import org.gbif.ws.server.filter.JsonpResponseFilter;
007
008import java.util.List;
009import javax.annotation.Nullable;
010
011import com.google.common.collect.Lists;
012import com.sun.jersey.spi.container.ContainerRequestFilter;
013import com.sun.jersey.spi.container.ContainerResponseFilter;
014
015/**
016 * WsJerseyModule configuration class with fluent interface.
017 * Default response filters represent filters that will be added in front of other response filters, including
018 * authentication filter.
019 */
020public class WsJerseyModuleConfiguration {
021
022  private String resourcePackages;
023
024  private Boolean installAuthenticationFilter;
025  private Class<? extends ContainerRequestFilter> authenticationFilterClass;
026
027  private List<Class<? extends ContainerResponseFilter>> defaultResponseFilters = Lists.newArrayList();
028  private List<Class<? extends ContainerResponseFilter>> responseFilters = Lists.newArrayList();
029  private List<Class<? extends ContainerRequestFilter>> requestFilters = Lists.newLinkedList();
030
031  /**
032   * Build WsJerseyModuleConfiguration with default configuration values.
033   */
034  public WsJerseyModuleConfiguration() {
035    defaultResponseFilters.add(JsonpResponseFilter.class);
036    defaultResponseFilters.add(CreatedResponseFilter.class);
037    defaultResponseFilters.add(CrossDomainResponseFilter.class);
038  }
039
040  /**
041   * To scan for the Jersey resources.
042   */
043  public WsJerseyModuleConfiguration resourcePackages(String resourcePackages) {
044    this.resourcePackages = resourcePackages;
045    return this;
046  }
047
048  /**
049   * Controls whether the GBIF default AuthFilter should be installed.
050   */
051  public WsJerseyModuleConfiguration installAuthenticationFilter(boolean installAuthenticationFilter) {
052    this.installAuthenticationFilter = installAuthenticationFilter;
053    return this;
054  }
055
056  /**
057   * Allows to use a custom authentication filter instead of the default {@link AuthFilter}.
058   */
059  public WsJerseyModuleConfiguration useAuthenticationFilter(Class<? extends ContainerRequestFilter>
060                                                                     authenticationFilterClass) {
061    this.installAuthenticationFilter = true;
062    this.authenticationFilterClass = authenticationFilterClass;
063    return this;
064  }
065
066  public WsJerseyModuleConfiguration responseFilters(@Nullable List<Class<? extends ContainerResponseFilter>> responseFilters) {
067    if (responseFilters != null) {
068      this.responseFilters.addAll(responseFilters);
069    }
070    return this;
071  }
072
073  public WsJerseyModuleConfiguration requestFilters(@Nullable List<Class<? extends ContainerRequestFilter>> requestFilters) {
074    if (requestFilters != null) {
075      this.requestFilters.addAll(requestFilters);
076    }
077    return this;
078  }
079
080  /**
081   * Replaces the defaultResponseFilters to be used by the WsJerseyModule.
082   * defaultResponseFilters represent filters that will be added in front of other response filters, including
083   * authentication filter.
084   */
085  public WsJerseyModuleConfiguration replaceDefaultResponseFilters(List<Class<? extends ContainerResponseFilter>> defaultResponseFilters) {
086    this.defaultResponseFilters.clear();
087    if (defaultResponseFilters != null) {
088      this.defaultResponseFilters.addAll(defaultResponseFilters);
089    }
090    return this;
091  }
092
093  public String getResourcePackages() {
094    return resourcePackages;
095  }
096
097  public Boolean isInstallAuthenticationFilter() {
098    return installAuthenticationFilter;
099  }
100
101  public Class<? extends ContainerRequestFilter> getAuthenticationFilterClass() {
102    return authenticationFilterClass;
103  }
104
105  public List<Class<? extends ContainerResponseFilter>> getDefaultResponseFilters() {
106    return defaultResponseFilters;
107  }
108
109  public List<Class<? extends ContainerResponseFilter>> getResponseFilters() {
110    return responseFilters;
111  }
112
113  public List<Class<? extends ContainerRequestFilter>> getRequestFilters() {
114    return requestFilters;
115  }
116
117}
