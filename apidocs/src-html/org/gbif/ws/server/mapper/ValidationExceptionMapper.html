001package org.gbif.ws.server.mapper;
002
003import com.google.common.base.Joiner;
004import com.google.common.collect.ImmutableList;
005import java.util.Comparator;
006import javax.validation.ConstraintViolation;
007import javax.validation.ConstraintViolationException;
008import javax.validation.Path;
009import org.slf4j.Logger;
010import org.slf4j.LoggerFactory;
011import org.springframework.http.HttpStatus;
012import org.springframework.http.MediaType;
013import org.springframework.http.ResponseEntity;
014import org.springframework.validation.FieldError;
015import org.springframework.web.bind.MethodArgumentNotValidException;
016import org.springframework.web.bind.annotation.ControllerAdvice;
017import org.springframework.web.bind.annotation.ExceptionHandler;
018
019/**
020 * Converts validation exceptions into a http 422 bad request and gives a meaningful messages on the
021 * issues.
022 */
023@ControllerAdvice
024public class ValidationExceptionMapper {
025
026  private static final Logger LOG = LoggerFactory.getLogger(ValidationExceptionMapper.class);
027
028  @ExceptionHandler(MethodArgumentNotValidException.class)
029  public ResponseEntity<Object> toResponse(MethodArgumentNotValidException exception) {
030    ImmutableList.Builder<String> builder = ImmutableList.builder();
031
032    exception.getBindingResult()
033        .getAllErrors()
034        .stream()
035        .map(error -> ((FieldError) error))
036        .sorted(Comparator.comparing(FieldError::getField, Comparator.naturalOrder()))
037        .forEach(error -> {
038          LOG.debug("Validation of [{}] failed: {}", error.getField(), error.getDefaultMessage());
039          builder.add(String.format("Validation of [%s] failed: %s", error.getField(),
040              error.getDefaultMessage()));
041        });
042
043    return ResponseEntity.status(HttpStatus.UNPROCESSABLE_ENTITY)
044        .contentType(MediaType.TEXT_PLAIN)
045        .body("<ul><li>" + Joiner.on("</li><li>").join(builder.build()) + "</li></ul>");
046  }
047
048  @ExceptionHandler(ConstraintViolationException.class)
049  public ResponseEntity<Object> toResponse(ConstraintViolationException exception) {
050    ImmutableList.Builder<String> builder = ImmutableList.builder();
051
052    exception.getConstraintViolations()
053        .stream()
054        .sorted(Comparator.comparing(cv -> getPropertyFromPropertyPath(cv.getPropertyPath()), Comparator.naturalOrder()))
055        .forEach(cv -> {
056          LOG.debug("Validation of [{}] failed: {}", cv.getPropertyPath(), cv.getMessage());
057          builder.add(String.format("Validation of [%s] failed: %s",
058              getPropertyFromPropertyPath(cv.getPropertyPath()), cv.getMessage()));
059        });
060
061    return ResponseEntity.status(HttpStatus.UNPROCESSABLE_ENTITY)
062        .contentType(MediaType.TEXT_PLAIN)
063        .body("<ul><li>" + Joiner.on("</li><li>").join(builder.build()) + "</li></ul>");
064  }
065
066  private String getPropertyFromPropertyPath(Path propertyPath) {
067    String resultProperty = null;
068
069    if (propertyPath != null) {
070      resultProperty = propertyPath.toString();
071      int lastDotIndex = resultProperty.lastIndexOf('.');
072      if (lastDotIndex != -1) {
073        resultProperty = resultProperty.substring(lastDotIndex + 1);
074      }
075    }
076
077    return resultProperty;
078  }
079}
