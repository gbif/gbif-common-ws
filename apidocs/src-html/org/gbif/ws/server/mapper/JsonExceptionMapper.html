001package org.gbif.ws.server.mapper;
002
003import com.fasterxml.jackson.core.JsonParseException;
004import com.fasterxml.jackson.core.JsonProcessingException;
005import com.fasterxml.jackson.databind.JsonMappingException;
006import com.fasterxml.jackson.databind.RuntimeJsonMappingException;
007import org.springframework.http.HttpStatus;
008import org.springframework.http.MediaType;
009import org.springframework.http.ResponseEntity;
010import org.springframework.web.bind.annotation.ControllerAdvice;
011import org.springframework.web.bind.annotation.ExceptionHandler;
012
013/**
014 * Converts json related exceptions into a http 400 bad request.
015 */
016@ControllerAdvice
017public class JsonExceptionMapper {
018
019  @ExceptionHandler({
020      JsonMappingException.class,
021      JsonParseException.class,
022      JsonProcessingException.class,
023      RuntimeJsonMappingException.class})
024  public ResponseEntity<Object> toResponse(Exception exception) {
025    return ResponseEntity.status(HttpStatus.BAD_REQUEST)
026        .contentType(MediaType.TEXT_PLAIN)
027        .body(exception.getMessage());
028  }
029}
