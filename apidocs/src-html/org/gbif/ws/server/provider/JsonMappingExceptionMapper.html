001package org.gbif.ws.server.provider;
002
003import javax.ws.rs.core.MediaType;
004import javax.ws.rs.core.Response;
005import javax.ws.rs.ext.ExceptionMapper;
006import javax.ws.rs.ext.Provider;
007
008import com.google.inject.Singleton;
009import org.codehaus.jackson.map.JsonMappingException;
010
011/**
012 * Converts a JsonMappingException into a http 400 bad request.
013 */
014@Provider
015@Singleton
016public class JsonMappingExceptionMapper implements ExceptionMapper<JsonMappingException> {
017
018  @Override
019  public Response toResponse(JsonMappingException exception) {
020    return Response.status(Response.Status.BAD_REQUEST).type(MediaType.TEXT_PLAIN).entity(exception.getMessage()).build();
021  }
022}
