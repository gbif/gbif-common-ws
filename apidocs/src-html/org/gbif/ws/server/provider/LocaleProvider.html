016package org.gbif.ws.server.provider;
017
018import java.lang.reflect.Type;
019import java.util.List;
020import java.util.Locale;
021import javax.ws.rs.core.Context;
022import javax.ws.rs.ext.Provider;
023
024import com.google.common.base.Strings;
025import com.google.inject.Singleton;
026import com.sun.jersey.api.core.HttpContext;
027import com.sun.jersey.core.spi.component.ComponentContext;
028import com.sun.jersey.core.spi.component.ComponentScope;
029import com.sun.jersey.server.impl.inject.AbstractHttpContextInjectable;
030import com.sun.jersey.spi.inject.Injectable;
031import com.sun.jersey.spi.inject.InjectableProvider;
032
033/**
034 * Jersey provider class that extracts the requested locale based on http header or language query parameter.
035 * This allows resources to access a locale context very easily while keeping all logic in this class.
036 * The accept any language value * will be converted into a null locale.
037 * Example resource use:
038 * <pre>
039 * {@code
040 * public String getVernacularName(@Context Locale locale) {
041 *   return "this is the " + locale + " vernacular name: xyz");
042 * }
043 * }
044 * </pre>
045 */
046@Provider
047@Singleton
048public class LocaleProvider extends AbstractHttpContextInjectable<Locale> implements InjectableProvider<Context, Type> {
049
050  private static final String LANGUAGE_PARAM = "language";
051
052  @Override
053  public Injectable<Locale> getInjectable(ComponentContext ic, Context a, Type c) {
054    if (c.equals(Locale.class)) {
055      return this;
056    }
057
058    return null;
059  }
060
061  @Override
062  public ComponentScope getScope() {
063    return ComponentScope.PerRequest;
064  }
065
066  @Override
067  public Locale getValue(HttpContext c) {
068    return getLocale(c);
069  }
070
071  public static Locale getLocale(HttpContext c) {
072    // try language parameter first
073    if (c.getRequest().getQueryParameters() != null && c.getRequest().getQueryParameters()
074      .containsKey(LANGUAGE_PARAM)) {
075      String lang = c.getRequest().getQueryParameters().getFirst(LANGUAGE_PARAM).trim().toLowerCase();
076      // iso language has to be 2 lower case letters!
077      if (!Strings.isNullOrEmpty(lang) && lang.length() == 2) {
078        return new Locale(lang);
079      }
080    }
081
082    // try headers next
083    final List<Locale> locales = c.getRequest().getAcceptableLanguages();
084    for (Locale loc : locales) {
085      // ignore accept any language value: * and non iso 2 letter codes
086      if (!Strings.isNullOrEmpty(loc.getLanguage()) && !loc.getLanguage().equalsIgnoreCase("*")
087          && loc.getLanguage().length() == 2) {
088        return loc;
089      }
090    }
091    return null;
092  }
093}
