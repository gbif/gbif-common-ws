013package org.gbif.ws.server.provider;
014
015import org.gbif.api.model.common.search.SearchParameter;
016import org.gbif.api.model.common.search.SearchRequest;
017import org.gbif.api.util.SearchTypeValidator;
018import org.gbif.api.util.VocabularyUtils;
019
020import java.lang.reflect.Type;
021import java.util.List;
022import java.util.Map.Entry;
023import javax.ws.rs.core.Context;
024import javax.ws.rs.core.MultivaluedMap;
025
026import com.google.common.base.Strings;
027import com.google.common.collect.Lists;
028import com.sun.jersey.api.core.HttpContext;
029import com.sun.jersey.core.spi.component.ComponentContext;
030import com.sun.jersey.core.spi.component.ComponentScope;
031import com.sun.jersey.server.impl.inject.AbstractHttpContextInjectable;
032import com.sun.jersey.spi.inject.Injectable;
033import com.sun.jersey.spi.inject.InjectableProvider;
034
035import static org.gbif.ws.util.WebserviceParameter.PARAM_HIGHLIGHT;
036import static org.gbif.ws.util.WebserviceParameter.PARAM_QUERY_STRING;
037
038/**
039 * Provider class that transforms a set of HTTP parameters into a SearchRequest class instance.
040 * This assumes the existence of the following parameters in the HTTP request:
041 * 'page_size', 'offset', 'q' and any of the search parameter enum member names case insensitively.
042 */
043public class SearchRequestProvider<RT extends SearchRequest<P>, P extends Enum<?> & SearchParameter>
044  extends AbstractHttpContextInjectable<RT> implements InjectableProvider<Context, Type> {
045
046  private final Class<P> searchParameterClass;
047  private final Class<RT> requestType;
048
049  public SearchRequestProvider(Class<RT> requestType, Class<P> searchParameterClass) {
050    this.requestType = requestType;
051    this.searchParameterClass = searchParameterClass;
052  }
053
054  /**
055   * Get an injectable.
056   *
057   * @param ic the injectable context
058   * @param context the annotation instance
059   * @param type the context instance
060   * @return an Injectable instance, otherwise null if an instance cannot
061   *         be created.
062   */
063  @Override
064  public Injectable<RT> getInjectable(ComponentContext ic, Context context, Type type) {
065    if (type.equals(requestType)) {
066      return this;
067    }
068    return null;
069  }
070
071  /**
072   * Get the scope of the injectable provider.
073   *
074   * @return the scope.
075   */
076  @Override
077  public ComponentScope getScope() {
078    return ComponentScope.PerRequest;
079  }
080
081  @Override
082  public RT getValue(HttpContext context) {
083    try {
084      RT req = requestType.newInstance();
085      return getSearchRequest(context, req);
086    } catch (InstantiationException e) {
087    } catch (IllegalAccessException e) {
088    }
089    return null;
090  }
091
092  protected P findSearchParam(String name) {
093    try {
094      return (P) VocabularyUtils.lookupEnum(name, searchParameterClass);
095    } catch (IllegalArgumentException e) {
096      // we have all params here, not only the enum ones, so this is ok to end up here a few times
097    }
098    return null;
099  }
100
101  protected RT getSearchRequest(HttpContext context, RT searchRequest) {
102    searchRequest.copyPagingValues(PageableProvider.getPagingRequest(context));
103
104    final MultivaluedMap<String, String> params = context.getRequest().getQueryParameters();
105    final String q = params.getFirst(PARAM_QUERY_STRING);
106    final String highlightValue = params.getFirst(PARAM_HIGHLIGHT);
107
108    if (!Strings.isNullOrEmpty(q)) {
109      searchRequest.setQ(q);
110    }
111    if (!Strings.isNullOrEmpty(highlightValue)) {
112      searchRequest.setHighlight(Boolean.parseBoolean(highlightValue));
113    }
114    // find search parameter enum based filters
115    setSearchParams(searchRequest, params);
116
117    return searchRequest;
118  }
119
120  /**
121   * Removes all empty and null parameters from the list.
122   * Each value is trimmed(String.trim()) in order to remove all sizes of empty parameters.
123   */
124  private List<String> removeEmptyParameters(List<String> parameters) {
125    List<String> cleanParameters = Lists.newArrayList();
126    for (String param : parameters) {
127      final String cleanParam = Strings.nullToEmpty(param).trim();
128      if (cleanParam.length() > 0) {
129        cleanParameters.add(cleanParam);
130      }
131    }
132    return cleanParameters;
133  }
134
135  /**
136   * Iterates over the params map and adds to the search request the recognized parameters (i.e.: those that have a
137   * correspondent value in the P generic parameter).
138   * Empty (of all size) and null parameters are discarded.
139   */
140  private void setSearchParams(RT searchRequest, MultivaluedMap<String, String> params) {
141    for (Entry<String, List<String>> entry : params.entrySet()) {
142      P p = findSearchParam(entry.getKey());
143      if (p != null) {
144        for (String val : removeEmptyParameters(entry.getValue())) {
145          // validate value for certain types
146          SearchTypeValidator.validate(p, val);
147          searchRequest.addParameter(p, val);
148        }
149      }
150    }
151  }
152}
