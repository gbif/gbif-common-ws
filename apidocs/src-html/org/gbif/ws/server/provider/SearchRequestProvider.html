013package org.gbif.ws.server.provider;
014
015import org.gbif.api.model.common.search.SearchParameter;
016import org.gbif.api.model.common.search.SearchRequest;
017import org.gbif.api.util.SearchTypeValidator;
018import org.gbif.api.util.VocabularyUtils;
019
020import java.lang.reflect.Type;
021import java.util.List;
022import java.util.Map.Entry;
023
024import javax.ws.rs.core.Context;
025import javax.ws.rs.core.MultivaluedMap;
026
027import com.google.common.base.Strings;
028import com.google.common.base.Throwables;
029import com.google.common.collect.Lists;
030import com.sun.jersey.api.core.HttpContext;
031import com.sun.jersey.core.spi.component.ComponentContext;
032import com.sun.jersey.core.spi.component.ComponentScope;
033import com.sun.jersey.server.impl.inject.AbstractHttpContextInjectable;
034import com.sun.jersey.spi.inject.Injectable;
035import com.sun.jersey.spi.inject.InjectableProvider;
036
037import static org.gbif.ws.util.WebserviceParameter.PARAM_HIGHLIGHT;
038import static org.gbif.ws.util.WebserviceParameter.PARAM_QUERY_STRING;
039import static org.gbif.ws.util.WebserviceParameter.PARAM_SPELLCHECK;
040import static org.gbif.ws.util.WebserviceParameter.PARAM_SPELLCHECK_COUNT;
041
042/**
043 * Provider class that transforms a set of HTTP parameters into a SearchRequest class instance.
044 * This assumes the existence of the following parameters in the HTTP request:
045 * 'page_size', 'offset', 'q' and any of the search parameter enum member names case insensitively.
046 */
047public class SearchRequestProvider<RT extends SearchRequest<P>, P extends Enum<?> & SearchParameter>
048  extends AbstractHttpContextInjectable<RT> implements InjectableProvider<Context, Type> {
049
050  private final Class<P> searchParameterClass;
051  private final Class<RT> requestType;
052  private static final int NON_SPELL_CHECK_COUNT = -1;
053
054  public SearchRequestProvider(Class<RT> requestType, Class<P> searchParameterClass) {
055    this.requestType = requestType;
056    this.searchParameterClass = searchParameterClass;
057  }
058
059  /**
060   * Get an injectable.
061   *
062   * @param ic the injectable context
063   * @param context the annotation instance
064   * @param type the context instance
065   * @return an Injectable instance, otherwise null if an instance cannot
066   *         be created.
067   */
068  @Override
069  public Injectable<RT> getInjectable(ComponentContext ic, Context context, Type type) {
070    if (type.equals(requestType)) {
071      return this;
072    }
073    return null;
074  }
075
076  /**
077   * Get the scope of the injectable provider.
078   *
079   * @return the scope.
080   */
081  @Override
082  public ComponentScope getScope() {
083    return ComponentScope.PerRequest;
084  }
085
086  @Override
087  public RT getValue(HttpContext context) {
088    try {
089      RT req = requestType.newInstance();
090      return getSearchRequest(context, req);
091
092    } catch (InstantiationException | IllegalAccessException e) {
093      // should never happen
094      Throwables.propagate(e);
095    }
096    return null;
097  }
098
099  protected P findSearchParam(String name) {
100    try {
101      return (P) VocabularyUtils.lookupEnum(name, searchParameterClass);
102    } catch (IllegalArgumentException e) {
103      // we have all params here, not only the enum ones, so this is ok to end up here a few times
104    }
105    return null;
106  }
107
108  protected RT getSearchRequest(HttpContext context, RT searchRequest) {
109    searchRequest.copyPagingValues(PageableProvider.getPagingRequest(context));
110
111    final MultivaluedMap<String, String> params = context.getRequest().getQueryParameters();
112
113    getSearchRequestFromQueryParams(searchRequest, params);
114
115    return searchRequest;
116  }
117
118  /**
119   * Override this method for populating specific search/suggest requests
120   */
121  protected void getSearchRequestFromQueryParams(RT searchRequest, final MultivaluedMap<String, String> params) {
122    final String q = params.getFirst(PARAM_QUERY_STRING);
123    final String highlightValue = params.getFirst(PARAM_HIGHLIGHT);
124    final String spellCheck = params.getFirst(PARAM_SPELLCHECK);
125    final String spellCheckCount = params.getFirst(PARAM_SPELLCHECK_COUNT);
126
127    if (!Strings.isNullOrEmpty(q)) {
128      searchRequest.setQ(q);
129    }
130
131    if (!Strings.isNullOrEmpty(highlightValue)) {
132      searchRequest.setHighlight(Boolean.parseBoolean(highlightValue));
133    }
134
135    if (!Strings.isNullOrEmpty(spellCheck)) {
136      searchRequest.setSpellCheck(Boolean.parseBoolean(spellCheck));
137    }
138
139    if (!Strings.isNullOrEmpty(spellCheckCount)) {
140      searchRequest.setSpellCheckCount(Integer.parseInt(spellCheckCount));
141    } else {
142      searchRequest.setSpellCheckCount(NON_SPELL_CHECK_COUNT);
143    }
144
145    // find search parameter enum based filters
146    setSearchParams(searchRequest, params);
147  }
148
149  /**
150   * Removes all empty and null parameters from the list.
151   * Each value is trimmed(String.trim()) in order to remove all sizes of empty parameters.
152   */
153  private static List<String> removeEmptyParameters(List<String> parameters) {
154    List<String> cleanParameters = Lists.newArrayListWithCapacity(parameters.size());
155    for (String param : parameters) {
156      String cleanParam = Strings.nullToEmpty(param).trim();
157      if (!cleanParam.isEmpty()) {
158        cleanParameters.add(cleanParam);
159      }
160    }
161    return cleanParameters;
162  }
163
164  /**
165   * Iterates over the params map and adds to the search request the recognized parameters (i.e.: those that have a
166   * correspondent value in the P generic parameter).
167   * Empty (of all size) and null parameters are discarded.
168   */
169  private void setSearchParams(RT searchRequest, MultivaluedMap<String, String> params) {
170    for (Entry<String, List<String>> entry : params.entrySet()) {
171      P p = findSearchParam(entry.getKey());
172      if (p != null) {
173        for (String val : removeEmptyParameters(entry.getValue())) {
174          // validate value for certain types
175          SearchTypeValidator.validate(p, val);
176          searchRequest.addParameter(p, val);
177        }
178      }
179    }
180  }
181}
