001package org.gbif.ws.server.provider;
002
003import javax.ws.rs.core.Context;
004import javax.ws.rs.core.MediaType;
005import javax.ws.rs.core.Response;
006import javax.ws.rs.ext.ExceptionMapper;
007import javax.ws.rs.ext.Provider;
008import javax.ws.rs.ext.Providers;
009
010import com.google.inject.Singleton;
011import com.sun.jersey.api.container.ContainerException;
012
013/**
014 * Mapper to handle ContainerException which wrap exceptions thrown from injectable providers.
015 * This mapper looks for the root cause and delegates to the appropriate wrapper where there is
016 * one registered for the cause of the exception, otherwise it returns 500.
017 */
018@Provider
019@Singleton
020public class ContainerExceptionMapper implements ExceptionMapper<ContainerException> {
021  @Context
022  private final Providers providers;
023  
024  public ContainerExceptionMapper(@Context Providers providers) {
025    this.providers = providers;
026  }
027
028  @Override
029  public Response toResponse(ContainerException e) {
030    Throwable cause = e.getCause();
031    if (e.getCause() != null) {
032      @SuppressWarnings("unchecked")
033      ExceptionMapper<Throwable> em = (ExceptionMapper<Throwable>) providers.getExceptionMapper(cause.getClass());  
034      if (em != null) {
035        return em.toResponse(e.getCause());
036      }
037    }
038    
039    // no cause or not handled, then we are really hosed
040    return Response.status(Response.Status.INTERNAL_SERVER_ERROR).type(MediaType.TEXT_PLAIN).entity(e.getMessage()).build();    
041  }
042}
