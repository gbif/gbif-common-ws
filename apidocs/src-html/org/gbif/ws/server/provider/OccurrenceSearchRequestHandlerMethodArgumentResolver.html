001package org.gbif.ws.server.provider;
002
003import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
004import org.gbif.api.model.occurrence.search.OccurrenceSearchRequest;
005
006import java.util.Optional;
007
008import org.springframework.core.MethodParameter;
009import org.springframework.web.bind.support.WebDataBinderFactory;
010import org.springframework.web.context.request.NativeWebRequest;
011import org.springframework.web.context.request.WebRequest;
012import org.springframework.web.method.support.HandlerMethodArgumentResolver;
013import org.springframework.web.method.support.ModelAndViewContainer;
014
015public class OccurrenceSearchRequestHandlerMethodArgumentResolver
016    extends FacetedSearchRequestProvider<OccurrenceSearchRequest, OccurrenceSearchParameter>
017    implements HandlerMethodArgumentResolver {
018
019  private static final String MATCH_CASE_PARAM = "matchCase";
020
021  public OccurrenceSearchRequestHandlerMethodArgumentResolver() {
022    super(OccurrenceSearchRequest.class, OccurrenceSearchParameter.class);
023  }
024
025  @Override
026  public boolean supportsParameter(MethodParameter parameter) {
027    return OccurrenceSearchRequest.class.equals(parameter.getParameterType());
028  }
029
030  @Override
031  public Object resolveArgument(MethodParameter parameter, ModelAndViewContainer mavContainer, NativeWebRequest webRequest, WebDataBinderFactory binderFactory) {
032    return getValue(webRequest);
033  }
034
035  @Override
036  protected OccurrenceSearchRequest getSearchRequest(WebRequest webRequest, OccurrenceSearchRequest searchRequest) {
037    OccurrenceSearchRequest occurrenceSearchRequest = super.getSearchRequest(webRequest, searchRequest);
038    Optional.ofNullable(webRequest.getParameter(MATCH_CASE_PARAM))
039      .ifPresent(matchVerbatim -> occurrenceSearchRequest.setMatchCase(Boolean.parseBoolean(matchVerbatim)));
040    return occurrenceSearchRequest;
041  }
042}
