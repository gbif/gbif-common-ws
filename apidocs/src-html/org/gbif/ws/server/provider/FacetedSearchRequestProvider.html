013package org.gbif.ws.server.provider;
014
015import org.gbif.api.model.common.search.FacetedSearchRequest;
016import org.gbif.api.model.common.search.SearchParameter;
017
018import java.lang.reflect.Type;
019import java.util.List;
020import javax.ws.rs.core.Context;
021import javax.ws.rs.core.MultivaluedMap;
022
023import com.google.common.base.Strings;
024import com.sun.jersey.api.core.HttpContext;
025import com.sun.jersey.spi.inject.InjectableProvider;
026
027import static org.gbif.ws.util.WebserviceParameter.PARAM_FACET;
028import static org.gbif.ws.util.WebserviceParameter.PARAM_FACETS_ONLY;
029import static org.gbif.ws.util.WebserviceParameter.PARAM_FACET_MINCOUNT;
030import static org.gbif.ws.util.WebserviceParameter.PARAM_FACET_MULTISELECT;
031
032/**
033 * Provider class that transforms a set of HTTP parameters into a FacetedSearchRequest class instance.
034 * This assumes the existence of the following parameters in the HTTP request:
035 * 'page_size', 'offset', 'facet', 'q' and any of the search parameter enum member names case insensitively.
036 */
037public class FacetedSearchRequestProvider<RT extends FacetedSearchRequest<P>, P extends Enum<?> & SearchParameter>
038  extends SearchRequestProvider<RT, P> implements InjectableProvider<Context, Type> {
039
040  public FacetedSearchRequestProvider(Class<RT> requestType, Class<P> searchParameterClass) {
041    super(requestType, searchParameterClass);
042  }
043
044  @Override
045  protected RT getSearchRequest(HttpContext context, RT searchRequest) {
046    RT request = super.getSearchRequest(context, searchRequest);
047
048    final MultivaluedMap<String, String> params = context.getRequest().getQueryParameters();
049
050    final String facetsOnlyValue = params.getFirst(PARAM_FACETS_ONLY);
051    if (!Strings.isNullOrEmpty(facetsOnlyValue)) {
052      searchRequest.setFacetsOnly(Boolean.parseBoolean(facetsOnlyValue));
053    }
054
055    final String facetMultiSelectValue = params.getFirst(PARAM_FACET_MULTISELECT);
056    if (!Strings.isNullOrEmpty(facetMultiSelectValue)) {
057      searchRequest.setMultiSelectFacets(Boolean.parseBoolean(facetMultiSelectValue));
058    }
059
060    final String facetMinCountValue = params.getFirst(PARAM_FACET_MINCOUNT);
061    if (!Strings.isNullOrEmpty(facetMinCountValue)) {
062      searchRequest.setFacetMinCount(Integer.parseInt(facetMinCountValue));
063    }
064
065    final List<String> facets = params.get(PARAM_FACET);
066    if (facets != null && !facets.isEmpty()) {
067      for (String f : facets) {
068        P p = findSearchParam(f);
069        if (p != null) {
070          searchRequest.addFacets(p);
071        }
072      }
073    }
074
075    return request;
076  }
077}
