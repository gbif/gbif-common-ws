001package org.gbif.ws.server.provider;
002
003import com.google.common.base.Strings;
004import org.gbif.api.model.common.search.FacetedSearchRequest;
005import org.gbif.api.model.common.search.SearchParameter;
006import org.springframework.web.context.request.WebRequest;
007
008import java.util.Arrays;
009import java.util.Collections;
010import java.util.List;
011import java.util.Map;
012
013import static org.gbif.ws.util.CommonWsUtils.getFirst;
014import static org.gbif.ws.util.WebserviceParameter.PARAM_FACET;
015import static org.gbif.ws.util.WebserviceParameter.PARAM_FACET_LIMIT;
016import static org.gbif.ws.util.WebserviceParameter.PARAM_FACET_MINCOUNT;
017import static org.gbif.ws.util.WebserviceParameter.PARAM_FACET_MULTISELECT;
018import static org.gbif.ws.util.WebserviceParameter.PARAM_FACET_OFFSET;
019
020/**
021 * Provider class that transforms a set of HTTP parameters into a FacetedSearchRequest class instance.
022 * This assumes the existence of the following parameters in the HTTP request:
023 * 'page_size', 'offset', 'facet', 'q' and any of the search parameter enum member names case insensitively.
024 */
025public class FacetedSearchRequestProvider<RT extends FacetedSearchRequest<P>, P extends Enum<?> & SearchParameter>
026    extends SearchRequestProvider<RT, P> {
027
028  private static final int DEFAULT_FACET_LIMIT = 10;
029
030  public FacetedSearchRequestProvider(Class<RT> requestType, Class<P> searchParameterClass) {
031    super(requestType, searchParameterClass);
032  }
033
034  @Override
035  protected RT getSearchRequest(WebRequest webRequest, RT searchRequest) {
036    RT request = super.getSearchRequest(webRequest, searchRequest);
037
038    final Map<String, String[]> params = webRequest.getParameterMap();
039
040    final String facetMultiSelectValue = getFirstIgnoringCase(PARAM_FACET_MULTISELECT, params);
041    if (facetMultiSelectValue != null) {
042      searchRequest.setMultiSelectFacets(Boolean.parseBoolean(facetMultiSelectValue));
043    }
044
045    final String facetMinCountValue = getFirstIgnoringCase(PARAM_FACET_MINCOUNT, params);
046    if (facetMinCountValue != null) {
047      searchRequest.setFacetMinCount(Integer.parseInt(facetMinCountValue));
048    }
049
050    final String facetLimit = getFirstIgnoringCase(PARAM_FACET_LIMIT, params);
051    if (facetLimit != null) {
052      searchRequest.setFacetLimit(Integer.parseInt(facetLimit));
053    }
054
055    final String facetOffset = getFirstIgnoringCase(PARAM_FACET_OFFSET, params);
056    if (facetOffset != null) {
057      searchRequest.setFacetOffset(Integer.parseInt(facetOffset));
058    }
059
060    final List<String> facets = params.get(PARAM_FACET) != null ? Arrays.asList(params.get(PARAM_FACET)) : Collections.emptyList();
061    if (!facets.isEmpty()) {
062      for (String f : facets) {
063        P p = findSearchParam(f);
064        if (p != null) {
065          searchRequest.addFacets(p);
066          String pFacetOffset = getFirstIgnoringCase(f + '.' + PARAM_FACET_OFFSET, params);
067          String pFacetLimit = getFirstIgnoringCase(f + '.' + PARAM_FACET_LIMIT, params);
068          if (pFacetLimit != null) {
069            if (pFacetOffset != null) {
070              searchRequest.addFacetPage(p, Integer.parseInt(pFacetOffset), Integer.parseInt(pFacetLimit));
071            } else {
072              searchRequest.addFacetPage(p, 0, Integer.parseInt(pFacetLimit));
073            }
074          } else if (pFacetOffset != null) {
075            searchRequest.addFacetPage(p, Integer.parseInt(pFacetOffset), DEFAULT_FACET_LIMIT);
076          }
077        }
078      }
079    }
080
081    return request;
082  }
083
084  /**
085   * Get the first parameter value, the parameter is searched in a case-insensitive manner.
086   * First tries with the exact match, then the lowercase and finally the uppercase value of the parameter.
087   */
088  private static String getFirstIgnoringCase(String parameter, Map<String, String[]> params) {
089    String value = getFirst(params, parameter);
090    if (!Strings.isNullOrEmpty(value)) {
091      return value;
092    }
093    value = getFirst(params, parameter.toLowerCase());
094    if (!Strings.isNullOrEmpty(value)) {
095      return value;
096    }
097    value = getFirst(params, parameter.toUpperCase());
098    if (!Strings.isNullOrEmpty(value)) {
099      return value;
100    }
101    return null;
102  }
103}
