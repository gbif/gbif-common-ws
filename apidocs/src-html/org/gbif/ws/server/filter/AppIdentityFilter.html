001package org.gbif.ws.server.filter;
002
003import org.gbif.api.model.common.AppPrincipal;
004import org.gbif.api.model.common.ExtendedPrincipal;
005import org.gbif.api.vocabulary.AppRole;
006import org.gbif.ws.security.GbifAuthService;
007
008import java.security.Principal;
009import java.util.ArrayList;
010import java.util.List;
011import javax.annotation.Nullable;
012import javax.validation.constraints.NotNull;
013import javax.ws.rs.WebApplicationException;
014import javax.ws.rs.core.Response;
015import javax.ws.rs.core.SecurityContext;
016
017import com.google.inject.Inject;
018import com.google.inject.name.Named;
019import com.sun.jersey.spi.container.ContainerRequest;
020import com.sun.jersey.spi.container.ContainerRequestFilter;
021import org.apache.commons.lang3.StringUtils;
022import org.slf4j.Logger;
023import org.slf4j.LoggerFactory;
024
025/**
026 * A filter that allows an application to identify itself as an application (as opposed to an application
027 * impersonating a user).
028 * In order to identify itself an application shall provide its appKey in the header x-gbif-user and sign the request
029 * accordingly.
030 * If the application can be authenticated AND its appKey is in the whitelist, the {@link Principal#getName()} will return the appKey and the
031 * role {@link AppRole#APP} will be assigned to it.
032 *
033 * We use an appKey whitelist to control which app should be allowed to have the {@link AppRole#APP} while letting
034 * the user impersonation available in {@link IdentityFilter}. If at some point multiple {@link AppRole} should be
035 * supported the whitelist should simply be changed for something more structured.
036 *
037 * This filter must run AFTER {@link IdentityFilter} if user impersonation using appKey is required.
038 * This filter will be skipped if the {@link ContainerRequest} already has a {@link Principal} attached.
039 * This filter operates on {@link GbifAuthService#GBIF_SCHEME} only.
040 * If the appKeyWhitelist list is not provided no apps will be authenticated by this filter.
041 *
042 */
043public class AppIdentityFilter implements ContainerRequestFilter {
044
045  public static final  String APPKEYS_WHITELIST = "identity.appkeys.whitelist";
046
047  //FIXME should probably have its own scheme but that would requires to change {@link GbifAuthService)
048  private static final String GBIF_SCHEME_PREFIX = GbifAuthService.GBIF_SCHEME + " ";
049  private static final Logger LOG = LoggerFactory.getLogger(AppIdentityFilter.class);
050
051  private final GbifAuthService authService;
052  private final List<String> appKeyWhitelist;
053
054  @Inject
055  public AppIdentityFilter(@NotNull GbifAuthService authService, @Nullable @Named(APPKEYS_WHITELIST) List<String> appKeyWhitelist) {
056    this.authService = authService;
057    //defensive copy or creation
058    this.appKeyWhitelist = appKeyWhitelist != null ? new ArrayList<>(appKeyWhitelist) : new ArrayList<>();
059  }
060
061  @Override
062  public ContainerRequest filter(final ContainerRequest containerRequest) {
063
064    // Only try if no user principal is already there
065    if (containerRequest.getSecurityContext() != null && containerRequest.getUserPrincipal() != null) {
066      return containerRequest;
067    }
068
069    String authorization = containerRequest.getHeaderValue(ContainerRequest.AUTHORIZATION);
070    if (StringUtils.startsWith(authorization, GBIF_SCHEME_PREFIX)) {
071      if (!authService.isValidRequest(containerRequest)) {
072        LOG.warn("Invalid GBIF authenticated request");
073        throw new WebApplicationException(Response.Status.UNAUTHORIZED);
074      }
075
076      String username = containerRequest.getHeaderValue(GbifAuthService.HEADER_GBIF_USER);
077      String appKey = GbifAuthService.getAppKeyFromRequest(containerRequest::getHeaderValue);
078
079      //check if it's an app by ensuring the appkey used to sign the request is the one used as x-gbif-user
080      if (StringUtils.equals(appKey, username) && appKeyWhitelist.contains(appKey)) {
081        containerRequest.setSecurityContext(new SecurityContext() {
082          private final ExtendedPrincipal principal = new AppPrincipal(appKey, AppRole.APP.name());
083
084          @Override
085          public Principal getUserPrincipal() {
086            return principal;
087          }
088
089          @Override
090          public boolean isUserInRole(String s) {
091            return principal.hasRole(s);
092          }
093
094          @Override
095          public boolean isSecure() {
096            return containerRequest.isSecure();
097          }
098
099          @Override
100          public String getAuthenticationScheme() {
101            return GbifAuthService.GBIF_SCHEME;
102          }
103        });
104      }
105    }
106    return containerRequest;
107  }
108}
