046import org.springframework.stereotype.Component;
047import org.springframework.web.filter.OncePerRequestFilter;

049/**
050 * A filter that allows an application to identify itself as an application (as opposed to an
051 * application impersonating a user). In order to identify itself an application shall provide its
052 * appKey in the header x-gbif-user and sign the request accordingly. If the application can be
053 * authenticated AND its appKey is in the whitelist, the {@link Principal#getName()} will return the
054 * appKey and the role {@link AppRole#APP} will be assigned to it.
055 *
056 * <p>We use an appKey whitelist to control which app should be allowed to have the {@link
057 * AppRole#APP} while letting the user impersonation available in {@link IdentityFilter}. If at some
058 * point multiple {@link AppRole} should be supported the whitelist should simply be changed for
059 * something more structured.
060 *
061 * <p>This filter must run AFTER {@link IdentityFilter} if user impersonation using appKey is
062 * required. This filter will be skipped if the request already has a {@link Principal} attached.
063 * This filter operates on {@link SecurityConstants#GBIF_SCHEME} only. If the appKeyWhitelist list
064 * is not provided no apps will be authenticated by this filter.
065 */
066@Component
067public class AppIdentityFilter extends OncePerRequestFilter {
068
069  private static final Logger LOG = LoggerFactory.getLogger(AppIdentityFilter.class);
070
071  private final GbifAuthService authService;
072  private final List<String> appKeyWhitelist;
073
074  public AppIdentityFilter(
075      @NotNull GbifAuthService authService, AppkeysConfigurationProperties appkeysConfiguration) {
076    this.authService = authService;
077    // defensive copy or creation
078    this.appKeyWhitelist =
079        appkeysConfiguration.getWhitelist() != null
080            ? new ArrayList<>(appkeysConfiguration.getWhitelist())
081            : new ArrayList<>();
082  }
083
084  @Override
085  protected void doFilterInternal(
086      HttpServletRequest request, HttpServletResponse response, FilterChain filterChain)
087      throws ServletException, IOException {
088    final Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
089
090    // Only try if no user principal is already there
091    if (authentication == null
092        || authentication.getPrincipal() == null
093        || authentication.getPrincipal() instanceof AnonymousUserPrincipal) {
094      String authorization = request.getHeader(HttpHeaders.AUTHORIZATION);
095      if (StringUtils.startsWith(authorization, SecurityConstants.GBIF_SCHEME_PREFIX)) {
096        if (authService.isValidRequest(
097            request instanceof GbifHttpServletRequestWrapper
098                ? (GbifHttpServletRequestWrapper) request
099                : new GbifHttpServletRequestWrapper(request))) {
100          String username = request.getHeader(SecurityConstants.HEADER_GBIF_USER);
101          String appKey = GbifAuthUtils.getAppKeyFromRequest(authorization);
102
103          // check if it's an app by ensuring the appkey used to sign the request is the one used as
104          // x-gbif-user
105          if (StringUtils.equals(appKey, username) && appKeyWhitelist.contains(appKey)) {
106            final List<SimpleGrantedAuthority> authorities =
107                Collections.singletonList(new SimpleGrantedAuthority(AppRole.APP.name()));
108            final AppPrincipal principal = new AppPrincipal(appKey, authorities);
109            final Authentication newAuthentication =
110                new GbifAuthenticationToken(principal, SecurityConstants.GBIF_SCHEME, authorities);
111
112            SecurityContextHolder.getContext().setAuthentication(newAuthentication);
113          }
114        } else {
115          LOG.warn("Invalid GBIF authenticated request");
116          response.setStatus(HttpStatus.UNAUTHORIZED.value());
117        }
118      }
119    }
120
121    filterChain.doFilter(request, response);
122  }
123}
