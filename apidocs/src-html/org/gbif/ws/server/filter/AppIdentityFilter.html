001package org.gbif.ws.server.filter;
002
003import java.io.IOException;
004import java.security.Principal;
005import java.util.ArrayList;
006import java.util.Collections;
007import java.util.List;
008import javax.servlet.FilterChain;
009import javax.servlet.ServletException;
010import javax.servlet.http.HttpServletRequest;
011import javax.servlet.http.HttpServletResponse;
012import javax.validation.constraints.NotNull;
013import org.apache.commons.lang3.StringUtils;
014import org.gbif.api.vocabulary.AppRole;
015import org.gbif.ws.security.AnonymousUserPrincipal;
016import org.gbif.ws.security.AppPrincipal;
017import org.gbif.ws.security.AppkeysConfigurationProperties;
018import org.gbif.ws.security.GbifAuthService;
019import org.gbif.ws.security.GbifAuthUtils;
020import org.gbif.ws.security.GbifAuthenticationToken;
021import org.gbif.ws.server.GbifHttpServletRequestWrapper;
022import org.gbif.ws.util.SecurityConstants;
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025import org.springframework.http.HttpHeaders;
026import org.springframework.http.HttpStatus;
027import org.springframework.security.core.Authentication;
028import org.springframework.security.core.authority.SimpleGrantedAuthority;
029import org.springframework.security.core.context.SecurityContextHolder;
030import org.springframework.stereotype.Component;
031import org.springframework.web.filter.OncePerRequestFilter;
032
033/**
034 * A filter that allows an application to identify itself as an application (as opposed to an
035 * application impersonating a user). In order to identify itself an application shall provide its
036 * appKey in the header x-gbif-user and sign the request accordingly. If the application can be
037 * authenticated AND its appKey is in the whitelist, the {@link Principal#getName()} will return the
038 * appKey and the role {@link AppRole#APP} will be assigned to it.
039 *
040 * <p>We use an appKey whitelist to control which app should be allowed to have the {@link
041 * AppRole#APP} while letting the user impersonation available in {@link IdentityFilter}. If at some
042 * point multiple {@link AppRole} should be supported the whitelist should simply be changed for
043 * something more structured.
044 *
045 * <p>This filter must run AFTER {@link IdentityFilter} if user impersonation using appKey is
046 * required. This filter will be skipped if the request already has a {@link Principal} attached.
047 * This filter operates on {@link SecurityConstants#GBIF_SCHEME} only. If the appKeyWhitelist list
048 * is not provided no apps will be authenticated by this filter.
049 */
050@Component
051public class AppIdentityFilter extends OncePerRequestFilter {
052
053  private static final Logger LOG = LoggerFactory.getLogger(AppIdentityFilter.class);
054
055  private final GbifAuthService authService;
056  private final List<String> appKeyWhitelist;
057
058  public AppIdentityFilter(
059      @NotNull GbifAuthService authService, AppkeysConfigurationProperties appkeysConfiguration) {
060    this.authService = authService;
061    // defensive copy or creation
062    this.appKeyWhitelist =
063        appkeysConfiguration.getWhitelist() != null
064            ? new ArrayList<>(appkeysConfiguration.getWhitelist())
065            : new ArrayList<>();
066  }
067
068  @Override
069  protected void doFilterInternal(
070      HttpServletRequest request, HttpServletResponse response, FilterChain filterChain)
071      throws ServletException, IOException {
072    final Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
073
074    // Only try if no user principal is already there
075    if (authentication == null
076        || authentication.getPrincipal() == null
077        || authentication.getPrincipal() instanceof AnonymousUserPrincipal) {
078      String authorization = request.getHeader(HttpHeaders.AUTHORIZATION);
079      if (StringUtils.startsWith(authorization, SecurityConstants.GBIF_SCHEME_PREFIX)) {
080        if (authService.isValidRequest(
081            request instanceof GbifHttpServletRequestWrapper
082                ? (GbifHttpServletRequestWrapper) request
083                : new GbifHttpServletRequestWrapper(request))) {
084          String username = request.getHeader(SecurityConstants.HEADER_GBIF_USER);
085          String appKey = GbifAuthUtils.getAppKeyFromRequest(authorization);
086
087          // check if it's an app by ensuring the appkey used to sign the request is the one used as
088          // x-gbif-user
089          if (StringUtils.equals(appKey, username) && appKeyWhitelist.contains(appKey)) {
090            final List<SimpleGrantedAuthority> authorities =
091                Collections.singletonList(new SimpleGrantedAuthority(AppRole.APP.name()));
092            final AppPrincipal principal = new AppPrincipal(appKey, authorities);
093            final Authentication newAuthentication =
094                new GbifAuthenticationToken(principal, SecurityConstants.GBIF_SCHEME, authorities);
095
096            SecurityContextHolder.getContext().setAuthentication(newAuthentication);
097          }
098        } else {
099          LOG.warn("Invalid GBIF authenticated request");
100          response.setStatus(HttpStatus.UNAUTHORIZED.value());
101        }
102      }
103    }
104
105    filterChain.doFilter(request, response);
106  }
107}
