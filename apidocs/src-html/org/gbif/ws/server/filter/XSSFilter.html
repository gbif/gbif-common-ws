001package org.gbif.ws.server.filter;
002
003import org.gbif.ws.util.XSSUtil;
004
005import java.io.IOException;
006import java.util.Enumeration;
007import javax.servlet.Filter;
008import javax.servlet.FilterChain;
009import javax.servlet.FilterConfig;
010import javax.servlet.ServletException;
011import javax.servlet.ServletRequest;
012import javax.servlet.ServletResponse;
013import javax.servlet.http.HttpServletRequest;
014import javax.servlet.http.HttpServletResponse;
015
016import com.google.inject.Singleton;
017
018/**
019 * Request filter that detects XSS in paramater, header or querystring values and responds with a 400 bad request in such cases.
020 */
021@Singleton
022public class XSSFilter implements Filter {
023
024  @Override
025  public void init(FilterConfig filterConfig) throws ServletException {
026  }
027
028  @Override
029  public void destroy() {
030  }
031
032  @Override
033  public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain) throws IOException, ServletException {
034    if (request instanceof HttpServletRequest) {
035      HttpServletRequest httpReq = (HttpServletRequest) request;
036      // test for xss in headers
037      Enumeration<String> headers =  httpReq.getHeaderNames();
038      while(headers.hasMoreElements()) {
039        if (XSSUtil.containsXSS(httpReq.getHeader(headers.nextElement()))) {
040          respond400(response);
041        }
042      }
043      // test for xss in querystring
044      // note: servlet container does not decode querystring when creating HttpServletRequest object
045      if (XSSUtil.containsXSS(httpReq.getQueryString())) {
046        respond400(response);
047      }
048      // test for xss in parameters
049      // note: servlet container decodes parameter strings when creating HttpServletRequest object
050      Enumeration<String> params =  httpReq.getParameterNames();
051      while(params.hasMoreElements()) {
052        if (XSSUtil.containsXSS(httpReq.getParameter(params.nextElement()))) {
053          respond400(response);
054        }
055      }
056    }
057    chain.doFilter(request, response);
058  }
059
060  /**
061   * Sends a 400 error response, provided the response hasn't already been committed.
062   *
063   * @param response response
064   *
065   * @throws IOException if there is a problem committing response
066   */
067  private void respond400(ServletResponse response) throws IOException {
068    HttpServletResponse resp = (HttpServletResponse) response;
069    if (!resp.isCommitted()) {
070      resp.sendError(400);
071    }
072  }
073
074}
