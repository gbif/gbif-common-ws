001package org.gbif.ws.server.filter;
002
003import java.net.HttpURLConnection;
004import java.net.URI;
005import java.util.UUID;
006import javax.ws.rs.core.Response;
007
008import com.sun.jersey.spi.container.ContainerRequest;
009import com.sun.jersey.spi.container.ContainerResponse;
010import com.sun.jersey.spi.container.ContainerResponseFilter;
011
012/**
013 * Filter that updates http headers when a new resource is successfully created via a POST request.
014 * The following headers are added or replaced if they existed:
015 * <ul>
016 *   <li>Http response code 201</li>
017 *   <li>Location header is set accordingly based on returned key</li>
018 * </ul>
019 */
020public class CreatedResponseFilter implements ContainerResponseFilter {
021
022  @Override
023  public ContainerResponse filter(ContainerRequest request, ContainerResponse response) {
024
025    if (request.getMethod() != null && request.getMethod().equalsIgnoreCase("post") && response.getStatusType() != null
026        && response.getStatusType().getFamily() == Response.Status.Family.SUCCESSFUL) {
027
028      response.setStatus(HttpURLConnection.HTTP_CREATED);
029      // if response contains the key, also set Location
030      if (response.getEntity() != null) {
031        Object key = response.getEntity();
032        // we use POSTs also for non Create method which can return large objects, e.g. a list of parsed names
033        // only set the location header if the object is one of the following simple primary key data types
034        if (key instanceof Number || key instanceof UUID || key instanceof String) {
035          // allow POSTing to resource with or without trailing slash
036          URI location = request.getRequestUriBuilder().path(key.toString()).build();
037          response.getHttpHeaders().putSingle("Location", location.toString());
038        }
039
040      }
041    }
042
043    return response;
044  }
045}
