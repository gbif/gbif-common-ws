001package org.gbif.ws.server.filter;
002
003import java.net.HttpURLConnection;
004import java.net.URI;
005import java.util.UUID;
006import javax.ws.rs.core.Response;
007
008import com.sun.jersey.spi.container.ContainerRequest;
009import com.sun.jersey.spi.container.ContainerResponse;
010import com.sun.jersey.spi.container.ContainerResponseFilter;
011
012/**
013 * Filter that updates http headers when a new resource is successfully created via a POST request unless
014 * the response returns 204 No Content.
015 *
016 * The following headers are added or replaced if they existed:
017 * <ul>
018 *   <li>Http response code 201</li>
019 *   <li>Location header is set accordingly based on returned key</li>
020 * </ul>
021 */
022public class CreatedResponseFilter implements ContainerResponseFilter {
023
024  @Override
025  public ContainerResponse filter(ContainerRequest request, ContainerResponse response) {
026
027    if (request.getMethod() != null && "post".equalsIgnoreCase(request.getMethod())
028            && response.getStatusType() != null
029            && !Response.Status.NO_CONTENT.equals(response.getStatusType())
030            && response.getStatusType().getFamily() == Response.Status.Family.SUCCESSFUL) {
031
032      response.setStatus(HttpURLConnection.HTTP_CREATED);
033      // if response contains the key, also set Location
034      if (response.getEntity() != null) {
035        Object key = response.getEntity();
036        // we use POSTs also for non Create method which can return large objects, e.g. a list of parsed names
037        // only set the location header if the object is one of the following simple primary key data types
038        if (key instanceof Number || key instanceof UUID || key instanceof String) {
039          // allow POSTing to resource with or without trailing slash
040          URI location = request.getRequestUriBuilder().path(key.toString()).build();
041          response.getHttpHeaders().putSingle("Location", location.toString());
042        }
043
044      }
045    }
046
047    return response;
048  }
049}
