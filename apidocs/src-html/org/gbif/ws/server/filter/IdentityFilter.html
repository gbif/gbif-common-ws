001package org.gbif.ws.server.filter;

003import org.gbif.api.model.common.ExtendedPrincipal;
004import org.gbif.api.model.common.GbifUser;
005import org.gbif.api.model.common.GbifUserPrincipal;
006import org.gbif.api.model.common.User;
007import org.gbif.api.service.common.IdentityAccessService;
008import org.gbif.ws.security.GbifAuthService;
009
010import java.security.Principal;
011import java.util.Objects;
012import java.util.Optional;
013import java.util.UUID;
014import java.util.regex.Pattern;
015import javax.annotation.Nullable;
016import javax.validation.constraints.NotNull;
017import javax.ws.rs.WebApplicationException;
018import javax.ws.rs.core.Response;
019import javax.ws.rs.core.SecurityContext;
020
021import com.google.common.base.Strings;
022import com.google.inject.Inject;
023import com.sun.jersey.core.util.Base64;
024import com.sun.jersey.spi.container.ContainerRequest;
025import com.sun.jersey.spi.container.ContainerRequestFilter;
026import org.slf4j.Logger;
027import org.slf4j.LoggerFactory;

029/**
030 *
031 * Replacement for {@link AuthFilter}.
032 *
033 * Server filter that looks for a http BasicAuthentication with user accounts based on a {@link IdentityAccessService}
034 * or GBIF trusted application schema to impersonate a user and populates the security context.
035 *
036 * As we have another custom authorization filter in the registry that understands a registry internal authentication,
037 * all Basic authentication requests that have a UUID as the username are simply passed through and passwords are not
038 * evaluated.
039 */
040public class IdentityFilter implements ContainerRequestFilter {
041
042  private static class Authorizer implements SecurityContext {
043
044    private final ExtendedPrincipal principal;
045    private final String authenticationScheme;
046    private final boolean isSecure;
047
048    private Authorizer(ExtendedPrincipal principal, String authenticationScheme, boolean isSecure) {
049      this.principal = principal;
050      this.authenticationScheme = authenticationScheme;
051      this.isSecure = isSecure;
052    }
053
054    /**
055     * Get an {@link Authorizer} for a specific {@link User}.
056     *
057     * @param user
058     * @param authenticationScheme
059     * @param isSecure
060     *
061     * @return
062     */
063    static Authorizer getAuthorizer(GbifUser user, String authenticationScheme, boolean isSecure) {
064      return new Authorizer(new GbifUserPrincipal(user), authenticationScheme, isSecure);
065    }
066
067    /**
068     * Get an anonymous {@link Authorizer}.
069     * Anonymous users do not have {@link Principal}.
070     *
071     * @param isSecure
072     *
073     * @return
074     */
075    static Authorizer getAnonymous(boolean isSecure) {
076      return new Authorizer(null, "", isSecure);
077    }
078
079    @Override
080    public String getAuthenticationScheme() {
081      return authenticationScheme;
082    }
083
084    @Override
085    public Principal getUserPrincipal() {
086      return principal;
087    }
088
089    @Override
090    public boolean isSecure() {
091      return isSecure;
092    }
093
094    @Override
095    public boolean isUserInRole(String role) {
096      return principal != null && principal.hasRole(role);
097    }
098  }
099
100  private static final Logger LOG = LoggerFactory.getLogger(IdentityFilter.class);
101
102  private static final Pattern COLON_PATTERN = Pattern.compile(":");
103  private final IdentityAccessService identityAccessService;
104  private final GbifAuthService authService;
105
106  private static final String GBIF_SCHEME_PREFIX = GbifAuthService.GBIF_SCHEME + " ";
107  private static final String BASIC_SCHEME_PREFIX = "Basic ";
108
109  /**
110   * IdentityFilter constructor
111   * In case {@link GbifAuthService} is not provided, this class will reject all authentications
112   * on the GBIF scheme prefix.
113   *
114   * @param identityAccessService
115   * @param authService     nullable GbifAuthService
116   */
117  @Inject
118  public IdentityFilter(@NotNull IdentityAccessService identityAccessService, @Nullable GbifAuthService authService) {
119    Objects.requireNonNull(identityAccessService, "identityAccessService shall be provided");
120    this.identityAccessService = identityAccessService;
121    this.authService = authService;
122  }
123
124  @Override
125  public ContainerRequest filter(ContainerRequest request) {
126
127    // authenticates the HTTP method, but ignores legacy UUID user names
128    Authorizer authorizer = authenticate(request);
129
130    if (authorizer == null) {
131      // default is the anonymous user
132      authorizer = Authorizer.getAnonymous(request.isSecure());
133    }
134    request.setSecurityContext(authorizer);
135    return request;
136  }
137
138  private Authorizer authenticate(ContainerRequest request) {
139    // Extract authentication credentials
140    String authentication = request.getHeaderValue(ContainerRequest.AUTHORIZATION);
141    if (authentication != null) {
142      if (authentication.startsWith(BASIC_SCHEME_PREFIX)) {
143        return basicAuthentication(authentication.substring(BASIC_SCHEME_PREFIX.length()), request.isSecure());
144      } else if (authentication.startsWith(GBIF_SCHEME_PREFIX)) {
145        return gbifAuthentication(request);
146      }
147    }
148    return Authorizer.getAnonymous(request.isSecure());
149  }
150
151  private Authorizer basicAuthentication(String authentication, boolean isSecure) {
152    String[] values = COLON_PATTERN.split(Base64.base64Decode(authentication));
153    if (values.length < 2) {
154      LOG.warn("Invalid syntax for username and password: {}", authentication);
155      throw new WebApplicationException(Response.Status.BAD_REQUEST);
156    }
157
158    String username = values[0];
159    String password = values[1];
160    if (username == null || password == null) {
161      LOG.warn("Missing basic authentication username or password: {}", authentication);
162      throw new WebApplicationException(Response.Status.BAD_REQUEST);
163    }
164
165    // ignore usernames which are UUIDs - these are registry legacy IPT calls and handled by a special security filter
166    try {
167      UUID.fromString(username);
168      return null;
169    } catch (IllegalArgumentException e) {
170      // no UUID, continue with regular drupal authentication
171    }
172
173    GbifUser user = identityAccessService.authenticate(username, password);
174    if (user == null) {
175      throw new WebApplicationException(Response.Status.UNAUTHORIZED);
176    }
177
178    LOG.debug("Authenticating user {} via scheme {}", username, SecurityContext.BASIC_AUTH);
179    return Authorizer.getAuthorizer(user, SecurityContext.BASIC_AUTH, isSecure);
180  }
181
182  private Authorizer gbifAuthentication(final ContainerRequest request) {
183    String username = request.getHeaderValue(GbifAuthService.HEADER_GBIF_USER);
184    if (Strings.isNullOrEmpty(username)) {
185      LOG.warn("Missing gbif username header {}", GbifAuthService.HEADER_GBIF_USER);
186      throw new WebApplicationException(Response.Status.BAD_REQUEST);
187    }
188    if (authService == null) {
189      LOG.warn("No GbifAuthService defined.");
190      throw new WebApplicationException(Response.Status.UNAUTHORIZED);
191    }
192    if (!authService.isValidRequest(request)) {
193      LOG.warn("Invalid GBIF authenticated request");
194      throw new WebApplicationException(Response.Status.UNAUTHORIZED);
195    }
196
197    LOG.debug("Authenticating user {} via scheme {}", username, GbifAuthService.GBIF_SCHEME);
198    if (identityAccessService == null) {
199      LOG.debug("No identityService configured! No roles assigned, using anonymous user instead.");
200      return Authorizer.getAnonymous(request.isSecure());
201    }
202
203    //check if we have a request that impersonates a user
204    GbifUser user = identityAccessService.get(username);
205    //Note: using an Anonymous Authorizer is probably not the best thing to do here
206    //we should consider simply return null to let another filter handle it
207    return user == null ? Authorizer.getAnonymous(request.isSecure())
208            : Authorizer.getAuthorizer(user, GbifAuthService.GBIF_SCHEME,
209            Optional.ofNullable(request.getSecurityContext()).map(sc -> request.isSecure()).orElse(Boolean.FALSE));
210  }
