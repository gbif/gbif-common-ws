001package org.gbif.ws.server.filter;
002
003import java.io.IOException;
004import java.util.Objects;
005import javax.servlet.FilterChain;
006import javax.servlet.ServletException;
007import javax.servlet.http.HttpServletRequest;
008import javax.servlet.http.HttpServletResponse;
009import org.gbif.api.service.common.IdentityAccessService;
010import org.gbif.ws.WebApplicationException;
011import org.gbif.ws.security.GbifAuthenticationManager;
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014import org.springframework.security.core.Authentication;
015import org.springframework.security.core.context.SecurityContextHolder;
016import org.springframework.stereotype.Component;
017import org.springframework.web.filter.OncePerRequestFilter;
018
019/**
020 * Override a built-in spring filter because of legacy behaviour.
021 *
022 * <p>Replacement for AuthFilter (legacy gbif-common-ws).
023 *
024 * <p>Server filter that looks for a http BasicAuthentication with user accounts based on a {@link
025 * IdentityAccessService} or GBIF trusted application schema to impersonate a user and populates the
026 * security context.
027 *
028 * <p>As we have another custom authorization filter in the registry that understands a registry
029 * internal authentication, all Basic authentication requests that have a UUID as the username are
030 * simply passed through and passwords are not evaluated.
031 */
032@Component
033public class IdentityFilter extends OncePerRequestFilter {
034
035  private static final Logger LOG = LoggerFactory.getLogger(IdentityFilter.class);
036
037  private GbifAuthenticationManager authenticationManager;
038
039  public IdentityFilter(GbifAuthenticationManager authenticationManager) {
040    this.authenticationManager = authenticationManager;
041  }
042
043  @Override
044  protected void doFilterInternal(
045      HttpServletRequest request, HttpServletResponse response, FilterChain filterChain)
046      throws ServletException, IOException {
047    Objects.requireNonNull(request, "Can't filter null request");
048    Objects.requireNonNull(response, "Can't filter null response");
049
050    // authenticates the HTTP method, but ignores legacy UUID user names
051    try {
052      final Authentication authentication = authenticationManager.authenticate(request);
053      SecurityContextHolder.getContext().setAuthentication(authentication);
054      filterChain.doFilter(request, response);
055    } catch (final WebApplicationException e) {
056      LOG.debug("Exception while authentication in IdentityFilter: {}", e.getMessage());
057      response.setStatus(e.getStatus());
058    }
059  }
060}
