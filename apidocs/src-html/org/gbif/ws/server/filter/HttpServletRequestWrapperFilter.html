001package org.gbif.ws.server.filter;
002
003import java.io.IOException;
004import javax.servlet.FilterChain;
005import javax.servlet.ServletException;
006import javax.servlet.http.HttpServletRequest;
007import javax.servlet.http.HttpServletResponse;
008import org.gbif.ws.server.GbifHttpServletRequestWrapper;
009import org.springframework.stereotype.Component;
010import org.springframework.web.filter.OncePerRequestFilter;
011
012@Component
013public class HttpServletRequestWrapperFilter extends OncePerRequestFilter {
014
015  @Override
016  protected void doFilterInternal(
017      HttpServletRequest request, HttpServletResponse response, FilterChain filterChain)
018      throws ServletException, IOException {
019    final GbifHttpServletRequestWrapper requestWrapper =
020        request instanceof GbifHttpServletRequestWrapper
021            ? (GbifHttpServletRequestWrapper) request
022            : new GbifHttpServletRequestWrapper(request);
023
024    filterChain.doFilter(requestWrapper, response);
025  }
026}
