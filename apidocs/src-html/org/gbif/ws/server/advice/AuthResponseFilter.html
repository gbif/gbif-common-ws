001package org.gbif.ws.server.advice;
002
003import org.springframework.core.MethodParameter;
004import org.springframework.http.HttpStatus;
005import org.springframework.http.MediaType;
006import org.springframework.http.converter.HttpMessageConverter;
007import org.springframework.http.server.ServerHttpRequest;
008import org.springframework.http.server.ServerHttpResponse;
009import org.springframework.http.server.ServletServerHttpResponse;
010import org.springframework.web.bind.annotation.ControllerAdvice;
011import org.springframework.web.servlet.mvc.method.annotation.ResponseBodyAdvice;
012
013import javax.validation.constraints.NotNull;
014
015/**
016 * Response filter that adds a WWW-Authenticate header if response status is Unauthenticated,
017 * indicating a Basic Authentication scheme to be used.
018 */
019@ControllerAdvice
020public class AuthResponseFilter implements ResponseBodyAdvice<Object> {
021
022  @Override
023  public boolean supports(@NotNull MethodParameter returnType,
024                          @NotNull Class<? extends HttpMessageConverter<?>> converterType) {
025    return true;
026  }
027
028  @Override
029  public Object beforeBodyWrite(Object body, @NotNull MethodParameter returnType,
030                                @NotNull MediaType selectedContentType,
031                                @NotNull Class<? extends HttpMessageConverter<?>> selectedConverterType,
032                                @NotNull ServerHttpRequest request,
033                                @NotNull ServerHttpResponse response) {
034    if (response instanceof ServletServerHttpResponse) {
035      int responseStatusCode = ((ServletServerHttpResponse) response).getServletResponse().getStatus();
036
037      if (responseStatusCode == HttpStatus.UNAUTHORIZED.value()) {
038        // TODO: 10/01/2020 realm GBIF only?
039        response.getHeaders().add("WWW-Authenticate", "Basic realm=\"GBIF\"");
040      }
041    }
042
043    return body;
044  }
045}
