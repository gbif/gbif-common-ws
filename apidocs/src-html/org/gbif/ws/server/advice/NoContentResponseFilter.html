001package org.gbif.ws.server.advice;
002
003import org.springframework.core.MethodParameter;
004import org.springframework.http.HttpStatus;
005import org.springframework.http.MediaType;
006import org.springframework.http.converter.HttpMessageConverter;
007import org.springframework.http.server.ServerHttpRequest;
008import org.springframework.http.server.ServerHttpResponse;
009import org.springframework.http.server.ServletServerHttpResponse;
010import org.springframework.web.bind.annotation.ControllerAdvice;
011import org.springframework.web.servlet.mvc.method.annotation.ResponseBodyAdvice;
012
013/**
014 * Response filter that converts response status to 204 NO_CONTENT if response body is null.
015 */
016@SuppressWarnings("NullableProblems")
017@ControllerAdvice
018public class NoContentResponseFilter implements ResponseBodyAdvice<Object> {
019
020  @Override
021  public boolean supports(MethodParameter returnType, Class<? extends HttpMessageConverter<?>> converterType) {
022    return true;
023  }
024
025  @Override
026  public Object beforeBodyWrite(Object body,
027                                MethodParameter returnType,
028                                MediaType selectedContentType,
029                                Class<? extends HttpMessageConverter<?>> selectedConverterType,
030                                ServerHttpRequest request,
031                                ServerHttpResponse response) {
032    final int intStatus = ((ServletServerHttpResponse) response).getServletResponse().getStatus();
033    final HttpStatus httpStatus = HttpStatus.resolve(intStatus);
034
035    if (body == null && httpStatus == HttpStatus.OK) {
036      response.setStatusCode(HttpStatus.NO_CONTENT);
037    }
038
039    return body;
040  }
041}
