001package org.gbif.ws.server.advice;
002
003import org.springframework.core.MethodParameter;
004import org.springframework.http.HttpMethod;
005import org.springframework.http.HttpStatus;
006import org.springframework.http.MediaType;
007import org.springframework.http.converter.HttpMessageConverter;
008import org.springframework.http.server.ServerHttpRequest;
009import org.springframework.http.server.ServerHttpResponse;
010import org.springframework.http.server.ServletServerHttpResponse;
011import org.springframework.web.bind.annotation.ControllerAdvice;
012import org.springframework.web.servlet.mvc.method.annotation.ResponseBodyAdvice;
013
014/**
015 * Filter that updates http headers when a new resource is successfully created via a POST request unless
016 * the response returns 204 No Content.
017 * <p>
018 * The following headers are added or replaced if they existed:
019 * <ul>
020 *   <li>Http response code 201</li>
021 *   <li>Location header is set accordingly based on returned key</li>
022 * </ul>
023 */
024@ControllerAdvice
025public class CreatedResponseFilter implements ResponseBodyAdvice<Object> {
026
027  @Override
028  public boolean supports(MethodParameter returnType, Class<? extends HttpMessageConverter<?>> converterType) {
029    return true;
030  }
031
032  @Override
033  public Object beforeBodyWrite(Object body,
034                                MethodParameter returnType,
035                                MediaType selectedContentType,
036                                Class<? extends HttpMessageConverter<?>> selectedConverterType,
037                                ServerHttpRequest request,
038                                ServerHttpResponse response) {
039    final int intStatus = ((ServletServerHttpResponse) response).getServletResponse().getStatus();
040    final HttpStatus httpStatus = HttpStatus.resolve(intStatus);
041
042    if (request.getMethod() != null && httpStatus != null && request.getMethod() == HttpMethod.POST
043        && httpStatus != HttpStatus.NO_CONTENT && httpStatus.is2xxSuccessful()) {
044      response.setStatusCode(HttpStatus.CREATED);
045
046      // TODO: 15/01/2020 if response contains the key, also set Location
047    }
048
049    return body;
050  }
051}
