001package org.gbif.ws.server.interceptor;
002
003import com.sun.jersey.api.NotFoundException;
004import org.aopalliance.intercept.MethodInterceptor;
005import org.aopalliance.intercept.MethodInvocation;
006
007/**
008 * This method interceptor throws a {@link NotFoundException} for every {@code null} return value of a method.
009 * <p/>
010 * This exception is mapped to a <em>404</em> response code (<em>Not found</em>).
011 */
012public class NullToNotFoundInterceptor implements MethodInterceptor {
013
014  @Override
015  public Object invoke(MethodInvocation invocation) throws Throwable {
016
017    Object result = invocation.proceed();
018
019    if (result == null) {
020      throw new NotFoundException();
021    }
022    return result;
023  }
024}
