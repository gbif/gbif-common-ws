001package org.gbif.ws.json;
002
003import java.util.Map;
004
005import javax.ws.rs.Produces;
006import javax.ws.rs.core.MediaType;
007import javax.ws.rs.ext.ContextResolver;
008import javax.ws.rs.ext.Provider;
009
010import com.google.inject.Singleton;
011import org.codehaus.jackson.map.DeserializationConfig;
012import org.codehaus.jackson.map.ObjectMapper;
013import org.codehaus.jackson.map.SerializationConfig;
014import org.codehaus.jackson.map.annotate.JsonSerialize;
015
016/**
017 * Provider that initializes the {@link ObjectMapper} to ignore {@code null} fields.
018 * NOTE! This class has the same name as a class provided by jersey, BUT THEY DO DIFFERENT THINGS! This
019 * class depends on the presence of the jersey version in order to do anything useful. If you get errors that
020 * suggest that you don't have an appropriate MessageBodyReader for the MIME type application/json, then chances
021 * are good that you have not included the jersey-provided JacksonJsonProvider in your configuration.
022 */
023@Provider
024@Produces(MediaType.APPLICATION_JSON)
025@Singleton
026public class JacksonJsonContextResolver implements ContextResolver<ObjectMapper> {
027
028  protected static final ObjectMapper MAPPER = new ObjectMapper();
029
030  static {
031    MAPPER.setSerializationInclusion(JsonSerialize.Inclusion.NON_NULL);
032    // determines whether encountering of unknown properties (ones that do not map to a property, and there is no
033    // "any setter" or handler that can handle it) should result in a failure (throwing a JsonMappingException) or not.
034    MAPPER.disable(DeserializationConfig.Feature.FAIL_ON_UNKNOWN_PROPERTIES);
035    
036    // Enforce use of ISO-8601 format dates (http://wiki.fasterxml.com/JacksonFAQDateHandling)
037    MAPPER.configure(SerializationConfig.Feature.WRITE_DATES_AS_TIMESTAMPS, false);
038  }
039
040  /**
041   * Specific polymorphic serialization is supported by passing a map of mixIns into the class.
042   * 
043   * @param mixIns class map
044   */
045  public static void addMixIns(Map<Class<?>, Class<?>> mixIns) {
046    // handle polymorphic JSON
047    for (Map.Entry<Class<?>, Class<?>> classClassEntry : mixIns.entrySet()) {
048      MAPPER.getSerializationConfig().addMixInAnnotations(classClassEntry.getKey(), classClassEntry.getValue());
049      MAPPER.getDeserializationConfig().addMixInAnnotations(classClassEntry.getKey(), classClassEntry.getValue());
050    }
051  }
052
053  @Override
054  public ObjectMapper getContext(Class<?> type) {
055    return MAPPER;
056  }
057}
