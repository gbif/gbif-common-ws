001package org.gbif.ws.json;
002
003import com.fasterxml.jackson.annotation.JsonInclude;
004import com.fasterxml.jackson.databind.DeserializationFeature;
005import com.fasterxml.jackson.databind.ObjectMapper;
006import com.fasterxml.jackson.databind.SerializationFeature;
007import org.gbif.api.ws.mixin.Mixins;
008
009public final class JacksonJsonObjectMapperProvider {
010
011  private JacksonJsonObjectMapperProvider() {
012  }
013
014  public static ObjectMapper getObjectMapper() {
015    final ObjectMapper objectMapper = new ObjectMapper();
016    objectMapper.setSerializationInclusion(JsonInclude.Include.NON_NULL);
017    // determines whether encountering of unknown properties (ones that do not map to a property,
018    // and there is no
019    // "any setter" or handler that can handle it) should result in a failure (throwing a
020    // JsonMappingException) or not.
021    objectMapper.disable(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES);
022
023    // Enforce use of ISO-8601 format dates (http://wiki.fasterxml.com/JacksonFAQDateHandling)
024    objectMapper.configure(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS, false);
025
026    Mixins.getPredefinedMixins().forEach(objectMapper::addMixIn);
027
028    return objectMapper;
029  }
030}
