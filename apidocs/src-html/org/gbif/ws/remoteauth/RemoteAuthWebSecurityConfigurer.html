014package org.gbif.ws.remoteauth;
015
016import org.gbif.ws.remoteauth.app.GbifAppRemoteAuthenticationProvider;
017import org.gbif.ws.remoteauth.app.GbifAppRequestFilter;
018import org.gbif.ws.remoteauth.basic.BasicRemoteAuthenticationProvider;
019import org.gbif.ws.remoteauth.jwt.JwtRemoteBasicAuthenticationProvider;
020import org.gbif.ws.remoteauth.jwt.JwtRequestFilter;
021import org.gbif.ws.server.filter.HttpServletRequestWrapperFilter;
022import org.gbif.ws.server.filter.RequestHeaderParamUpdateFilter;
023
024import java.util.Arrays;
025import java.util.Collections;
026
027import org.springframework.context.ApplicationContext;
028import org.springframework.context.annotation.Bean;
029import org.springframework.security.config.annotation.authentication.builders.AuthenticationManagerBuilder;
030import org.springframework.security.config.annotation.web.builders.HttpSecurity;
031import org.springframework.security.config.annotation.web.configuration.WebSecurityConfigurerAdapter;
032import org.springframework.security.config.http.SessionCreationPolicy;
033import org.springframework.security.web.authentication.www.BasicAuthenticationFilter;
034import org.springframework.security.web.csrf.CsrfFilter;
035import org.springframework.web.cors.CorsConfiguration;
036import org.springframework.web.cors.CorsConfigurationSource;
037import org.springframework.web.cors.UrlBasedCorsConfigurationSource;
038
039/**
040 * Security Adapter that disables the authentication redirect and use GBIF remote services.
041 * Supports Basic and JWT authentication through JwtRemoteBasicAuthenticationProvider and
042 * JwtRemoteBasicAuthenticationProvider.
043 */
044public class RemoteAuthWebSecurityConfigurer extends WebSecurityConfigurerAdapter {
045
046  private final RemoteAuthClient remoteAuthClient;
047
048  public RemoteAuthWebSecurityConfigurer(
049      ApplicationContext context, RemoteAuthClient remoteAuthClient) {
050
051    setApplicationContext(context);
052    this.remoteAuthClient = remoteAuthClient;
053  }
054
055  @Override
056  protected void configure(AuthenticationManagerBuilder auth) {
057    auth.authenticationProvider(new BasicRemoteAuthenticationProvider(remoteAuthClient));
058    auth.authenticationProvider(new JwtRemoteBasicAuthenticationProvider(remoteAuthClient));
059    auth.authenticationProvider(new GbifAppRemoteAuthenticationProvider(remoteAuthClient));
060  }
061
062  @Override
063  protected void configure(HttpSecurity http) throws Exception {
064    http.httpBasic()
065        .and()
066        .addFilterAfter(
067            getApplicationContext().getBean(HttpServletRequestWrapperFilter.class),
068            CsrfFilter.class)
069        .addFilterAfter(
070            getApplicationContext().getBean(RequestHeaderParamUpdateFilter.class),
071            HttpServletRequestWrapperFilter.class)
072        .addFilterBefore(
073            new JwtRequestFilter(authenticationManager()), BasicAuthenticationFilter.class)
074        .addFilterAfter(
075            new GbifAppRequestFilter(authenticationManager()), BasicAuthenticationFilter.class)
076        .csrf()
077        .disable()
078        .cors()
079        .and()
080        .authorizeRequests()
081        .anyRequest()
082        .authenticated();
083
084    http.sessionManagement().sessionCreationPolicy(SessionCreationPolicy.STATELESS);
085  }
086
087  /**
088   * Cors configuration, allows all methods and origins.
089   */
090  @Bean
091  CorsConfigurationSource corsConfigurationSource() {
092    // CorsFilter only applies this if the origin header is present in the request
093    CorsConfiguration configuration = new CorsConfiguration();
094    configuration.setAllowedHeaders(Arrays.asList("authorization", "content-type"));
095    configuration.setAllowedOrigins(Collections.singletonList("*"));
096    configuration.setAllowedMethods(
097        Arrays.asList("HEAD", "GET", "POST", "DELETE", "PUT", "OPTIONS"));
098    configuration.setExposedHeaders(
099        Arrays.asList(
100            "Access-Control-Allow-Origin",
101            "Access-Control-Allow-Methods",
102            "Access-Control-Allow-Headers"));
103    UrlBasedCorsConfigurationSource source = new UrlBasedCorsConfigurationSource();
104    source.registerCorsConfiguration("/**", configuration);
105    return source;
106  }
107}
