014package org.gbif.ws.remoteauth.app;
015
016import org.gbif.api.vocabulary.AppRole;
017import org.gbif.ws.remoteauth.AbstractRemoteAuthenticationProvider;
018import org.gbif.ws.remoteauth.RemoteAuthClient;
019import org.gbif.ws.security.AppPrincipal;
020import org.gbif.ws.security.GbifAuthenticationToken;
021import org.gbif.ws.util.SecurityConstants;
022
023import java.util.ArrayList;
024import java.util.Collection;
025
026import org.springframework.http.HttpHeaders;
027import org.springframework.http.ResponseEntity;
028import org.springframework.security.core.Authentication;
029import org.springframework.security.core.authority.SimpleGrantedAuthority;
030
031import lombok.extern.slf4j.Slf4j;
032
033import static org.gbif.ws.util.SecurityConstants.GBIF_SCHEME;
034
035/**
036 * GBIF APP authentication against the registry.
037 */
038@Slf4j
039public class GbifAppRemoteAuthenticationProvider
040    extends AbstractRemoteAuthenticationProvider<GbifAppAuthentication> {
041
042  private static final String AUTH_PATH = "/user/auth/app";
043
044  public GbifAppRemoteAuthenticationProvider(RemoteAuthClient remoteAuthClient) {
045    super(GbifAppAuthentication.class, AUTH_PATH, remoteAuthClient);
046  }
047
048  @Override
049  public HttpHeaders createHttpHeaders(Authentication authentication) {
050    HttpHeaders headers = new HttpHeaders();
051    GbifAppAuthentication gbifAppAuthentication = (GbifAppAuthentication) authentication;
052    headers.add(HttpHeaders.AUTHORIZATION, gbifAppAuthentication.getGbifScheme());
053    headers.add(SecurityConstants.HEADER_CONTENT_MD5, gbifAppAuthentication.getContentMd5());
054    if (authentication.getPrincipal() != null) {
055      headers.add(SecurityConstants.HEADER_GBIF_USER, authentication.getPrincipal().toString());
056    }
057    if (gbifAppAuthentication.getOriginalRequestUrl() != null
058        && !gbifAppAuthentication.getOriginalRequestUrl().isEmpty()) {
059      headers.add(
060          SecurityConstants.HEADER_ORIGINAL_REQUEST_URL,
061          gbifAppAuthentication.getOriginalRequestUrl());
062    }
063    return headers;
064  }
065
066  @Override
067  protected Authentication createSuccessAuthentication(
068      ResponseEntity<String> response, Authentication authentication) {
069    Collection<SimpleGrantedAuthority> authorities = extractRoles(readUserFromResponse(response));
070    authorities.add(new SimpleGrantedAuthority(AppRole.APP.name()));
071
072    return new GbifAuthenticationToken(
073        new AppPrincipal(
074            ((GbifAppAuthentication) authentication).getAppKey(), new ArrayList<>(authorities)),
075        GBIF_SCHEME,
076        authorities);
077  }
078}
