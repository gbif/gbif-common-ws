017package org.gbif.ws.util.spring;
018
019import com.google.common.base.Strings;
020
021/**
022 * Assertion utility class that assists in validating arguments.
023 * Useful for identifying programmer errors early and clearly at runtime.
024 * <p>For example, if the contract of a public method states it does not
025 * allow <code>null</code> arguments, Assert can be used to validate that
026 * contract. Doing this clearly indicates a contract violation when it
027 * occurs and protects the class's invariants.
028 * <p>Typically used to validate method arguments rather than configuration
029 * properties, to check for cases that are usually programmer errors rather than
030 * configuration errors. In contrast to config initialization code, there is
031 * usally no point in falling back to defaults in such methods.
032 * <p>This class is similar to JUnit's assertion library. If an argument value is
033 * deemed invalid, an {@link IllegalArgumentException} is thrown (typically).
034 * For example:
035 * <pre class="code">
036 * Assert.notNull(clazz, "The class must not be null");
037 * Assert.isTrue(i > 0, "The value must be greater than zero");</pre>
038 * Mainly for internal use within the framework; consider Jakarta's Commons Lang
039 * >= 2.0 for a more comprehensive suite of assertion utilities.
040 *
041 * @author Keith Donald
042 * @author Juergen Hoeller
043 * @author Colin Sampaleanu
044 * @author Rob Harrop
045 * @since 1.1.2
046 */
047public abstract class Assert {
048
049  /**
050   * Assert a boolean expression, throwing <code>IllegalArgumentException</code>
051   * if the test result is <code>false</code>.
052   * <pre class="code">Assert.isTrue(i &gt; 0, "The value must be greater than zero");</pre>
053   *
054   * @param expression a boolean expression
055   * @param message    the exception message to use if the assertion fails
056   *
057   * @throws IllegalArgumentException if expression is <code>false</code>
058   */
059  public static void isTrue(boolean expression, String message) {
060    if (!expression) {
061      throw new IllegalArgumentException(message);
062    }
063  }
064
065  /**
066   * Assert a boolean expression, throwing <code>IllegalArgumentException</code>
067   * if the test result is <code>false</code>.
068   * <pre class="code">Assert.isTrue(i &gt; 0);</pre>
069   *
070   * @param expression a boolean expression
071   *
072   * @throws IllegalArgumentException if expression is <code>false</code>
073   */
074  public static void isTrue(boolean expression) {
075    isTrue(expression, "[Assertion failed] - this expression must be true");
076  }
077
078  /**
079   * Assert that an object is <code>null</code> .
080   * <pre class="code">Assert.isNull(value, "The value must be null");</pre>
081   *
082   * @param object  the object to check
083   * @param message the exception message to use if the assertion fails
084   *
085   * @throws IllegalArgumentException if the object is not <code>null</code>
086   */
087  public static void isNull(Object object, String message) {
088    if (object != null) {
089      throw new IllegalArgumentException(message);
090    }
091  }
092
093  /**
094   * Assert that an object is <code>null</code> .
095   * <pre class="code">Assert.isNull(value);</pre>
096   *
097   * @param object the object to check
098   *
099   * @throws IllegalArgumentException if the object is not <code>null</code>
100   */
101  public static void isNull(Object object) {
102    isNull(object, "[Assertion failed] - the object argument must be null");
103  }
104
105  /**
106   * Assert that an object is not <code>null</code> .
107   * <pre class="code">Assert.notNull(clazz, "The class must not be null");</pre>
108   *
109   * @param object  the object to check
110   * @param message the exception message to use if the assertion fails
111   *
112   * @throws IllegalArgumentException if the object is <code>null</code>
113   */
114  public static void notNull(Object object, String message) {
115    if (object == null) {
116      throw new IllegalArgumentException(message);
117    }
118  }
119
120  /**
121   * Assert that an object is not <code>null</code> .
122   * <pre class="code">Assert.notNull(clazz);</pre>
123   *
124   * @param object the object to check
125   *
126   * @throws IllegalArgumentException if the object is <code>null</code>
127   */
128  public static void notNull(Object object) {
129    notNull(object, "[Assertion failed] - this argument is required; it must not be null");
130  }
131
132
133  /**
134   * Assert that the given String has valid text content; that is, it must not
135   * be <code>null</code> and must contain at least one non-whitespace character.
136   * <pre class="code">Assert.hasText(name, "'name' must not be empty");</pre>
137   *
138   * @param text    the String to check
139   * @param message the exception message to use if the assertion fails
140   */
141  public static void hasText(String text, String message) {
142    if (Strings.isNullOrEmpty(text)) {
143      throw new IllegalArgumentException(message);
144    }
145  }
146
147  /**
148   * Assert that the given String has valid text content; that is, it must not
149   * be <code>null</code> and must contain at least one non-whitespace character.
150   * <pre class="code">Assert.hasText(name, "'name' must not be empty");</pre>
151   *
152   * @param text the String to check
153   */
154  public static void hasText(String text) {
155    hasText(text,
156      "[Assertion failed] - this String argument must have text; it must not be <code>null</code>, empty, or blank");
157  }
158
159
160}
