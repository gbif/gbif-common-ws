001package org.gbif.ws.util;
002
003/**
004 * Extra media types used in Http responses.
005 */
006public class ExtraMediaTypes {
007
008  /**
009   * Darwin Core archive media type with underlying zip structure.
010   * Use carefully, it's an unregistered media type, in most of the cases it is more appropriate to return a simple
011   * application/zip
012   * http://www.iana.org/assignments/media-types/media-types.xhtml
013   * Currently used for experimenting in OAI-PMH DublinCore resources.
014   */
015  public static final String APPLICATION_DWCA = "application/dwca+zip";
016
017  private ExtraMediaTypes() {
018    throw new UnsupportedOperationException("Can't initialize class");
019  }
020}
