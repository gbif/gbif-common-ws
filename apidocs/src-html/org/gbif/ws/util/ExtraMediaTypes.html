016package org.gbif.ws.util;
017
018import javax.ws.rs.core.MediaType;
019
020/**
021 * Extra media types used in Http responses.
022 */
023public class ExtraMediaTypes {
024
025  public static final String APPLICATION_JAVASCRIPT = "application/javascript";
026  public static final MediaType APPLICATION_JAVASCRIPT_TYPE = new MediaType("application", "javascript");
027
028  public static final String TEXT_CSV = "text/csv";
029  public static final MediaType TEXT_CSV_TYPE = new MediaType("text", "csv");
030
031  public static final String TEXT_TSV = "text/tab-separated-values";
032  public static final MediaType TEXT_TSV_TYPE = new MediaType("text", "tab-separated-values");
033
034  //.xls
035  public static final String APPLICATION_EXCEL = "application/vnd.ms-excel";
036  public static final MediaType APPLICATION_EXCEL_TYPE = new MediaType("application", "vnd.ms-excel");
037
038  //.xlsx
039  public static final String APPLICATION_OFFICE_SPREADSHEET = "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet";
040  public static final MediaType APPLICATION_OFFICE_SPREADSHEET_TYPE = new MediaType("application", "vnd.openxmlformats-officedocument.spreadsheetml.sheet");
041
042  //.ods
043  public static final String APPLICATION_OPEN_DOC_SPREADSHEET = "application/vnd.oasis.opendocument.spreadsheet";
044  public static final MediaType APPLICATION_OPEN_DOC_SPREADSHEET_TYPE = new MediaType("application", "vnd.oasis.opendocument.spreadsheet");
045
046
047  /**
048   * Darwin Core archive media type with underlying zip structure.
049   * Use carefully, it's an unregistered media type, in most of the cases it is more appropriate to return a simple
050   * application/zip
051   * http://www.iana.org/assignments/media-types/media-types.xhtml
052   * Currently used for experimenting in OAI-PMH DublinCore resources.
053   */
054  public static final String APPLICATION_DWCA = "application/dwca+zip";
055
056  private ExtraMediaTypes() {
057    throw new UnsupportedOperationException("Can't initialize class");
058  }
059
060}
