001package org.gbif.ws.util;
002
003import com.google.common.base.Strings;
004import org.springframework.http.MediaType;
005
006import javax.annotation.Nullable;
007import javax.validation.constraints.NotNull;
008import java.util.Map;
009
010/**
011 * Class with util methods for WS.
012 */
013public final class CommonWsUtils {
014
015  private CommonWsUtils() {
016  }
017
018  /**
019   * Retrieve the first occurrence of the param in params.
020   * Can be applied for HttpHeaders.
021   */
022  public static String getFirst(Map<String, String[]> params, String param) {
023    final String[] values = params.get(param);
024    String resultValue = null;
025
026    if (values != null && values[0] != null) {
027      resultValue = values[0];
028    }
029
030    return resultValue;
031  }
032
033  /**
034   * MediaType by string extension.
035   */
036  @Nullable
037  public static String getResponseTypeByExtension(@Nullable String extension, @NotNull String defaultMediaType) {
038    if (Strings.isNullOrEmpty(extension)) {
039      return defaultMediaType;
040    } else if (".xml".equals(extension)) {
041      return MediaType.APPLICATION_XML_VALUE;
042    } else if (".json".equals(extension)) {
043      return MediaType.APPLICATION_JSON_VALUE;
044    } else {
045      return null;
046    }
047  }
048}
