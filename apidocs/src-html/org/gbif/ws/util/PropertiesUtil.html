001package org.gbif.ws.util;
002
003import java.io.File;
004import java.io.FileReader;
005import java.io.IOException;
006import java.io.InputStream;
007import java.util.Properties;
008
009import com.google.common.base.Strings;
010import com.google.common.io.Closeables;
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013
014/**
015 * Common utilities for dealing with Properties.
016 *
017 * @deprecated use org.gbif.utils.file.properties.PropertiesUtil instead
018 */
019@Deprecated
020public class PropertiesUtil {
021
022  private static final Logger LOG = LoggerFactory.getLogger(PropertiesUtil.class);
023
024  /**
025   * Reads a property file from the classpath.
026   */
027  public static Properties readFromClasspath(String filename) {
028    InputStream inputStream = PropertiesUtil.class.getClassLoader().getResourceAsStream(filename);
029    Properties properties = new Properties();
030    try {
031      properties.load(inputStream);
032    } catch (IOException e) {
033      LOG.error("Cannot read property file {}", filename, e);
034    } catch (NullPointerException e) {
035      LOG.error("Cannot read property file {}", filename, e);
036    } finally {
037      Closeables.closeQuietly(inputStream);
038    }
039    return properties;
040  }
041
042  /**
043   * Reads a property file from an absolute filepath.
044   */
045  public static Properties readFromFile(String filepath) throws IOException {
046    if (Strings.isNullOrEmpty(filepath)) {
047      throw new IOException("No properties file given");
048    }
049    File pf = new File(filepath);
050    if (!pf.exists()) {
051      throw new IOException("Cannot find properties file " + filepath);
052    }
053    FileReader reader = new FileReader(pf);
054    Properties properties = new Properties();
055    try {
056      properties.load(reader);
057    } finally {
058      Closeables.closeQuietly(reader);
059    }
060    return properties;
061  }
062
063  private PropertiesUtil() {
064  }
065}
