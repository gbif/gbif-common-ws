001package org.gbif.ws.util;
002
003import java.util.regex.Matcher;
004import java.util.regex.Pattern;
005
006import org.slf4j.Logger;
007import org.slf4j.LoggerFactory;
008
009/**
010 * Class with utility methods for XSS filtering.
011 */
012public class XSSUtil {
013  private static final Logger LOG = LoggerFactory.getLogger(XSSUtil.class);
014
015  private XSSUtil() {
016    // empty private constructor
017  }
018
019  private static final String NULL_CHAR = "\0";
020  private static final Pattern[] PATTERNS = new Pattern[] {
021    // Avoid anything in a <script> type of expression
022    Pattern.compile("<script>(.*?)</script>", Pattern.CASE_INSENSITIVE),
023    // Avoid anything in a src='...' type of expression
024    Pattern.compile("src[\r\n]*=[\r\n]*\\\'(.*?)\\\'", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL),
025    Pattern.compile("src[\r\n]*=[\r\n]*\\\"(.*?)\\\"", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL),
026    // Remove any lonesome </script> tag
027    Pattern.compile("</script>", Pattern.CASE_INSENSITIVE),
028    // Remove any lonesome <script ...> tag
029    Pattern.compile("<script(.*?)>", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL),
030    // Avoid eval(...) expressions
031    Pattern.compile("eval\\((.*?)\\)", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL),
032    // Avoid expression(...) expressions
033    Pattern.compile("expression\\((.*?)\\)", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL),
034    // Avoid javascript:... expressions
035    Pattern.compile("javascript:", Pattern.CASE_INSENSITIVE),
036    // Avoid vbscript:... expressions
037    Pattern.compile("vbscript:", Pattern.CASE_INSENSITIVE),
038    // Avoid onload= expressions
039    Pattern.compile("onload(.*?)=", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL)
040  };
041
042
043  public static String stripXSS(String value) {
044    if (value != null) {
045
046      // Avoid null characters
047      value = value.replaceAll(NULL_CHAR, "");
048
049      // Remove all sections that match a pattern
050      boolean found = false;
051      for (Pattern scriptPattern : PATTERNS) {
052          Matcher matcher = scriptPattern.matcher(value);
053          if (matcher.find()) {
054            found = true;
055            value = matcher.replaceAll("");
056          }
057      }
058      if (found) {
059        LOG.warn("Malicious XSS script found: {}", value);
060      }
061    }
062    return value;
063  }
064}
