001package org.gbif.ws.util;
002
003import java.util.regex.Pattern;
004
005/**
006 * Class with utility methods for XSS filtering.
007 */
008public class XSSUtil {
009
010  private XSSUtil() {
011    // empty private constructor
012  }
013
014  private static final String NULL_CHAR = "\0";
015  private static final Pattern[] PATTERNS = new Pattern[] {
016    // Avoid anything in a <script> type of expression
017    Pattern.compile("<script>(.*?)</script>", Pattern.CASE_INSENSITIVE),
018    // Avoid anything in a src='...' type of expression
019    Pattern.compile("src[\r\n]*=[\r\n]*\\\'(.*?)\\\'", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL),
020    Pattern.compile("src[\r\n]*=[\r\n]*\\\"(.*?)\\\"", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL),
021    // Remove any lonesome </script> tag
022    Pattern.compile("</script>", Pattern.CASE_INSENSITIVE),
023    // Remove any lonesome <script ...> tag
024    Pattern.compile("<script(.*?)>", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL),
025    // Avoid eval(...) expressions
026    Pattern.compile("eval\\((.*?)\\)", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL),
027    // Avoid expression(...) expressions
028    Pattern.compile("expression\\((.*?)\\)", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL),
029    // Avoid javascript:... expressions
030    Pattern.compile("javascript:", Pattern.CASE_INSENSITIVE),
031    // Avoid vbscript:... expressions
032    Pattern.compile("vbscript:", Pattern.CASE_INSENSITIVE),
033    // Avoid onload= expressions
034    Pattern.compile("onload(.*?)=", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL)
035  };
036
037
038  public static String stripXSS(String value) {
039    if (value != null) {
040
041      // Avoid null characters
042      value = value.replaceAll(NULL_CHAR, "");
043
044      // Remove all sections that match a pattern
045      for (Pattern scriptPattern : PATTERNS) {
046        value = scriptPattern.matcher(value).replaceAll("");
047      }
048    }
049    return value;
050  }
051}
