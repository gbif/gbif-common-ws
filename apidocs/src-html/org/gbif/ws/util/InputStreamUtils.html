001package org.gbif.ws.util;
002
003import org.gbif.api.exception.ServiceUnavailableException;
004
005import java.io.InputStream;
006
007import com.sun.jersey.api.client.ClientResponse;
008import com.sun.jersey.api.client.WebResource;
009
010/**
011 * Utility class that validates response statuses and extract the input stream from a HTTP response.
012 */
013public class InputStreamUtils {
014
015  private InputStreamUtils() {
016    //empty constructor
017  }
018
019  /**
020   * Only allow status 200 responses and throw NotFound or ServiceUnavailable exceptions otherwise.
021   * @return the http entities input stream if http status=200 or null for 404
022   * @throws org.gbif.api.exception.ServiceUnavailableException in all other cases
023   */
024  public static InputStream wrapStream(WebResource resource) {
025
026    final ClientResponse response = resource.get(ClientResponse.class);
027
028    if (response.getStatus() == ClientResponse.Status.NOT_FOUND.getStatusCode()) {
029      response.close();
030      return null;
031    }
032
033    if (response.getStatus() != ClientResponse.Status.OK.getStatusCode()) {
034      response.close();
035      throw new ServiceUnavailableException("HTTP " + response.getStatus() + ": " +
036                                            response.getStatusInfo().getReasonPhrase());
037    }
038
039    return response.getEntityInputStream();
040  }
041}
