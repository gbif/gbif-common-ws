1   package org.gbif.ws.security;

3   import org.slf4j.Logger;
4   import org.slf4j.LoggerFactory;
5   import org.springframework.context.annotation.Primary;
6   import org.springframework.stereotype.Service;
7   
8   @Primary
9   @Service
10  public class AppKeySigningService extends BaseSigningService {
11  
12    private static final Logger LOG = LoggerFactory.getLogger(AppKeySigningService.class);
13  
14    private final KeyStore keyStore;
15  
16    public AppKeySigningService(KeyStore keyStore) {
17      this.keyStore = keyStore;
18    }
19  
20    /**
21     * Generates a Base64 encoded HMAC-SHA1 signature of the passed request data with the secret key
22     * associated with the given application key. See Message Authentication Code specs
23     * http://tools.ietf.org/html/rfc2104
24     *
25     * @param requestDataToSign the request data to be signed
26     * @param appKey            the application key
27     */
28    @Override
29    public String buildSignature(RequestDataToSign requestDataToSign, String appKey) {
30      // find private key for this app
31      final String secretKey = keyStore.getPrivateKey(appKey);
32      if (secretKey == null) {
33        LOG.error("Unknown application key: {}", appKey);
34        throw new PrivateKeyNotFoundException();
35      }
36  
37      // sign
38      return super.buildSignature(requestDataToSign, secretKey);
39    }
40  }
