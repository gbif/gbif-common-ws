29  /**
30   * The GBIF authentication scheme is modelled after the Amazon scheme on how to sign REST http requests
31   * using a private key. It uses the standard Http Authorization Header to transport the following information:
32   * Authorization: GBIF applicationKey:Signature
33   *
34   * <br/>
35   * The header starts with the authentication scheme (GBIF), followed by the plain applicationKey (the public key)
36   * and a unique signature for the very request which is generated using a fixed set of request attributes
37   * which are then encrypted by a standard HMAC-SHA1 algorithm.
38   *
39   * <br/>
40   * A POST request with a GBIF header would look like this:
41   *
42   * <pre>
43   * POST /dataset HTTP/1.1
44   * Host: johnsmith.s3.amazonaws.com
45   * Date: Mon, 26 Mar 2007 19:37:58 +0000
46   * x-gbif-user: trobertson
47   * Content-MD5: LiFThEP4Pj2TODQXa/oFPg==
48   * Authorization: GBIF gbif.portal:frJIUN8DYpKDtOLCwo//yllqDzg=
49   * </pre>
50   *
51   * When signing an http request in addition to the Authentication header some additional custom headers are added
52   * which are used to sign and digest the message.
53   * <br/>
54   * x-gbif-user is added to transport a proxied user in which the application is acting.
55   * <br/>
56   * Content-MD5 is added if a body entity exists.
57   * See Concent-MD5 header specs: http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.15
58   */
59  @Singleton
60  public class GbifAuthService {

102   /**
103    * Extracts the information to be encrypted from a request and concatenates them into a single String.
104    * When the server receives an authenticated request, it compares the computed request signature
105    * with the signature provided in the request in StringToSign.
106    * For that reason this string may only contain information also available in the exact same form to the server.
107    *
108    * @return unique string for a request
109    *
110    * @see <a href="http://docs.amazonwebservices.com/AmazonS3/latest/dev/RESTAuthentication.html">AWS Docs</a>
111    */
112   private String buildStringToSign(ContainerRequest request) {
113     StringBuilder sb = new StringBuilder();
114 
115     sb.append(request.getMethod());
116     sb.append(NEWLINE);
117     // custom header set by varnish overrides real URI
118     // see http://dev.gbif.org/issues/browse/GBIFCOM-137
119     if (request.getRequestHeaders().containsKey(HEADER_ORIGINAL_REQUEST_URL)) {
120       sb.append(request.getRequestHeaders().getFirst(HEADER_ORIGINAL_REQUEST_URL));
121     } else {
122       sb.append(getCanonicalizedPath(request.getRequestUri()));
123     }
124 
125     appendHeader(sb, request.getRequestHeaders(), HEADER_CONTENT_TYPE, false);
126     appendHeader(sb, request.getRequestHeaders(), HEADER_CONTENT_MD5, true);
127     appendHeader(sb, request.getRequestHeaders(), HEADER_GBIF_USER, true);
128 
129     return sb.toString();
130   }
131 
132   /**
133    * Build the string to be signed for a client request by extracting header information from the request.
134    * For PUT/POST requests that contain a body content it is required that the Content-MD5 header
135    * is already present on the request instance!
136    */
137   private String buildStringToSign(ClientRequest request) {
138     StringBuilder sb = new StringBuilder();
139 
140     sb.append(request.getMethod());
141     sb.append(NEWLINE);
142     sb.append(getCanonicalizedPath(request.getURI()));
143 
144     appendHeader(sb, request.getHeaders(), HEADER_CONTENT_TYPE, false);
145     appendHeader(sb, request.getHeaders(), HEADER_CONTENT_MD5, true);
146     appendHeader(sb, request.getHeaders(), HEADER_GBIF_USER, true);
147 
148     return sb.toString();
149   }
150 
151   private void appendHeader(StringBuilder sb, MultivaluedMap<String, ?> request, String header, boolean caseSensitive) {
152     if (request.containsKey(header)) {
153       sb.append(NEWLINE);
154       if (caseSensitive) {
155         sb.append(request.getFirst(header));
156       } else {
157         sb.append(request.getFirst(header).toString().toLowerCase());
158       }
159     }
160   }

162   /**
163    * @return an absolute uri of the resource path alone, excluding host, scheme and query parameters

166     return uri.normalize().getPath();

169   private String buildAuthHeader(String applicationKey, String signature) {
170     return GBIF_SCHEME + " " + applicationKey + ":" + signature;
171   }

173   /**
174    * Generates a Base64 encoded HMAC-SHA1 signature of the passed in string with the given secret key.
175    * See Message Authentication Code specs http://tools.ietf.org/html/rfc2104
176    * @param stringToSign the string to be signed
177    * @param secretKey the secret key to use in the
178    */
179   private String buildSignature(String stringToSign, String secretKey) {
180     try {
181       Mac mac = Mac.getInstance(ALGORITHM);
182       SecretKeySpec secret = new SecretKeySpec(secretKey.getBytes(Charset.forName("UTF8")), ALGORITHM);
183       mac.init(secret);
184       byte[] digest = mac.doFinal(stringToSign.getBytes());
185 
186       return new String(Base64.encode(digest), "ASCII");
187 
188     } catch (NoSuchAlgorithmException e) {
189       throw new RuntimeException("Cant find " + ALGORITHM + " message digester", e);
190     } catch (UnsupportedEncodingException e) {
191       throw new RuntimeException("Unsupported character encoding " + CHAR_ENCODING, e);
192     } catch (InvalidKeyException e) {
193       throw new RuntimeException("Invalid secret key " + secretKey, e);
194     }
195   }

197   /**
198    * Signs a request by adding a Content-MD5 and Authorization header.
199    * For PUT/POST requests that contain a body entity the Content-MD5 header is created using the same
200    * JSON mapper for serialization as the clients use.
201    *
202    * Other format than JSON are not supported currently !!!
203    */
204   public void signRequest(String username, ClientRequest request) {
205     Preconditions.checkNotNull(appKey, "To sign request a single application key is required");
206     // first add custom GBIF headers so we can use them to build the string to sign
207 
208     // the proxied username
209     request.getHeaders().putSingle(HEADER_GBIF_USER, username);
210     // the canonical path header
211     request.getHeaders().putSingle(HEADER_ORIGINAL_REQUEST_URL, getCanonicalizedPath(request.getURI()));
212     // adds content md5
213     if (request.getEntity() != null) {
214       request.getHeaders().putSingle(HEADER_CONTENT_MD5, buildContentMD5(request.getEntity()));
215     }
216 
217     // build the unique string to sign
218     final String stringToSign = buildStringToSign(request);
219     System.out.println(stringToSign);
220     // find private key for this app
221     final String secretKey = getPrivateKey(appKey);
222     if (secretKey == null) {
223       LOG.warn("Skip signing request with unknown application key: {}", appKey);
224       return;
225     }
226     // sign
227     String signature = buildSignature(stringToSign, secretKey);
228     // build authorization header string
229     String header = buildAuthHeader(appKey, signature);
230     // add authorization header
231     LOG.debug("Adding authentication header to request {} for proxied user {} : {}", request.getURI(), username, header);
232     request.getHeaders().putSingle(HEADER_AUTHORIZATION, header);
233   }

235   /**
236    * Generates the Base64 encoded 128 bit MD5 digest of the entire content string suitable for the
237    * Content-MD5 header value.
238    * See http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.15
239    */
240   private String buildContentMD5(Object entity) {
241     try {
242       byte[] content = mapper.writeValueAsBytes(entity);
243       return new String(Base64.encode(DigestUtils.md5(content)), "ASCII");
244 
245     } catch (IOException e) {
246       LOG.error("Failed to serialize http entity [{}]", entity);
247       throw new RuntimeException(e);
248     }
249   }

251   public boolean isValidRequest(ContainerRequest request) {
252     // parse auth header
253     final String authHeader = request.getHeaderValue(HEADER_AUTHORIZATION);
254     if (Strings.isNullOrEmpty(authHeader) || !authHeader.startsWith(GBIF_SCHEME + " ")) {
255       LOG.info(HEADER_AUTHORIZATION + " header is no GBIF scheme");
256       return false;
257     }
258 
259     String[] values = COLON_PATTERN.split(authHeader.substring(5), 2);
260     if (values.length < 2) {
261       LOG.warn("Invalid syntax for application key and signature: {}", authHeader);
262       return false;
263     }
264 
265     final String appKey = values[0];
266     final String signatureFound = values[1];
267     if (appKey == null || signatureFound == null) {
268       LOG.warn("Authentication header missing applicationKey or signature: {}", authHeader);
269       return false;
270     }
271 
272     String secretKey = getPrivateKey(appKey);
273     if (secretKey == null) {
274       LOG.warn("Unknown application key: {}", appKey);
275       return false;
276     }
277     //
278     String stringToSign = buildStringToSign(request);
279     // sign
280     String signature = buildSignature(stringToSign, secretKey);
281     // compare signatures
282     if (signatureFound.equals(signature)) {
283       LOG.debug("Trusted application with matching signatures");
284       return true;
285 
286     }
287     LOG.info("Invalid signature: {}", authHeader);
288     LOG.debug("StringToSign: {}", stringToSign);
289     return false;
290   }

292   private String getPrivateKey(String appKey) {
293     return keyStore.get(appKey);
