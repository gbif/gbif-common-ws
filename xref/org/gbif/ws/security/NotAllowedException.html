1   package org.gbif.ws.security;
2   
3   import org.gbif.api.model.common.User;
4   
5   import javax.ws.rs.WebApplicationException;
6   import javax.ws.rs.core.Response;
7   
8   /**
9    * HTTP 403 jersey exception when a request has a valid authentication, but the authenticated user is not allowed
10   * to execute the resource.
11   */
12  public class NotAllowedException extends WebApplicationException {
13  
14       public NotAllowedException() {
15           super(Response.status(Response.Status.FORBIDDEN).build());
16       }
17  
18       /**
19        * @param message the String that is the entity of the 403 response.
20        */
21       public NotAllowedException(String message) {
22           super(Response.status(Response.Status.FORBIDDEN).entity(message).type("text/plain").build());
23       }
24  
25    /**
26     * @param user that is authenticated
27     */
28    public NotAllowedException(User user) {
29      this (user.getUserName() + " is not allowed to access this resource.");
30    }
31  
32  }
