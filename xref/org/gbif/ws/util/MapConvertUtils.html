27    /**
28     * Converts a {@link MultivaluedMap} into a {@link Multimap}.
29     * If the parameter is null an empty multimap is returned.
30     */
31    public static <K, V> Multimap<K, V> toMultiMap(MultivaluedMap<K, V> multivaluedMap) {
32      Multimap<K, V> multimap = HashMultimap.create();
33      if (multivaluedMap != null) {
34        for (Entry<K, List<V>> entry : multivaluedMap.entrySet()) {
35          multimap.putAll(entry.getKey(), entry.getValue());
36        }
37      }
38      return multimap;
39    }

44     */
45    public static MultivaluedMap<String, String> toMultivaluedMap(Multimap<String, String> multimap) {
46      MultivaluedMap<String, String> multivaluedMap = new MultivaluedMapImpl();
47      if (multimap != null) {
48        for (Entry<String, String> entry : multimap.entries()) {
49          multivaluedMap.add(entry.getKey(), entry.getValue());
50        }
51      }
52      return multivaluedMap;
53    }
