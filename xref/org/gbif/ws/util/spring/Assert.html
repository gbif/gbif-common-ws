17  package org.gbif.ws.util.spring;
18  
19  import com.google.common.base.Strings;
20  
21  /**
22   * Assertion utility class that assists in validating arguments.
23   * Useful for identifying programmer errors early and clearly at runtime.
24   * <p>For example, if the contract of a public method states it does not
25   * allow <code>null</code> arguments, Assert can be used to validate that
26   * contract. Doing this clearly indicates a contract violation when it
27   * occurs and protects the class's invariants.
28   * <p>Typically used to validate method arguments rather than configuration
29   * properties, to check for cases that are usually programmer errors rather than
30   * configuration errors. In contrast to config initialization code, there is
31   * usally no point in falling back to defaults in such methods.
32   * <p>This class is similar to JUnit's assertion library. If an argument value is
33   * deemed invalid, an {@link IllegalArgumentException} is thrown (typically).
34   * For example:
35   * <pre class="code">
36   * Assert.notNull(clazz, "The class must not be null");
37   * Assert.isTrue(i > 0, "The value must be greater than zero");</pre>
38   * Mainly for internal use within the framework; consider Jakarta's Commons Lang
39   * >= 2.0 for a more comprehensive suite of assertion utilities.
40   *
41   * @author Keith Donald
42   * @author Juergen Hoeller
43   * @author Colin Sampaleanu
44   * @author Rob Harrop
45   * @since 1.1.2
46   */
47  public abstract class Assert {
48  
49    /**
50     * Assert a boolean expression, throwing <code>IllegalArgumentException</code>
51     * if the test result is <code>false</code>.
52     * <pre class="code">Assert.isTrue(i &gt; 0, "The value must be greater than zero");</pre>
53     *
54     * @param expression a boolean expression
55     * @param message    the exception message to use if the assertion fails
56     *
57     * @throws IllegalArgumentException if expression is <code>false</code>
58     */
59    public static void isTrue(boolean expression, String message) {
60      if (!expression) {
61        throw new IllegalArgumentException(message);
62      }
63    }
64  
65    /**
66     * Assert a boolean expression, throwing <code>IllegalArgumentException</code>
67     * if the test result is <code>false</code>.
68     * <pre class="code">Assert.isTrue(i &gt; 0);</pre>
69     *
70     * @param expression a boolean expression
71     *
72     * @throws IllegalArgumentException if expression is <code>false</code>
73     */
74    public static void isTrue(boolean expression) {
75      isTrue(expression, "[Assertion failed] - this expression must be true");
76    }
77  
78    /**
79     * Assert that an object is <code>null</code> .
80     * <pre class="code">Assert.isNull(value, "The value must be null");</pre>
81     *
82     * @param object  the object to check
83     * @param message the exception message to use if the assertion fails
84     *
85     * @throws IllegalArgumentException if the object is not <code>null</code>
86     */
87    public static void isNull(Object object, String message) {
88      if (object != null) {
89        throw new IllegalArgumentException(message);
90      }
91    }
92  
93    /**
94     * Assert that an object is <code>null</code> .
95     * <pre class="code">Assert.isNull(value);</pre>
96     *
97     * @param object the object to check
98     *
99     * @throws IllegalArgumentException if the object is not <code>null</code>
100    */
101   public static void isNull(Object object) {
102     isNull(object, "[Assertion failed] - the object argument must be null");
103   }
104 
105   /**
106    * Assert that an object is not <code>null</code> .
107    * <pre class="code">Assert.notNull(clazz, "The class must not be null");</pre>
108    *
109    * @param object  the object to check
110    * @param message the exception message to use if the assertion fails
111    *
112    * @throws IllegalArgumentException if the object is <code>null</code>
113    */
114   public static void notNull(Object object, String message) {
115     if (object == null) {
116       throw new IllegalArgumentException(message);
117     }
118   }
119 
120   /**
121    * Assert that an object is not <code>null</code> .
122    * <pre class="code">Assert.notNull(clazz);</pre>
123    *
124    * @param object the object to check
125    *
126    * @throws IllegalArgumentException if the object is <code>null</code>
127    */
128   public static void notNull(Object object) {
129     notNull(object, "[Assertion failed] - this argument is required; it must not be null");
130   }
131 
132 
133   /**
134    * Assert that the given String has valid text content; that is, it must not
135    * be <code>null</code> and must contain at least one non-whitespace character.
136    * <pre class="code">Assert.hasText(name, "'name' must not be empty");</pre>
137    *
138    * @param text    the String to check
139    * @param message the exception message to use if the assertion fails
140    */
141   public static void hasText(String text, String message) {
142     if (Strings.isNullOrEmpty(text)) {
143       throw new IllegalArgumentException(message);
144     }
145   }
146 
147   /**
148    * Assert that the given String has valid text content; that is, it must not
149    * be <code>null</code> and must contain at least one non-whitespace character.
150    * <pre class="code">Assert.hasText(name, "'name' must not be empty");</pre>
151    *
152    * @param text the String to check
153    */
154   public static void hasText(String text) {
155     hasText(text,
156       "[Assertion failed] - this String argument must have text; it must not be <code>null</code>, empty, or blank");
157   }
158
