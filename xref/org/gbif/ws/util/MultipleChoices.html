3   import java.util.List;
4   import javax.annotation.Nullable;
5   
6   import static com.google.common.collect.Lists.newArrayList;

8   /**
9    * A simple bean for a http 300 MultipleChoices response listing the available options.
10   *
11   * @see <a href="https://gbif.basecamphq.com/projects/7935093-portal/todo_items/107116544/comments">Basecamp</a>
12   * @see <a href="http://tools.ietf.org/html/rfc2616#page-61">HTTP 1.1 specs</a>
13   */
14  public class MultipleChoices {
15  
16    private final List<Choice> choices = newArrayList();
17  
18    public List<Choice> getChoices() {
19      return choices;
20    }
21  
22    public void addChoice(Choice choice) {
23      choices.add(choice);
24    }
25  
26    public void addChoice(@Nullable Object key, String url, String title, @Nullable String description) {
27      addChoice(new Choice(key, url, title, description));
28    }
29  
30  }
