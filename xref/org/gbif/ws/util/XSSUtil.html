1   package org.gbif.ws.util;
2   
3   import java.util.regex.Matcher;
4   import java.util.regex.Pattern;
5   
6   import org.slf4j.Logger;
7   import org.slf4j.LoggerFactory;
8   
9   /**
10   * Class with utility methods for XSS filtering.
11   */
12  public class XSSUtil {
13    private static final Logger LOG = LoggerFactory.getLogger(XSSUtil.class);
14  
15    private XSSUtil() {
16      // empty private constructor
17    }
18  
19    private static final String NULL_CHAR = "\0";
20    private static final Pattern[] PATTERNS = new Pattern[] {
21      // Avoid anything in a <script> type of expression
22      Pattern.compile("<script>(.*?)</script>", Pattern.CASE_INSENSITIVE),
23      // Avoid anything in a src='...' type of expression
24      Pattern.compile("src[\r\n]*=[\r\n]*\\\'(.*?)\\\'", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL),
25      Pattern.compile("src[\r\n]*=[\r\n]*\\\"(.*?)\\\"", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL),
26      // Remove any lonesome </script> tag
27      Pattern.compile("</script>", Pattern.CASE_INSENSITIVE),
28      // Remove any lonesome <script ...> tag
29      Pattern.compile("<script(.*?)>", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL),
30      // Avoid eval(...) expressions
31      Pattern.compile("eval\\((.*?)\\)", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL),
32      // Avoid expression(...) expressions
33      Pattern.compile("expression\\((.*?)\\)", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL),
34      // Avoid javascript:... expressions
35      Pattern.compile("javascript:", Pattern.CASE_INSENSITIVE),
36      // Avoid vbscript:... expressions
37      Pattern.compile("vbscript:", Pattern.CASE_INSENSITIVE),
38      // Avoid onload= expressions
39      Pattern.compile("onload(.*?)=", Pattern.CASE_INSENSITIVE | Pattern.MULTILINE | Pattern.DOTALL)
40    };
41  
42    /**
43     * Method tests whether a string contains malicious XSS script or not.
44     *
45     * @param value decoded string to test
46     *
47     * @return true if string matches at least one XSS pattern, or false otherwise
48     */
49    public static boolean containsXSS(String value) {
50      if (value != null) {
51  
52        // Avoid null characters
53        value = value.replaceAll(NULL_CHAR, "");
54  
55        // Remove all sections that match a pattern
56        for (Pattern scriptPattern : PATTERNS) {
57          Matcher matcher = scriptPattern.matcher(value);
58          if (matcher.find()) {
59            LOG.warn("Malicious XSS script found: {}", value);
60            return true;
61          }
62        }
63      }
64      return false;
65    }
66  }
