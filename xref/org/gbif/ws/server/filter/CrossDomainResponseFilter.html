7   /**
8    * Filter that always adds a Access-Control-Allow-Origin: * header to the response
9    * that will make all GBIF webservices available for simple cross domain calls without JSONP.
10   *
11   * @see <a href="http://en.wikipedia.org/wiki/Cross-origin_resource_sharing">Wikipedia</a>
12   * @see <a href="http://www.cypressnorth.com/blog/programming/cross-domain-ajax-request-with-xml-response-for-iefirefoxchrome-safari-jquery/">Cross domain blog</a>
13   */
14  public class CrossDomainResponseFilter implements ContainerResponseFilter {
15  
16    @Override
17    public ContainerResponse filter(ContainerRequest request, ContainerResponse response) {
18      response.getHttpHeaders().putSingle("Access-Control-Allow-Origin", "*");
19      return response;
20    }
