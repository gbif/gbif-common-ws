1   package org.gbif.ws.server.filter;

3   import java.util.List;
4   
5   import com.sun.jersey.spi.container.ContainerRequest;
6   import com.sun.jersey.spi.container.ContainerResponse;
7   import com.sun.jersey.spi.container.ContainerResponseFilter;
8   
9   /**
10   * Filter that always adds a CORS related headers to the response
11   * that will make all GBIF webservices available for simple cross domain calls without JSONP.
12   *
13   *
14   * @see <a href="https://developer.mozilla.org/en-US/docs/Web/HTTP/Access_control_CORS">Mozilla documentation</a>
15   * @see <a href="http://en.wikipedia.org/wiki/Cross-origin_resource_sharing">Wikipedia</a>
16   * @see <a href="http://www.cypressnorth.com/blog/programming/cross-domain-ajax-request-with-xml-response-for-iefirefoxchrome-safari-jquery/">Cross domain blog</a>
17   */
18  public class CrossDomainResponseFilter implements ContainerResponseFilter {
19  
20    @Override
21    public ContainerResponse filter(ContainerRequest request, ContainerResponse response) {
22      response.getHttpHeaders().putSingle("Access-Control-Allow-Origin", "*");
23  
24      response.getHttpHeaders().putSingle("Access-Control-Allow-Methods", "HEAD, GET, POST, DELETE, PUT");
25  
26      //Used in response to a preflight request to indicate which HTTP headers can be used when making the actual request.
27      //https://developer.mozilla.org/en-US/docs/Web/HTTP/Access_control_CORS#Preflighted_requests
28      //we reflect the headers specified in the Access-Control-Request-Headers header of the request
29      if(request.getRequestHeaders().containsKey("Access-Control-Request-Headers")){
30        response.getHttpHeaders().put("Access-Control-Allow-Headers", (List)request.getRequestHeader("Access-Control-Request-Headers"));
31      }
32  
33      return response;
34    }
35  }
