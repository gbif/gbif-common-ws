1   package org.gbif.ws.server.filter;
2   
3   import java.net.HttpURLConnection;
4   import java.net.URI;
5   import java.util.UUID;
6   import javax.ws.rs.core.Response;
7   
8   import com.sun.jersey.spi.container.ContainerRequest;
9   import com.sun.jersey.spi.container.ContainerResponse;
10  import com.sun.jersey.spi.container.ContainerResponseFilter;
11  
12  /**
13   * Filter that updates http headers when a new resource is successfully created via a POST request unless
14   * the response returns 204 No Content.
15   * 
16   * The following headers are added or replaced if they existed:
17   * <ul>
18   *   <li>Http response code 201</li>
19   *   <li>Location header is set accordingly based on returned key</li>
20   * </ul>
21   */
22  public class CreatedResponseFilter implements ContainerResponseFilter {
23  
24    @Override
25    public ContainerResponse filter(ContainerRequest request, ContainerResponse response) {
26  
27      if (request.getMethod() != null && "post".equalsIgnoreCase(request.getMethod())
28              && response.getStatusType() != null
29              && !Response.Status.NO_CONTENT.equals(response.getStatusType())
30              && response.getStatusType().getFamily() == Response.Status.Family.SUCCESSFUL) {
31  
32        response.setStatus(HttpURLConnection.HTTP_CREATED);
33        // if response contains the key, also set Location
34        if (response.getEntity() != null) {
35          Object key = response.getEntity();
36          // we use POSTs also for non Create method which can return large objects, e.g. a list of parsed names
37          // only set the location header if the object is one of the following simple primary key data types
38          if (key instanceof Number || key instanceof UUID || key instanceof String) {
39            // allow POSTing to resource with or without trailing slash
40            URI location = request.getRequestUriBuilder().path(key.toString()).build();
41            response.getHttpHeaders().putSingle("Location", location.toString());
42          }
43  
44        }
45      }
46  
47      return response;
48    }
49  }
