12  /**
13   * Filter that updates http headers when a new resource is successfully created via a POST request.
14   * The following headers are added or replaced if they existed:
15   * <ul>
16   *   <li>Http response code 201</li>
17   *   <li>Location header is set accordingly based on returned key</li>
18   * </ul>
19   */
20  public class CreatedResponseFilter implements ContainerResponseFilter {
21  
22    @Override
23    public ContainerResponse filter(ContainerRequest request, ContainerResponse response) {
24  
25      if (request.getMethod() != null && request.getMethod().equalsIgnoreCase("post") && response.getStatusType() != null
26          && response.getStatusType().getFamily() == Response.Status.Family.SUCCESSFUL) {
27  
28        response.setStatus(HttpURLConnection.HTTP_CREATED);
29        // if response contains the key, also set Location
30        if (response.getEntity() != null) {
31          Object key = response.getEntity();
32          // we use POSTs also for non Create method which can return large objects, e.g. a list of parsed names
33          // only set the location header if the object is one of the following simple primary key data types
34          if (key instanceof Number || key instanceof UUID || key instanceof String) {
35            // allow POSTing to resource with or without trailing slash
36            URI location = request.getRequestUriBuilder().path(key.toString()).build();
37            response.getHttpHeaders().putSingle("Location", location.toString());
38          }
39  
40        }
41      }
42  
43      return response;
44    }
45  }
