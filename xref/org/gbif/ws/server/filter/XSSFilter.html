3   import org.gbif.api.model.common.messaging.Response;
4   import org.gbif.ws.util.XSSUtil;

6   import java.io.IOException;
7   import java.util.Enumeration;
8   import javax.servlet.Filter;
9   import javax.servlet.FilterChain;
10  import javax.servlet.FilterConfig;
11  import javax.servlet.ServletException;
12  import javax.servlet.ServletRequest;
13  import javax.servlet.ServletResponse;
14  import javax.servlet.http.HttpServletRequest;
15  import javax.servlet.http.HttpServletResponse;
16  
17  import com.google.inject.Singleton;
18  /**
19   * Request filter that detects XSS in paramater, header or querystring values and responds with a 400 bad request in such cases.
20   */
21  @Singleton
22  public class XSSFilter implements Filter {
23  
24    @Override
25    public void init(FilterConfig filterConfig) throws ServletException {
26    }
27  
28    @Override
29    public void destroy() {
30    }
31  
32    @Override
33    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain) throws IOException, ServletException {
34      if (request instanceof HttpServletRequest) {
35        HttpServletRequest httpRequest = (HttpServletRequest) request;
36        // test for xss in headers
37        checkParams(httpRequest.getHeaderNames(),httpRequest,response);
38  
39        // test for xss in querystring
40        // note: servlet container does not decode querystring when creating HttpServletRequest object
41        if (XSSUtil.containsXSS(httpRequest.getQueryString())) {
42          respondBadRequest(response);
43        }
44        // test for xss in parameters
45        // note: servlet container decodes parameter strings when creating HttpServletRequest object
46        checkParams(httpRequest.getParameterNames(),httpRequest,response);
47      }
48      chain.doFilter(request, response);
49    }
50  
51    /**
52     * Utility method that validates a enumeration of http parameters.
53     */
54    private static void checkParams(Enumeration<String> params, HttpServletRequest request,
55                                            ServletResponse response) throws IOException {
56      while(params.hasMoreElements()) {
57        if (XSSUtil.containsXSS(request.getParameter(params.nextElement()))) {
58          respondBadRequest(response);
59        }
60      }
61    }
62  
63    /**
64     * Sends a Response.StatusCode.BAD_REQUEST error response, provided the response hasn't already been committed.
65     *
66     * @param response response that will contain the error
67     *
68     * @throws IOException if there is a problem committing response
69     */
70    private static void respondBadRequest(ServletResponse response) throws IOException {
71      HttpServletResponse resp = (HttpServletResponse) response;
72      if (!resp.isCommitted()) {
73        resp.sendError(Response.StatusCode.BAD_REQUEST.getCode());
74      }
75    }
76  
77  }
