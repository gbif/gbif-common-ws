25  /**
26   * A filter that allows an application to identify itself as an application (as opposed to an application
27   * impersonating a user).
28   * In order to identify itself an application shall provide its appKey in the header x-gbif-user and sign the request
29   * accordingly.
30   * If the application can be authenticated AND its appKey is in the whitelist, the {@link Principal#getName()} will return the appKey and the
31   * role {@link AppRole#APP} will be assigned to it.
32   *
33   * We use an appKey whitelist to control which app should be allowed to have the {@link AppRole#APP} while letting
34   * the user impersonation available in {@link IdentityFilter}. If at some point multiple {@link AppRole} should be
35   * supported the whitelist should simply be changed for something more structured.
36   *
37   * This filter must run AFTER {@link IdentityFilter} if user impersonation using appKey is required.
38   * This filter will be skipped if the {@link ContainerRequest} already has a {@link Principal} attached.
39   * This filter operates on {@link GbifAuthService#GBIF_SCHEME} only.
40   * If the appKeyWhitelist list is not provided no apps will be authenticated by this filter.
41   *
42   */
43  public class AppIdentityFilter implements ContainerRequestFilter {
44  
45    public static final  String APPKEYS_WHITELIST = "identity.appkeys.whitelist";
46  
47    //FIXME should probably have its own scheme but that would requires to change {@link GbifAuthService)
48    private static final String GBIF_SCHEME_PREFIX = GbifAuthService.GBIF_SCHEME + " ";
49    private static final Logger LOG = LoggerFactory.getLogger(AppIdentityFilter.class);
50  
51    private final GbifAuthService authService;
52    private final List<String> appKeyWhitelist;
53  
54    @Inject
55    public AppIdentityFilter(@NotNull GbifAuthService authService, @Nullable @Named(APPKEYS_WHITELIST) List<String> appKeyWhitelist) {
56      this.authService = authService;
57      //defensive copy or creation
58      this.appKeyWhitelist = appKeyWhitelist != null ? new ArrayList<>(appKeyWhitelist) : new ArrayList<>();
59    }
60  
61    @Override
62    public ContainerRequest filter(final ContainerRequest containerRequest) {
63  
64      // Only try if no user principal is already there
65      if (containerRequest.getSecurityContext() != null && containerRequest.getUserPrincipal() != null) {
66        return containerRequest;
67      }
68  
69      String authorization = containerRequest.getHeaderValue(ContainerRequest.AUTHORIZATION);
70      if (StringUtils.startsWith(authorization, GBIF_SCHEME_PREFIX)) {
71        if (!authService.isValidRequest(containerRequest)) {
72          LOG.warn("Invalid GBIF authenticated request");
73          throw new WebApplicationException(Response.Status.UNAUTHORIZED);
74        }
75  
76        String username = containerRequest.getHeaderValue(GbifAuthService.HEADER_GBIF_USER);
77        String appKey = GbifAuthService.getAppKeyFromRequest(containerRequest::getHeaderValue);
78  
79        //check if it's an app by ensuring the appkey used to sign the request is the one used as x-gbif-user
80        if (StringUtils.equals(appKey, username) && appKeyWhitelist.contains(appKey)) {
81          containerRequest.setSecurityContext(new SecurityContext() {
82            private final ExtendedPrincipal principal = new AppPrincipal(appKey, AppRole.APP.name());
83  
84            @Override
85            public Principal getUserPrincipal() {
86              return principal;
87            }
88  
89            @Override
90            public boolean isUserInRole(String s) {
91              return principal.hasRole(s);
92            }
93  
94            @Override
95            public boolean isSecure() {
96              return containerRequest.isSecure();
97            }
98  
99            @Override
100           public String getAuthenticationScheme() {
101             return GbifAuthService.GBIF_SCHEME;
102           }
103         });
104       }
105     }
106     return containerRequest;
107   }
108 }
