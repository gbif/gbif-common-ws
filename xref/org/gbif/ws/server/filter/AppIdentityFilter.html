33  /**
34   * A filter that allows an application to identify itself as an application (as opposed to an
35   * application impersonating a user). In order to identify itself an application shall provide its
36   * appKey in the header x-gbif-user and sign the request accordingly. If the application can be
37   * authenticated AND its appKey is in the whitelist, the {@link Principal#getName()} will return the
38   * appKey and the role {@link AppRole#APP} will be assigned to it.
39   *
40   * <p>We use an appKey whitelist to control which app should be allowed to have the {@link
41   * AppRole#APP} while letting the user impersonation available in {@link IdentityFilter}. If at some
42   * point multiple {@link AppRole} should be supported the whitelist should simply be changed for
43   * something more structured.
44   *
45   * <p>This filter must run AFTER {@link IdentityFilter} if user impersonation using appKey is
46   * required. This filter will be skipped if the request already has a {@link Principal} attached.
47   * This filter operates on {@link SecurityConstants#GBIF_SCHEME} only. If the appKeyWhitelist list
48   * is not provided no apps will be authenticated by this filter.
49   */
50  @Component
51  public class AppIdentityFilter extends OncePerRequestFilter {
52  
53    private static final Logger LOG = LoggerFactory.getLogger(AppIdentityFilter.class);
54  
55    private final GbifAuthService authService;
56    private final List<String> appKeyWhitelist;
57  
58    public AppIdentityFilter(
59        @NotNull GbifAuthService authService, AppkeysConfigurationProperties appkeysConfiguration) {
60      this.authService = authService;
61      // defensive copy or creation
62      this.appKeyWhitelist =
63          appkeysConfiguration.getWhitelist() != null
64              ? new ArrayList<>(appkeysConfiguration.getWhitelist())
65              : new ArrayList<>();
66    }
67  
68    @Override
69    protected void doFilterInternal(
70        HttpServletRequest request, HttpServletResponse response, FilterChain filterChain)
71        throws ServletException, IOException {
72      final Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
73  
74      // Only try if no user principal is already there
75      if (authentication == null
76          || authentication.getPrincipal() == null
77          || authentication.getPrincipal() instanceof AnonymousUserPrincipal) {
78        String authorization = request.getHeader(HttpHeaders.AUTHORIZATION);
79        if (StringUtils.startsWith(authorization, SecurityConstants.GBIF_SCHEME_PREFIX)) {
80          if (authService.isValidRequest(
81              request instanceof GbifHttpServletRequestWrapper
82                  ? (GbifHttpServletRequestWrapper) request
83                  : new GbifHttpServletRequestWrapper(request))) {
84            String username = request.getHeader(SecurityConstants.HEADER_GBIF_USER);
85            String appKey = GbifAuthUtils.getAppKeyFromRequest(authorization);
86  
87            // check if it's an app by ensuring the appkey used to sign the request is the one used as
88            // x-gbif-user
89            if (StringUtils.equals(appKey, username) && appKeyWhitelist.contains(appKey)) {
90              final List<SimpleGrantedAuthority> authorities =
91                  Collections.singletonList(new SimpleGrantedAuthority(AppRole.APP.name()));
92              final AppPrincipal principal = new AppPrincipal(appKey, authorities);
93              final Authentication newAuthentication =
94                  new GbifAuthenticationToken(principal, SecurityConstants.GBIF_SCHEME, authorities);
95  
96              SecurityContextHolder.getContext().setAuthentication(newAuthentication);
97            }
98          } else {
99            LOG.warn("Invalid GBIF authenticated request");
100           response.setStatus(HttpStatus.UNAUTHORIZED.value());
101         }
102       }
103     }
104 
105     filterChain.doFilter(request, response);
106   }
107 }
