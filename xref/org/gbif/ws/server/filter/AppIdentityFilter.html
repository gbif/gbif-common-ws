51  /**
52   * A filter that allows an application to identify itself as an application (as opposed to an
53   * application impersonating a user). In order to identify itself an application shall provide its
54   * appKey in the header x-gbif-user and sign the request accordingly. If the application can be
55   * authenticated AND its appKey is in the whitelist, the {@link Principal#getName()} will return the
56   * appKey and the role {@link AppRole#APP} will be assigned to it.
57   *
58   * <p>We use an appKey whitelist to control which app should be allowed to have the {@link
59   * AppRole#APP} while letting the user impersonation available in {@link IdentityFilter}. If at some
60   * point multiple {@link AppRole} should be supported the whitelist should simply be changed for
61   * something more structured.
62   *
63   * <p>This filter must run AFTER {@link IdentityFilter} if user impersonation using appKey is
64   * required. This filter will be skipped if the request already has a {@link Principal} attached.
65   * This filter operates on {@link SecurityConstants#GBIF_SCHEME} only. If the appKeyWhitelist list
66   * is not provided no apps will be authenticated by this filter.
67   */
68  @Component
69  public class AppIdentityFilter extends OncePerRequestFilter {
70  
71    private static final Logger LOG = LoggerFactory.getLogger(AppIdentityFilter.class);
72  
73    private final GbifAuthService authService;
74    private final List<String> appKeyWhitelist;
75  
76    public AppIdentityFilter(
77        @NotNull GbifAuthService authService, AppkeysConfigurationProperties appkeysConfiguration) {
78      this.authService = authService;
79      // defensive copy or creation
80      this.appKeyWhitelist =
81          appkeysConfiguration.getWhitelist() != null
82              ? new ArrayList<>(appkeysConfiguration.getWhitelist())
83              : new ArrayList<>();
84    }
85  
86    @Override
87    protected void doFilterInternal(
88        HttpServletRequest request, HttpServletResponse response, FilterChain filterChain)
89        throws ServletException, IOException {
90      final Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
91  
92      // Only try if no user principal is already there
93      if (authentication == null
94          || authentication.getPrincipal() == null
95          || authentication.getPrincipal() instanceof AnonymousUserPrincipal) {
96        String authorization = request.getHeader(HttpHeaders.AUTHORIZATION);
97        if (StringUtils.startsWith(authorization, SecurityConstants.GBIF_SCHEME_PREFIX)) {
98          if (authService.isValidRequest(
99              request instanceof GbifHttpServletRequestWrapper
100                 ? (GbifHttpServletRequestWrapper) request
101                 : new GbifHttpServletRequestWrapper(request))) {
102           String username = request.getHeader(SecurityConstants.HEADER_GBIF_USER);
103           String appKey = GbifAuthUtils.getAppKeyFromRequest(authorization);
104 
105           // check if it's an app by ensuring the appkey used to sign the request is the one used as
106           // x-gbif-user
107           if (StringUtils.equals(appKey, username) && appKeyWhitelist.contains(appKey)) {
108             final List<SimpleGrantedAuthority> authorities =
109                 Collections.singletonList(new SimpleGrantedAuthority(AppRole.APP.name()));
110             final AppPrincipal principal = new AppPrincipal(appKey, authorities);
111             final Authentication newAuthentication =
112                 new GbifAuthenticationToken(principal, SecurityConstants.GBIF_SCHEME, authorities);
113 
114             SecurityContextHolder.getContext().setAuthentication(newAuthentication);
115           }
116         } else {
117           LOG.warn("Invalid GBIF authenticated request");
118           response.setStatus(HttpStatus.UNAUTHORIZED.value());
119         }
120       }
121     }
122 
123     filterChain.doFilter(request, response);
124   }
125 }
