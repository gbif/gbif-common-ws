3   import javax.validation.ConstraintViolation;
4   import javax.validation.ConstraintViolationException;
5   import javax.ws.rs.core.MediaType;
6   import javax.ws.rs.core.Response;
7   import javax.ws.rs.ext.ExceptionMapper;
8   import javax.ws.rs.ext.Provider;
9   
10  import com.google.common.base.Joiner;
11  import com.google.common.collect.ImmutableList;
12  import com.google.inject.Singleton;
13  import org.slf4j.Logger;
14  import org.slf4j.LoggerFactory;
15  
16  /**
17   * Converts a ConstraintViolationException into a http 422 bad request and gives a meaningful messages on the issues.
18   */
19  @Provider
20  @Singleton
21  public class ValidationExceptionMapper implements ExceptionMapper<ConstraintViolationException> {
22  
23    private static final int HTTP_CODE = 422;
24    private static Logger LOG = LoggerFactory.getLogger(ValidationExceptionMapper.class);
25  
26    @Override
27    public Response toResponse(ConstraintViolationException e) {
28      LOG.error(e.getMessage(), e);
29      ImmutableList.Builder<String> b = ImmutableList.builder();
30      for (ConstraintViolation<?> cv : e.getConstraintViolations()) {
31        LOG.debug("Validation of [{}] failed: {}", cv.getPropertyPath(), cv.getMessage());
32        b.add(String.format("Validation of [%s] failed: %s", cv.getPropertyPath(), cv.getMessage()));
33      }
34      return Response.status(HTTP_CODE).type(MediaType.TEXT_PLAIN)
35        .entity("<ul><li>" + Joiner.on("</li><li>").join(b.build()) + "</li></ul>").build();
36    }
37  }
