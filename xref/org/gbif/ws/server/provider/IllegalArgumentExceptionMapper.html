1   package org.gbif.ws.server.provider;

3   import javax.ws.rs.core.MediaType;
4   import javax.ws.rs.core.Response;
5   import javax.ws.rs.ext.ExceptionMapper;
6   import javax.ws.rs.ext.Provider;
7   
8   import com.google.inject.Singleton;
9   
10  /**
11   * Converts a IllegalArgumentException into a http 400 bad request.
12   */
13  @Provider
14  @Singleton
15  public class IllegalArgumentExceptionMapper implements ExceptionMapper<IllegalArgumentException> {
16  
17    @Override
18    public Response toResponse(IllegalArgumentException exception) {
19      return Response.status(Response.Status.BAD_REQUEST).type(MediaType.TEXT_PLAIN).entity(exception.getMessage()).build();
20    }
21  }
