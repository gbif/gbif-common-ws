67    /**
68     * Creates the module using default configuration, optionally installing authentication filters and customising the
69     * responses. Note that the request filters will be handled as follows if provided:
70     * <ol>
71     * <li>The default filters will always be first</li>
72     * <li>The authentication filter will be next if enabled (installAuth=true)</li>
73     * <li>The provided filters will be last in the given order</li>
74     * </ol>
75     * 
76     * @param resourcePackages To scan for the Jersey resources
77     * @param installAuth Controls whether the authentication filter should be configured
78     * @param responseFilters To customise the response if necessary
79     * @param requestFilters To apply in the chain of request filtering (see note above)
80     */
81    public WsJerseyModule(String resourcePackages, boolean installAuth,
82      @Nullable List<Class<? extends ContainerResponseFilter>> responseFilters,
83      List<Class<? extends ContainerRequestFilter>> requestFilters) {
84      // Let Jersey look for root resources automatically
85      this.resourcePackages = "org.gbif.ws.server.provider," + resourcePackages;
86      this.installAuth = installAuth;
87      if (responseFilters != null) {
88        this.responseFilters.addAll(responseFilters);
89      }
90      if (requestFilters != null) {
91        this.requestFilters.addAll(requestFilters);
92      }
93    }

95    @Override
96    protected void configureServlets() {
97      Map<String, String> params = new HashMap<String, String>(5);
98  
99      bind(JacksonJsonContextResolver.class);
100     params.put(JSONConfiguration.FEATURE_POJO_MAPPING, "true");
101 
102     // allow resources to use suffices to ask for media types
103     Map<String, String> mediaTypes = Maps.newHashMap();
104     mediaTypes.put("json", MediaType.APPLICATION_JSON);
105     mediaTypes.put("xml", MediaType.APPLICATION_XML);
106     mediaTypes.put("txt", MediaType.TEXT_PLAIN);
107     mediaTypes.put("zip", MediaType.APPLICATION_OCTET_STREAM);
108     params.put(ResourceConfig.PROPERTY_MEDIA_TYPE_MAPPINGS,
109       Joiner.on(", ").withKeyValueSeparator(" : ").join(mediaTypes));
110 
111     // Let Jersey look for root resources automatically
112     params.put(PackagesResourceConfig.PROPERTY_PACKAGES, resourcePackages);
113 
114     // secure resources via JSR 250
115     params.put(ResourceFilters.class.getName(), RolesAllowedResourceFilterFactory.class.getName());
116 
117     // The default filters are always added to start of the chain followed by authentication
118     if (installAuth) {
119       requestFilters.addFirst(AuthFilter.class);
120     }
121     requestFilters.addFirst(RequestHeaderParamUpdateFilter.class);
122     params.put(ResourceConfig.PROPERTY_CONTAINER_REQUEST_FILTERS,
123       Joiner.on(";").join(Lists.transform(requestFilters, fnClassName)));
124 
125     // response filters
126     List<Class<? extends ContainerResponseFilter>> respFilters = Lists.newArrayList();
127     respFilters.add(JsonpResponseFilter.class);
128     respFilters.add(CreatedResponseFilter.class);
129     respFilters.add(CrossDomainResponseFilter.class);
130     if (installAuth) {
131       respFilters.add(AuthResponseFilter.class);
132     }
133     respFilters.addAll(responseFilters);
134     params.put(ResourceConfig.PROPERTY_CONTAINER_RESPONSE_FILTERS,
135       Joiner.on(";").join(Lists.transform(respFilters, fnClassName)));
136 
137     serve("/*").with(GuiceContainer.class, params);
138 
139     bindInterceptor(Matchers.any(), Matchers.annotatedWith(NullToNotFound.class), new NullToNotFoundInterceptor());
