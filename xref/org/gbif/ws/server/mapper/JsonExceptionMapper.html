3   import com.fasterxml.jackson.core.JsonParseException;
4   import com.fasterxml.jackson.core.JsonProcessingException;
5   import com.fasterxml.jackson.databind.JsonMappingException;
6   import com.fasterxml.jackson.databind.RuntimeJsonMappingException;
7   import org.springframework.http.HttpStatus;
8   import org.springframework.http.MediaType;
9   import org.springframework.http.ResponseEntity;
10  import org.springframework.web.bind.annotation.ControllerAdvice;
11  import org.springframework.web.bind.annotation.ExceptionHandler;
12  
13  /**
14   * Converts json related exceptions into a http 400 bad request.
15   */
16  @ControllerAdvice
17  public class JsonExceptionMapper {
18  
19    @ExceptionHandler({
20        JsonMappingException.class,
21        JsonParseException.class,
22        JsonProcessingException.class,
23        RuntimeJsonMappingException.class})
24    public ResponseEntity<Object> toResponse(Exception exception) {
25      return ResponseEntity.status(HttpStatus.BAD_REQUEST)
26          .contentType(MediaType.TEXT_PLAIN)
27          .body(exception.getMessage());
28    }
29  }
