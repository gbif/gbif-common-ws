20  /**
21   * Base web service search client supporting a {@link FacetedSearchRequest}.
22   * 
23   * @param <T> type of the response content
24   * @param <P> search parameter type
25   * @param <R> search request type
26   */
27  public abstract class BaseWsFacetedSearchClient<T, P extends Enum<?> & SearchParameter, R extends FacetedSearchRequest<P>>
28    extends BaseWsSearchClient<T, P, R> implements SearchService<T, P, R> {
29  
30    protected BaseWsFacetedSearchClient(WebResource resource, GenericType<SearchResponse<T, P>> gType) {
31      super(resource, gType);
32    }
33  
34    /**
35     * Adds support for faceted search request parameters.
36     */
37    @Override
38    protected MultivaluedMap<String, String> getParameterFromRequest(@Nullable R searchRequest) {
39      // The searchRequest is transformed in a parameter map
40      MultivaluedMap<String, String> parameters = super.getParameterFromRequest(searchRequest);
41  
42      if (searchRequest != null) {
43        parameters.putSingle(PARAM_FACET_MULTISELECT, Boolean.toString(searchRequest.isMultiSelectFacets()));
44        if (searchRequest.getFacetMinCount() != null) {
45          parameters.putSingle(PARAM_FACET_MINCOUNT, Integer.toString(searchRequest.getFacetMinCount()));
46        }
47        if (searchRequest.getFacets() != null) {
48          for (P facet : searchRequest.getFacets()) {
49            parameters.add(PARAM_FACET, facet.name());
50          }
51        }
52      }
53  
54      return parameters;
55    }
56  }
