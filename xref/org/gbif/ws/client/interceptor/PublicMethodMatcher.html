1   package org.gbif.ws.client.interceptor;

3   import java.io.Serializable;
4   import java.lang.reflect.Method;
5   import java.lang.reflect.Modifier;
6   
7   import com.google.inject.matcher.AbstractMatcher;
8   
9   /**
10   * Matches public, non synthetic methods only which are no Object methods, i.e. equals, hashCode and toString.
11   */
12  public class PublicMethodMatcher extends AbstractMatcher<Object> implements Serializable {
13  
14    private static final long serialVersionUID = 0;
15  
16    @Override
17    public boolean matches(Object o) {
18      if (o instanceof Method) {
19        Method m = (Method) o;
20        if (Modifier.isPublic(m.getModifiers()) && (m.isBridge() || !m.isSynthetic())) {
21          for (Method om : Object.class.getMethods()) {
22            if (om.getName().equals(m.getName()) && om.getParameterTypes().length == m.getParameterTypes().length) {
23              return false;
24            }
25          }
26          return true;
27        }
28      }
29      return false;
30    }
31  }
