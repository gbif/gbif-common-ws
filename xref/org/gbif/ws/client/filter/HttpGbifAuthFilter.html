40  package org.gbif.ws.client.filter;
41  
42  import org.gbif.ws.security.GbifAppAuthService;
43  
44  import java.security.Principal;
45  
46  import com.google.inject.Provider;
47  import com.sun.jersey.api.client.ClientHandlerException;
48  import com.sun.jersey.api.client.ClientRequest;
49  import com.sun.jersey.api.client.ClientResponse;
50  import com.sun.jersey.api.client.filter.ClientFilter;
51  
52  /**
53   * Thread safe client filter adding an HTTP Authentication header to the HTTP request using the custom GBIF schema
54   * for trusted applications.
55   * In addition to the Authentication this filter will add these headers to the request:
56   * <ul>
57   * <li>Content-MD5: the MD5 hash for the request body</li>
58   * <li>x-gbif-user: the username of the proxied user</li>
59   * <li>x-gbif-date: the current datetime stamp</li>
60   * </ul>
61   * The custom date header is added in addition to the regular date header because it is not possible
62   * to access the generated date header in a jersey ClientFilter. The regular date header is generated by
63   * the RequestDate HttpRequestInterceptor of HttpClient and is therefore not available to jersey client.
64   * A current timestamp is important for signing the request though.
65   */
66  public final class HttpGbifAuthFilter extends ClientFilter {
67  
68    private final String applicationId;
69    private final Provider<Principal> principalProvider;
70    private final GbifAppAuthService authService;
71  
72    /**
73     * Creates a new HTTP GBIF Authentication filter using provided application ID and trusted principal provider.
74     */
75    public HttpGbifAuthFilter(final String applicationId, GbifAppAuthService authService,
76      Provider<Principal> principalProvider) {
77      this.applicationId = applicationId;
78      this.principalProvider = principalProvider;
79      this.authService = authService;
80    }
81  
82    @Override
83    public ClientResponse handle(final ClientRequest cr) throws ClientHandlerException {
84  
85      if (principalProvider != null) {
86        Principal prince = principalProvider.get();
87        if (prince != null) {
88          authService.signRequest(applicationId, prince.getName(), cr);
89        }
90      }
91      return getNext().handle(cr);
92    }
93  
94  }
