1   package org.gbif.ws.client;
2   
3   import feign.MethodMetadata;
4   import feign.Util;
5   import java.lang.annotation.Annotation;
6   import java.lang.reflect.Method;
7   import java.util.Collection;
8   import org.gbif.api.annotation.PartialDate;
9   import org.springframework.cloud.openfeign.AnnotatedParameterProcessor;
10  
11  /**
12   * Process method arguments annotated with annotation {@link PartialDate}.
13   * Should be used instead of {@link org.springframework.web.bind.annotation.RequestParam}.
14   * Only for date parameters!
15   */
16  public class PartialDateParameterProcessor implements AnnotatedParameterProcessor {
17  
18    private static final Class<PartialDate> ANNOTATION = PartialDate.class;
19  
20    private PartialDateExpander partialDateExpander = new PartialDateExpander();
21  
22    @Override
23    public Class<? extends Annotation> getAnnotationType() {
24      return ANNOTATION;
25    }
26  
27    @Override
28    public boolean processArgument(
29        AnnotatedParameterContext context, Annotation annotation, Method method) {
30      int parameterIndex = context.getParameterIndex();
31      MethodMetadata data = context.getMethodMetadata();
32      PartialDate requestParam = ANNOTATION.cast(annotation);
33      String name = requestParam.value();
34  
35      Util.checkState(Util.emptyToNull(name) != null,
36          "PartialDate.value() was null");
37  
38      context.setParameterName(name);
39  
40      data.indexToExpander().put(parameterIndex, partialDateExpander);
41      Collection<String> query =
42          context.setTemplateParameter(name, data.template().queries().get(name));
43      data.template().query(name, query);
44  
45      return true;
46    }
47  }
