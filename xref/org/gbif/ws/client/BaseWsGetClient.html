14  /**
15   * Base client providing a get by key method for a specific entity type T.
16   * The client applies a json media type and provides various convenience methods to execute CRUD requests
17   * and can be configured to work with authentication.
18   * The client can return other types than the main T class via generic types passed in.
19   *
20   * @param <T> the main entity class
21   */

27    /**
28     * @param resourceClass the main entity class
29     * @param resource      the base url to the underlying webservice
30     * @param authFilter    optional authentication filter, can be null
31     */
32    protected BaseWsGetClient(Class<T> resourceClass, WebResource resource, @Nullable ClientFilter authFilter) {
33      super(resource);
34      this.resourceClass = resourceClass;
35      if (authFilter != null) {
36        this.resource.addFilter(authFilter);
37      }
38    }

40    /**
41     * Gets a resource by its key by executing a json type http GET request.
42     * The clients resourceClass is used to convert the response into an object.
43     *
44     * @return requested resource or {@code null} if it couldn't be found
45     */
46    @Nullable
47    public T get(K key) {
48      if (key == null) {

50      }
51      return resource.path(key.toString()).type(MediaType.APPLICATION_JSON).get(resourceClass);
52    }

59     *
60     * @return requested resource or null

87     * Also applies a language http header.
88     * The clients resourceClass is used to convert the response into an object.

91     * @param path   optional paths to be appended to the base resource.

94     */
95    @Nullable
96    protected T get(Locale locale, String... path) {
97      WebResource.Builder res = getResource(path).type(MediaType.APPLICATION_JSON);
98      if (locale != null) {
99        res = res.acceptLanguage(locale);
100     }
101     return res.get(resourceClass);

114   /**
115    * Convert the http entity into a byte array to avoid jackson creating a chunked encoding request!
116    */
117   private byte[] toBytes(Object entity) {
118     try {
119       return mapper.writeValueAsBytes(entity);

121       log.error("Failed to serialize http entity [{}]", entity);
