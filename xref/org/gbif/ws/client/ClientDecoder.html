1   package org.gbif.ws.client;
2   
3   import com.fasterxml.jackson.databind.ObjectMapper;
4   import feign.FeignException;
5   import feign.Response;
6   import feign.Util;
7   import feign.codec.DecodeException;
8   import feign.codec.Decoder;
9   import feign.jackson.JacksonDecoder;
10  import java.io.IOException;
11  import java.lang.reflect.Type;
12  
13  public class ClientDecoder implements Decoder {
14  
15    private JacksonDecoder jacksonDecoder;
16  
17    public ClientDecoder(ObjectMapper objectMapper) {
18      this.jacksonDecoder = new JacksonDecoder(objectMapper);
19    }
20  
21    @Override
22    public Object decode(Response response, Type type)
23        throws IOException, DecodeException, FeignException {
24      if (byte[].class.equals(type)) {
25        return Util.toByteArray(response.body().asInputStream());
26      } else {
27        return jacksonDecoder.decode(response, type);
28      }
29    }
30  }
