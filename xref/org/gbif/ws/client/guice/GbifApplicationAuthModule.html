9   import com.google.inject.AbstractModule;
10  import com.sun.jersey.api.client.filter.ClientFilter;
11  
12  /**
13   * An authentication module for trusted GBIF applications.
14   * It requires an application key and can then act on behalf of any username.
15   *
16   * To set the proxied username you can either set it directly on the module instance
17   * or inject the SimplePrincipalProvider into your code anywhere to set the current user:
18   *
19   * <pre>
20   * {@code
21   * GbifApplicationAuthModule authMod = new GbifApplicationAuthModule();
22   * authMod.setPrincipal("admin");
23   *
24   * // set user via SimplePrincipalProvider:
25   *
26   * {@literal @}Inject
27   * SimplePrincipalProvider pp;
28   * pp.setPrincipal("admin");
29   * }
30   * </pre>
31   *
32   * Note that the users role still depends on the roles defined in the user account in Drupal and will not default
33   * to ADMIN. The user also has to exist in Drupal for the webservice authorization to work.
34   *
35   */
36  // TODO: Make the application key configurable (Requires significant changes in the webservice AuthModule)
37  public class GbifApplicationAuthModule extends AbstractModule {
38  
39    public static final String PROPERTY_APP_KEY = "application.key";
40    public static final String PROPERTY_APP_SECRET = "application.secret";
41  
42    private final SimplePrincipalProvider pp = new SimplePrincipalProvider();
43    private final GbifAuthService authService;
44  
45    /**
46     * Creates a new authentication guice module for a trusted application with its public and secret key.
47     * The proxied user will be set to match the application key.
48     *
49     * @param appKey       the trusted application public key
50     * @param appSecretKey the secret key issued for this application
51     */
52    public GbifApplicationAuthModule(String appKey, String appSecretKey) {
53      authService = GbifAuthService.singleKeyAuthService(appKey, appSecretKey);
54      pp.setPrincipal(appKey);
55    }
56  
57    /**
58     * Creates a new authentication guice module for a trusted application with its id and secret key given as
59     * properties.
60     * The proxied user will be set to match the application.key.
61     *
62     * @param properties that holds an entry for application.key and application.secret
63     */
64    public GbifApplicationAuthModule(Properties properties) {
65      this(properties.getProperty(PROPERTY_APP_KEY), properties.getProperty(PROPERTY_APP_SECRET));
66    }
67  
68    @Override
69    protected void configure() {
70      bind(SimplePrincipalProvider.class).toInstance(pp);
71      bind(GbifAuthService.class).toInstance(authService);
72      HttpGbifAuthFilter authFilter = new HttpGbifAuthFilter(authService, pp);
73      bind(ClientFilter.class).toInstance(authFilter);
74    }
75  
76    public void setPrincipal(String username) {
77      pp.setPrincipal(username);
78    }
79  }
