12  /**
13   * An authentication module for trusted GBIF applications.
14   * It requires an application key and can then act on behalf of any username.
15   *
16   * To set the proxied username you can either set it directly on the module instance
17   * or inject the SimplePrincipalProvider into your code anywhere to set the current user:
18   *
19   * <pre>
20   * {@code
21   * GbifApplicationAuthModule authMod = new GbifApplicationAuthModule();
22   * authMod.setPrincipal("admin");
23   *
24   * // set user via SimplePrincipalProvider:
25   *
26   * {@literal @}Inject
27   * SimplePrincipalProvider pp;
28   * pp.setPrincipal("admin");
29   * }
30   * </pre>
31   *
32   * Note that the users role still depends on the roles defined in the user account in Drupal and will not default
33   * to ADMIN. The user also has to exist in Drupal for the webservice authorization to work.
34   *
35   */
36  // TODO: Make the application key configurable (Requires significant changes in the webservice AuthModule)
37  public class GbifApplicationAuthModule extends AbstractModule {
38  
39    public static final String PROPERTY_APP_KEY = "application.key";
40    public static final String PROPERTY_APP_SECRET = "application.secret";
41  
42    private final SimplePrincipalProvider pp = new SimplePrincipalProvider();
43    private final GbifAppAuthService authService;
44    private final String applicationID;
45  
46    /**
47     * Creates a new authentication guice module for a trusted application with its id and secret key.
48     * The proxied user will be set to match the application ID.
49     *
50     * @param appId        the trusted application id
51     * @param appSecretKey the secret key issued for this application
52     */
53    public GbifApplicationAuthModule(String appId, String appSecretKey) {
54      applicationID = appId;
55      authService = new GbifAppAuthService(appId, appSecretKey);
56      pp.setPrincipal(appId);
57    }
58  
59    /**
60     * Creates a new authentication guice module for a trusted application with its id and secret key given as
61     * properties.
62     * The proxied user will be set to match the application.key.
63     *
64     * @param properties that holds an entry for application.key and application.secret
65     */
66    public GbifApplicationAuthModule(Properties properties) {
67      this(properties.getProperty(PROPERTY_APP_KEY), properties.getProperty(PROPERTY_APP_SECRET));
68    }
69  
70    @Override
71    protected void configure() {
72      bind(SimplePrincipalProvider.class).toInstance(pp);
73      bind(GbifAppAuthService.class).toInstance(authService);
74      HttpGbifAuthFilter authFilter = new HttpGbifAuthFilter(applicationID, authService, pp);
75      bind(ClientFilter.class).toInstance(authFilter);
76    }
77  
78    public void setPrincipal(String username) {
79      pp.setPrincipal(username);
80    }
81  }
