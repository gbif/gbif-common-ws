103   @Provides
104   @Singleton
105   public HttpClient provideHttpClient() {
106     int httpTimeout = DEFAULT_HTTP_TIMEOUT_MSECS;
107     int maxHttpConnections = DEFAULT_MAX_HTTP_CONNECTIONS;
108     int maxHttpConnectionsPerRoute = DEFAULT_MAX_HTTP_CONNECTIONS_PER_ROUTE;
109 
110     if (properties.containsKey(HttpClientConnParams.HTTP_TIMEOUT)) {
111       httpTimeout = Integer.parseInt(properties.getProperty(HttpClientConnParams.HTTP_TIMEOUT));
112     }
113     if (properties.containsKey(HttpClientConnParams.MAX_HTTP_CONNECTIONS)) {
114       maxHttpConnections = Integer.parseInt(properties.getProperty(HttpClientConnParams.MAX_HTTP_CONNECTIONS));
115     }
116     if (properties.containsKey(HttpClientConnParams.MAX_HTTP_CONNECTIONS_PER_ROUTE)) {
117       maxHttpConnectionsPerRoute =
118         Integer.parseInt(properties.getProperty(HttpClientConnParams.MAX_HTTP_CONNECTIONS_PER_ROUTE));
119     }
120     return HttpUtil.newMultithreadedClient(httpTimeout, maxHttpConnections, maxHttpConnectionsPerRoute);
121   }
