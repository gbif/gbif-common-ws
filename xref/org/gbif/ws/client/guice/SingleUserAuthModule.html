1   package org.gbif.ws.client.guice;
2   
3   import com.google.inject.AbstractModule;
4   import com.sun.jersey.api.client.filter.ClientFilter;
5   import com.sun.jersey.api.client.filter.HTTPBasicAuthFilter;
6   
7   /**
8    * A small module that installs a {@link com.sun.jersey.api.client.filter.HTTPBasicAuthFilter}
9    * for a single, fixed user.
10   *
11   * If application need to write to webservices, e.g the registry, you need to make sure the user
12   * has admin rights in drupal.
13   *
14   * @see <a href="http://staging.gbif.org/drupal/user/register">register a new user in drupal</a>
15   */
16  public class SingleUserAuthModule extends AbstractModule {
17  
18    private final HTTPBasicAuthFilter filter;
19  
20    public SingleUserAuthModule(String user, String password) {
21      filter = new HTTPBasicAuthFilter(user, password);
22    }
23  
24    @Override
25    protected void configure() {
26      bind(ClientFilter.class).toInstance(filter);
27    }
28  }
