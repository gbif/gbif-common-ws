25   *

27   * @param <P> search parameter type

35    /**
36     * @param resource the base web resource. /search will be added
37     * @param gType the generic type for the search response.
38     */
39    protected BaseWsSearchClient(WebResource resource, GenericType<SearchResponse<T, P>> gType) {
40      super(resource);
41      this.gType = gType;
42    }

45     * Generic search operation.
46     * 
47     * @param searchRequest to issue the search operation. If null matches everything
48     * @return a SearchResponse object with the operation result.

50    public SearchResponse<T, P> search(@Nullable R searchRequest) {
51      return get(gType, null, getParameterFromRequest(searchRequest), searchRequest, "search");
52    }

54    /**
55     * Transforms the attributes of the {@link SearchRequest} parameter into a map of parameters.
56     * The output of this method is intended to be used with the operation WebResource.queryParams(parameters).
57     * 
58     * @return a {@link MultivaluedMap} containing the HTTP parameters taken from the {@link SearchRequest} object.
59     */
60    protected MultivaluedMap<String, String> getParameterFromRequest(@Nullable R searchRequest) {
61      return getParameterFromSearchRequest(searchRequest);
62    }
63  
64    protected MultivaluedMap<String, String> getParameterFromSearchRequest(@Nullable SearchRequest<P> searchRequest) {
65  
66      // The searchRequest is transformed in a parameter map
67      MultivaluedMap<String, String> parameters = new MultivaluedMapImpl();
68  
69      if (searchRequest == null) {
70        parameters.putSingle(PARAM_QUERY_STRING, DEFAULT_SEARCH_PARAM_VALUE);
71  
72      } else {
73        String searchParamValue = searchRequest.getQ();
74        if (Strings.isNullOrEmpty(searchParamValue)) {
75          searchParamValue = DEFAULT_SEARCH_PARAM_VALUE;
76        }
77        parameters.putSingle(PARAM_QUERY_STRING, searchParamValue);
78        parameters.putSingle(PARAM_HIGHLIGHT, Boolean.toString(searchRequest.isHighlight()));
79  
80        Multimap<P, String> requestParameters = searchRequest.getParameters();
81        if (requestParameters != null) {
82          for (P param : requestParameters.keySet()) {
83            parameters.put(param.name(), Lists.newArrayList(requestParameters.get(param)));
84          }
85        }
86      }
87      return parameters;
88    }
