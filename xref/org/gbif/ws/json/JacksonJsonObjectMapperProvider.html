1   package org.gbif.ws.json;

3   import com.fasterxml.jackson.annotation.JsonInclude;
4   import com.fasterxml.jackson.databind.DeserializationFeature;
5   import com.fasterxml.jackson.databind.ObjectMapper;
6   import com.fasterxml.jackson.databind.SerializationFeature;
7   import org.gbif.api.ws.mixin.Mixins;
8   
9   public final class JacksonJsonObjectMapperProvider {
10  
11    private JacksonJsonObjectMapperProvider() {
12    }
13  
14    public static ObjectMapper getObjectMapper() {
15      final ObjectMapper objectMapper = new ObjectMapper();
16      objectMapper.setSerializationInclusion(JsonInclude.Include.NON_NULL);
17      // determines whether encountering of unknown properties (ones that do not map to a property,
18      // and there is no
19      // "any setter" or handler that can handle it) should result in a failure (throwing a
20      // JsonMappingException) or not.
21      objectMapper.disable(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES);
22  
23      // Enforce use of ISO-8601 format dates (http://wiki.fasterxml.com/JacksonFAQDateHandling)
24      objectMapper.configure(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS, false);
25  
26      Mixins.getPredefinedMixins().forEach(objectMapper::addMixIn);
27  
28      return objectMapper;
29    }
30  }
