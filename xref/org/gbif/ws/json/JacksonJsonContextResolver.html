3   import java.util.Map;

5   import javax.ws.rs.Produces;
6   import javax.ws.rs.core.MediaType;
7   import javax.ws.rs.ext.ContextResolver;
8   import javax.ws.rs.ext.Provider;
9   
10  import com.google.inject.Singleton;
11  import org.codehaus.jackson.map.DeserializationConfig;
12  import org.codehaus.jackson.map.ObjectMapper;
13  import org.codehaus.jackson.map.SerializationConfig;
14  import org.codehaus.jackson.map.annotate.JsonSerialize;
15  
16  /**
17   * Provider that initializes the {@link ObjectMapper} to ignore {@code null} fields.
18   * NOTE! This class has the same name as a class provided by jersey, BUT THEY DO DIFFERENT THINGS! This
19   * class depends on the presence of the jersey version in order to do anything useful. If you get errors that
20   * suggest that you don't have an appropriate MessageBodyReader for the MIME type application/json, then chances
21   * are good that you have not included the jersey-provided JacksonJsonProvider in your configuration.
22   */
23  @Provider
24  @Produces(MediaType.APPLICATION_JSON)
25  @Singleton
26  public class JacksonJsonContextResolver implements ContextResolver<ObjectMapper> {
27  
28    protected static final ObjectMapper MAPPER = new ObjectMapper();
29  
30    static {
31      MAPPER.setSerializationInclusion(JsonSerialize.Inclusion.NON_NULL);
32      // determines whether encountering of unknown properties (ones that do not map to a property, and there is no
33      // "any setter" or handler that can handle it) should result in a failure (throwing a JsonMappingException) or not.
34      MAPPER.disable(DeserializationConfig.Feature.FAIL_ON_UNKNOWN_PROPERTIES);
35      
36      // Enforce use of ISO-8601 format dates (http://wiki.fasterxml.com/JacksonFAQDateHandling)
37      MAPPER.configure(SerializationConfig.Feature.WRITE_DATES_AS_TIMESTAMPS, false);
38    }
39  
40    /**
41     * Specific polymorphic serialization is supported by passing a map of mixIns into the class.
42     * 
43     * @param mixIns class map
44     */
45    public static void addMixIns(Map<Class<?>, Class<?>> mixIns) {
46      // handle polymorphic JSON
47      for (Map.Entry<Class<?>, Class<?>> classClassEntry : mixIns.entrySet()) {
48        MAPPER.getSerializationConfig().addMixInAnnotations(classClassEntry.getKey(), classClassEntry.getValue());
49        MAPPER.getDeserializationConfig().addMixInAnnotations(classClassEntry.getKey(), classClassEntry.getValue());
50      }
51    }
52  
53    @Override
54    public ObjectMapper getContext(Class<?> type) {
55      return MAPPER;
56    }
57  }
