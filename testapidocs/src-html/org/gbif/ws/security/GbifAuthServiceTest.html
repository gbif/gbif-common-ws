001package org.gbif.ws.security;
002
003import org.gbif.ws.json.JacksonJsonObjectMapperProvider;
004import org.gbif.ws.server.GbifHttpServletRequestWrapper;
005import org.junit.Before;
006import org.junit.Test;
007import org.junit.runner.RunWith;
008import org.mockito.junit.MockitoJUnitRunner;
009import org.springframework.http.HttpHeaders;
010
011import java.net.URI;
012import java.net.URL;
013import java.nio.file.Paths;
014
015import static org.gbif.ws.util.SecurityConstants.HEADER_CONTENT_MD5;
016import static org.gbif.ws.util.SecurityConstants.HEADER_GBIF_USER;
017import static org.junit.Assert.assertEquals;
018import static org.junit.Assert.assertFalse;
019import static org.junit.Assert.assertNotNull;
020import static org.junit.Assert.assertTrue;
021import static org.mockito.Mockito.mock;
022import static org.mockito.Mockito.when;
023import static org.springframework.http.HttpHeaders.AUTHORIZATION;
024
025@RunWith(MockitoJUnitRunner.class)
026public class GbifAuthServiceTest {
027
028  public static final String APPKEY = "appKey";
029
030  private URI testUri;
031  private GbifAuthService service;
032
033  @Before
034  public void setUp() throws Exception {
035    testUri = new URI("http://api.gbif.org/v1/dataset");
036    service = prepareGbifAuthService();
037  }
038
039  public static GbifAuthService prepareGbifAuthService() throws Exception {
040    URL resource = GbifAuthServiceTest.class.getClassLoader().getResource("appkeys.properties");
041    String stringPath = Paths.get(resource.toURI()).toAbsolutePath().toString();
042
043    AppkeysConfigurationProperties mockAppkeysConfiguration = mock(AppkeysConfigurationProperties.class);
044    when(mockAppkeysConfiguration.getFile()).thenReturn(stringPath);
045
046    return new GbifAuthServiceImpl(
047        new AppKeySigningService(new FileSystemKeyStore(mockAppkeysConfiguration)),
048        new Md5EncodeServiceImpl(JacksonJsonObjectMapperProvider.getObjectMapper()),
049        () -> APPKEY
050    );
051  }
052
053  @Test
054  public void testSignRequest() {
055    // given
056    GbifHttpServletRequestWrapper requestObjectMock = mock(GbifHttpServletRequestWrapper.class);
057    HttpHeaders headers = new HttpHeaders();
058    when(requestObjectMock.getRequestURI()).thenReturn(testUri.toString());
059    when(requestObjectMock.getHttpHeaders()).thenReturn(headers);
060    when(requestObjectMock.getMethod()).thenReturn("POST");
061    when(requestObjectMock.getContent()).thenReturn("Simsalabim");
062
063    // when
064    service.signRequest("heinz", requestObjectMock);
065
066    // then
067    assertNotNull(headers.getFirst(HEADER_CONTENT_MD5));
068    assertEquals("heinz", headers.getFirst(HEADER_GBIF_USER));
069    assertNotNull(headers.getFirst(AUTHORIZATION));
070    assertTrue(headers.getFirst(AUTHORIZATION).startsWith("GBIF appKey:"));
071  }
072
073  @Test
074  public void testIsValid() {
075    // given
076    GbifHttpServletRequestWrapper requestObjectMock = mock(GbifHttpServletRequestWrapper.class);
077    HttpHeaders headers = new HttpHeaders();
078    when(requestObjectMock.getRequestURI()).thenReturn(testUri.toString());
079    when(requestObjectMock.getHttpHeaders()).thenReturn(headers);
080    when(requestObjectMock.getMethod()).thenReturn("POST");
081    when(requestObjectMock.getContent()).thenReturn("Simsalabim");
082    when(requestObjectMock.getHeader(AUTHORIZATION)).thenCallRealMethod();
083    service.signRequest("heinz", requestObjectMock);
084
085    // when
086    boolean isRequestValidCorrectContentActual = service.isValidRequest(requestObjectMock);
087    headers.set(HEADER_CONTENT_MD5, "73");
088    boolean isRequestValidWrongContentActual = service.isValidRequest(requestObjectMock);
089
090    // then
091    assertTrue(isRequestValidCorrectContentActual);
092    assertFalse(service.isValidRequest(requestObjectMock));
093  }
094
095  @Test
096  public void testGetAppKeyFromRequest() {
097    // given
098    GbifHttpServletRequestWrapper requestObjectMock = mock(GbifHttpServletRequestWrapper.class);
099    HttpHeaders headers = new HttpHeaders();
100    when(requestObjectMock.getRequestURI()).thenReturn(testUri.toString());
101    when(requestObjectMock.getHttpHeaders()).thenReturn(headers);
102    when(requestObjectMock.getMethod()).thenReturn("POST");
103    when(requestObjectMock.getContent()).thenReturn("Simsalabim");
104    when(requestObjectMock.getHeader(AUTHORIZATION)).thenCallRealMethod();
105
106    // when
107    service.signRequest("heinz", requestObjectMock);
108    String actualAppKey = GbifAuthUtils.getAppKeyFromRequest(requestObjectMock.getHeader(AUTHORIZATION));
109
110    // then
111    assertEquals(APPKEY, actualAppKey);
112  }
113}
