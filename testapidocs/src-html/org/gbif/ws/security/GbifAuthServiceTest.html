073    }
074

078    }
079
080    @Override

083    }
084
085    @Override
086    public boolean isEmpty() {
087      return map.isEmpty();
088    }
089
090    @Override
091    public boolean containsKey(Object o) {
092      return map.containsKey(o);
093    }
094
095    @Override
096    public boolean containsValue(Object o) {
097      return map.containsValue(o);
098    }
099
100    @Override

102      final List<Object> src = map.get(o);
103      if (src == null) {
104        return null;
105      }
106      List<String> l = Lists.newArrayList();

108        l.add(lo.toString());
109      }
110      return l;
111    }

126    }

154  @Test
155  public void testSignRequest() throws Exception {
156    when(mockRequest.getURI()).thenReturn(testUri);
157
158    when(mockRequest.getMethod()).thenReturn("POST");
159
160    Object entity = "Simsalabim";
161    when(mockRequest.getEntity()).thenReturn(entity);
162    // we instantiate a real request here, because there is no implementation of MultivaluedMap<String, Object>
163    MultivaluedMap<String, Object> headers = new OutBoundHeaders();
164    when(mockRequest.getHeaders()).thenReturn(headers);
165          
166    GbifAuthService service = GbifAuthService.singleKeyAuthService(APPKEY, APPSECRET);
167    service.signRequest("heinz", mockRequest);
168
169    assertNotNull(headers.getFirst(HEADER_CONTENT_MD5));
170    assertEquals("heinz", headers.getFirst(HEADER_GBIF_USER));
171    assertTrue(headers.getFirst(HEADER_AUTHORIZATION).toString().startsWith("GBIF appKey:"));
172  }

174  @Test
175  public void testIsValid() throws Exception {
176    when(mockRequest.getURI()).thenReturn(testUri);
177
178    when(mockRequest.getMethod()).thenReturn("POST");
179    when(containerRequest.getMethod()).thenReturn("POST");
180
181    Object entity = "Simsalabim";
182    when(mockRequest.getEntity()).thenReturn(entity);
183    // we instantiate a real request here, because there is no implementation of MultivaluedMap<String, Object>
184    MultivaluedMap<String, Object> headers = new OutBoundHeaders();
185    when(mockRequest.getHeaders()).thenReturn(headers);
186    when(containerRequest.getRequestHeaders()).thenReturn(new HeaderWrapper(headers));
187    when(containerRequest.getHeaderValue(eq(HEADER_AUTHORIZATION))).thenCallRealMethod();
188
189    GbifAuthService service = GbifAuthService.singleKeyAuthService(APPKEY, APPSECRET);
190    service.signRequest("heinz", mockRequest);
191    assertTrue(service.isValidRequest(containerRequest));
192
193    headers.putSingle(HEADER_CONTENT_MD5, "73");
194    assertFalse(service.isValidRequest(containerRequest));
195  }

210  /**
211   * Creates a mock POST to the /dataset endpoint with a String entity.
212   * Jersey 2.x has ContainerRequestBuilder to create a proper mock but Jersey 1.x doesn't.
213   * Only methods that we are currently using in unit tests are 'mapped' with Mockito.
214   *
215   * @param username
216   * @param service
217   * @return
218   * @throws URISyntaxException
219   */
220  public static ContainerRequest createMockContainerRequestFor(String username, GbifAuthService service) throws URISyntaxException {
221    ContainerRequest containerRequest = mock(ContainerRequest.class, Mockito.CALLS_REAL_METHODS);
222    ClientRequest mockRequest = mock(ClientRequest.class, Mockito.CALLS_REAL_METHODS);
223    when(mockRequest.getURI()).thenReturn(new URI("http://api.gbif.org/v1/dataset"));
224    //shared map
225    MultivaluedMap<String, Object> headers = new OutBoundHeaders();
226
227    when(containerRequest.getMethod()).thenReturn("POST");
228    when(mockRequest.getMethod()).thenReturn("POST");
229    Object entity = "My Entity";
230    when(mockRequest.getEntity()).thenReturn(entity);
231    when(mockRequest.getHeaders()).thenReturn(headers);
232
233    service.signRequest(username, mockRequest);
234
235    HeaderWrapper headersWrapper = new HeaderWrapper(headers);
236    when(containerRequest.getRequestHeaders()).thenReturn(headersWrapper);
237
238    return containerRequest;
239  }
