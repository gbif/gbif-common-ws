034@RunWith(MockitoJUnitRunner.class)
035public class GbifAppAuthServiceTest {
036
037  public static final String APPKEY = "appKey";
038  private static final String APPSECRET = "fghj56g66b676DFG";
039
040  public static Map<String, String> buildAppKeyMap() {
041    Map<String, String> map = Maps.newHashMap();
042    map.put(APPKEY, APPSECRET);
043    return map;
044  }
045
046  GbifAppAuthService service = new GbifAppAuthService(buildAppKeyMap());
047
048  @Mock
049  ContainerRequest containerRequest;
050  @Mock
051  ClientRequest mockRequest;
052  MultivaluedMap<String, Object> headers;
053
054  private static class HeaderWrapper implements MultivaluedMap<String, String> {
055
056    private final MultivaluedMap<String, Object> map;
057
058    public HeaderWrapper(MultivaluedMap<String, Object> map) {
059      this.map = map;
060    }
061
062    @Override
063    public void putSingle(String key, String value) {
064      map.putSingle(key, value);
065    }
066
067    @Override
068    public void add(String key, String value) {
069      map.add(key, value);
070    }
071
072    @Override
073    public String getFirst(String key) {
074      return (String) map.getFirst(key);
075    }
076
077    @Override
078    public int size() {
079      return map.size();
080    }
081
082    @Override
083    public boolean isEmpty() {
084      return map.isEmpty();
085    }
086
087    @Override
088    public boolean containsKey(Object o) {
089      return map.containsKey(o);
090    }
091
092    @Override
093    public boolean containsValue(Object o) {
094      return map.containsValue(o);
095    }
096
097    @Override
098    public List<String> get(Object o) {
099      final List<Object> src = map.get(o);
100      if (src == null) {
101        return null;
102      }
103      List<String> l = Lists.newArrayList();
104      for (Object lo : src) {
105        l.add(lo.toString());
106      }
107      return l;
108    }
109
110    @Override
111    public List<String> put(String s, List<String> strings) {
112      throw new UnsupportedOperationException();
113    }
114
115    @Override
116    public List<String> remove(Object o) {
117      throw new UnsupportedOperationException();
118    }
119
120    @Override
121    public void putAll(Map<? extends String, ? extends List<String>> map) {
122      throw new UnsupportedOperationException();
123    }
124
125    @Override
126    public void clear() {
127      map.clear();
128    }
129
130    @Override
131    public Set<String> keySet() {
132      return map.keySet();
133    }
134
135    @Override
136    public Collection<List<String>> values() {
137      throw new UnsupportedOperationException();
138    }
139
140    @Override
141    public Set<Entry<String, List<String>>> entrySet() {
142      throw new UnsupportedOperationException();
143    }
144  }
145
146  @Before
147  public void initMocks() throws Exception {
148    URI uri = new URI("http://www.gbif.org/dataset");
149    when(mockRequest.getURI()).thenReturn(uri);
150    when(containerRequest.getRequestUri()).thenReturn(uri);
151    when(containerRequest.getAbsolutePath()).thenReturn(uri);
152
153    when(mockRequest.getMethod()).thenReturn("POST");
154    when(containerRequest.getMethod()).thenReturn("POST");
155
156    Object entity = "Simsalabim";
157    when(mockRequest.getEntity()).thenReturn(entity);
158    // we instantiate a real request here, because there is no implementation of MultivaluedMap<String, Object>
159    headers = new OutBoundHeaders();
160    when(mockRequest.getHeaders()).thenReturn(headers);
161    when(containerRequest.getRequestHeaders()).thenReturn(new HeaderWrapper(headers));
162    when(containerRequest.getHeaderValue(eq(HEADER_AUTHORIZATION))).thenCallRealMethod();
163    when(containerRequest.getHeaderValue(eq(HEADER_CONTENT_TYPE))).thenCallRealMethod();
164
165  }
166
167  @Test
168  public void testSignRequest() throws Exception {
169
170    service.signRequest("appKey", "heinz", mockRequest);
171
172    assertNotNull(headers.getFirst(HEADER_GBIF_CONTENT_HASH));
173    assertNotNull(headers.getFirst(HEADER_GBIF_DATE));
174    assertEquals("heinz", headers.getFirst(HEADER_GBIF_USER));
175    assertTrue(headers.getFirst(HEADER_AUTHORIZATION).toString().startsWith("GBIF appKey:"));
176  }
177
178  @Test
179  public void testIsValid() throws Exception {
180
181    service.signRequest("appKey", "heinz", mockRequest);
182
183    assertTrue(service.isValidRequest(containerRequest));
184
185    headers.putSingle(HEADER_GBIF_CONTENT_HASH, 73);
186    assertFalse(service.isValidRequest(containerRequest));
187  }
