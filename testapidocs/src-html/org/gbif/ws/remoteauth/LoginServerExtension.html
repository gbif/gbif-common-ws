014package org.gbif.ws.remoteauth;
015
016import org.gbif.ws.security.UserRoles;
017
018import java.util.Collections;
019import java.util.UUID;
020
021import org.junit.jupiter.api.extension.AfterAllCallback;
022import org.junit.jupiter.api.extension.BeforeAllCallback;
023import org.junit.jupiter.api.extension.ExtensionContext;
024import org.springframework.http.HttpStatus;
025
026import com.fasterxml.jackson.databind.ObjectMapper;
027import com.github.tomakehurst.wiremock.WireMockServer;
028import com.github.tomakehurst.wiremock.core.WireMockConfiguration;
029
030import lombok.Builder;
031import lombok.Data;
032
033import static com.github.tomakehurst.wiremock.client.WireMock.aResponse;
034import static com.github.tomakehurst.wiremock.client.WireMock.equalTo;
035import static com.github.tomakehurst.wiremock.client.WireMock.matching;
036import static com.github.tomakehurst.wiremock.client.WireMock.post;
037import static org.gbif.ws.util.SecurityConstants.GBIF_SCHEME_PREFIX;
038import static org.gbif.ws.util.SecurityConstants.HEADER_CONTENT_MD5;
039import static org.gbif.ws.util.SecurityConstants.HEADER_GBIF_USER;
040import static org.springframework.http.HttpHeaders.*;
041
042/**
043 * This class mocks the auth services of the registry-ws that are used in the authentication providers of the remote
044 * auth classes.
045 */
046public class LoginServerExtension implements BeforeAllCallback, AfterAllCallback {
047
048  private static final ObjectMapper OBJECT_MAPPER = new ObjectMapper();
049
050  public static final TestUser ADMIN_USER =
051      TestUser.builder()
052          .loggedUser(
053              LoggedUser.builder()
054                  .userName("admin")
055                  .email("admin@gbif.org")
056                  .roles(Collections.singleton(UserRoles.ADMIN_ROLE))
057                  .build())
058          .password("pass1")
059          .jwtToken("token1")
060          .appKey("app1")
061          .appSecret("secret1")
062          .build();
063
064  public static final TestUser USER =
065      TestUser.builder()
066          .loggedUser(
067              LoggedUser.builder()
068                  .userName("user")
069                  .email("user@gbif.org")
070                  .roles(Collections.singleton(UserRoles.USER_ROLE))
071                  .build())
072          .password("pass2")
073          .jwtToken("token2")
074          .appKey("app2")
075          .appSecret("secret2")
076          .build();
077
078  public static final TestUser INVALID_USER =
079      TestUser.builder()
080          .loggedUser(
081              LoggedUser.builder()
082                  .userName("invalid")
083                  .email("invalid@gbif.org")
084                  .roles(Collections.singleton(UserRoles.USER_ROLE))
085                  .build())
086          .password("pass3")
087          .jwtToken("token3")
088          .appKey("app3")
089          .appSecret("secret3")
090          .build();
091
092  private final WireMockServer wireMockServer =
093      new WireMockServer(WireMockConfiguration.DYNAMIC_PORT);
094
095  @Override
096  public void beforeAll(ExtensionContext context) throws Exception {
097    wireMockServer.stubFor(
098        post("/user/auth/basic")
099            .withBasicAuth(USER.getLoggedUser().getUserName(), USER.getPassword())
100            .willReturn(
101                aResponse().withBody(OBJECT_MAPPER.writeValueAsString(USER.getLoggedUser()))));
102    wireMockServer.stubFor(
103        post("/user/auth/basic")
104            .withBasicAuth(ADMIN_USER.getLoggedUser().getUserName(), ADMIN_USER.getPassword())
105            .willReturn(
106                aResponse()
107                    .withBody(OBJECT_MAPPER.writeValueAsString(ADMIN_USER.getLoggedUser()))));
108    wireMockServer.stubFor(
109        post("/user/auth/basic")
110            .withBasicAuth(INVALID_USER.getLoggedUser().getUserName(), INVALID_USER.getPassword())
111            .willReturn(aResponse().withStatus(HttpStatus.UNAUTHORIZED.value())));
112    wireMockServer.stubFor(
113        post("/user/auth/jwt")
114            .withHeader(AUTHORIZATION, equalTo("Bearer " + USER.getJwtToken()))
115            .willReturn(
116                aResponse()
117                    .withBody(OBJECT_MAPPER.writeValueAsString(USER.getLoggedUser()))
118                    .withHeader("token", UUID.randomUUID().toString())));
119    wireMockServer.stubFor(
120        post("/user/auth/jwt")
121            .withHeader(AUTHORIZATION, equalTo("Bearer " + ADMIN_USER.getJwtToken()))
122            .willReturn(
123                aResponse()
124                    .withBody(OBJECT_MAPPER.writeValueAsString(ADMIN_USER.getLoggedUser()))
125                    .withHeader("token", UUID.randomUUID().toString())));
126    wireMockServer.stubFor(
127        post("/user/auth/jwt")
128            .withHeader(AUTHORIZATION, equalTo("Bearer " + INVALID_USER.getJwtToken()))
129            .willReturn(aResponse().withStatus(HttpStatus.UNAUTHORIZED.value())));
130    wireMockServer.stubFor(
131        post("/user/auth/app")
132            .withHeader(AUTHORIZATION, equalTo(USER.getGbifSchemeHeader()))
133            .withHeader(HEADER_GBIF_USER, equalTo(USER.getLoggedUser().getUserName()))
134            .withHeader(HEADER_CONTENT_MD5, matching(".*"))
135            .withHeader(CONTENT_TYPE, matching(".*"))
136            .willReturn(
137                aResponse().withBody(OBJECT_MAPPER.writeValueAsString(USER.getLoggedUser()))));
138    wireMockServer.stubFor(
139        post("/user/auth/app")
140            .withHeader(AUTHORIZATION, equalTo(ADMIN_USER.getGbifSchemeHeader()))
141            .withHeader(HEADER_GBIF_USER, equalTo(ADMIN_USER.getLoggedUser().getUserName()))
142            .withHeader(HEADER_CONTENT_MD5, matching(".*"))
143            .withHeader(CONTENT_TYPE, matching(".*"))
144            .willReturn(
145                aResponse()
146                    .withBody(OBJECT_MAPPER.writeValueAsString(ADMIN_USER.getLoggedUser()))));
147    wireMockServer.stubFor(
148        post("/user/auth/app")
149            .withHeader(AUTHORIZATION, equalTo(INVALID_USER.getGbifSchemeHeader()))
150            .withHeader(HEADER_GBIF_USER, equalTo(INVALID_USER.getLoggedUser().getUserName()))
151            .withHeader(HEADER_CONTENT_MD5, matching(".*"))
152            .withHeader(CONTENT_TYPE, matching(".*"))
153            .willReturn(aResponse().withStatus(HttpStatus.UNAUTHORIZED.value())));
154
155    wireMockServer.start();
156  }
157
158  @Override
159  public void afterAll(ExtensionContext context) throws Exception {
160    wireMockServer.stop();
161  }
162
163  public WireMockServer getWireMockServer() {
164    return wireMockServer;
165  }
166
167  @Builder
168  @Data
169  static class TestUser {
170
171    LoggedUser loggedUser;
172    String password;
173    String jwtToken;
174    String appKey;
175    String appSecret;
176
177    String getGbifSchemeHeader() {
178      return GBIF_SCHEME_PREFIX + appKey + ":" + appSecret;
179    }
180  }
181}
