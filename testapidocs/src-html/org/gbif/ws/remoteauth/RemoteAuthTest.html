014package org.gbif.ws.remoteauth;
015
016import org.gbif.ws.remoteauth.TestApplication.TestController;
017
018import java.nio.charset.StandardCharsets;
019
020import org.junit.jupiter.api.Test;
021import org.junit.jupiter.api.extension.ExtendWith;
022import org.junit.jupiter.api.extension.RegisterExtension;
023import org.springframework.beans.factory.annotation.Autowired;
024import org.springframework.boot.test.autoconfigure.web.client.AutoConfigureWebClient;
025import org.springframework.boot.test.autoconfigure.web.servlet.AutoConfigureMockMvc;
026import org.springframework.boot.test.autoconfigure.web.servlet.WebMvcTest;
027import org.springframework.boot.test.util.TestPropertyValues;
028import org.springframework.context.ApplicationContextInitializer;
029import org.springframework.context.ConfigurableApplicationContext;
030import org.springframework.http.HttpHeaders;
031import org.springframework.test.context.ContextConfiguration;
032import org.springframework.test.context.junit.jupiter.SpringExtension;
033import org.springframework.test.web.servlet.MockMvc;
034import org.springframework.test.web.servlet.request.MockMvcRequestBuilders;
035import org.springframework.test.web.servlet.result.MockMvcResultMatchers;
036
037import static org.gbif.ws.remoteauth.LoginServerExtension.ADMIN_USER;
038import static org.gbif.ws.remoteauth.LoginServerExtension.INVALID_USER;
039import static org.gbif.ws.remoteauth.LoginServerExtension.USER;
040import static org.gbif.ws.util.SecurityConstants.HEADER_CONTENT_MD5;
041import static org.gbif.ws.util.SecurityConstants.HEADER_GBIF_USER;
042
043/**
044 * Tests the remote auth classes
045 *
046 * @see org.gbif.ws.remoteauth
047 */
048@ExtendWith(SpringExtension.class)
049@WebMvcTest(controllers = TestController.class)
050@AutoConfigureMockMvc
051@AutoConfigureWebClient
052@ContextConfiguration(initializers = {RemoteAuthTest.ContextInitializer.class})
053public class RemoteAuthTest {
054
055  @RegisterExtension static LoginServerExtension loginServer = new LoginServerExtension();
056
057  @Autowired MockMvc mockMvc;
058
059  @Test
060  public void noAuthSentTest() throws Exception {
061    // non-secured endpoint
062    mockMvc
063        .perform(MockMvcRequestBuilders.get("/noAuth"))
064        .andExpect(MockMvcResultMatchers.status().isOk());
065
066    // secured endpoint
067    mockMvc
068        .perform(MockMvcRequestBuilders.get("/admin"))
069        .andExpect(MockMvcResultMatchers.status().isForbidden());
070  }
071
072  @Test
073  public void securedBasicAuthTest() throws Exception {
074    // user with no permissions
075    mockMvc
076        .perform(
077            MockMvcRequestBuilders.get("/admin")
078                .header(
079                    HttpHeaders.AUTHORIZATION,
080                    "Basic "
081                        + HttpHeaders.encodeBasicAuth(
082                            USER.getLoggedUser().getUserName(),
083                            USER.getPassword(),
084                            StandardCharsets.UTF_8)))
085        .andExpect(MockMvcResultMatchers.status().isForbidden());
086
087    // user with permissions
088    mockMvc
089        .perform(
090            MockMvcRequestBuilders.get("/admin")
091                .header(
092                    HttpHeaders.AUTHORIZATION,
093                    "Basic "
094                        + HttpHeaders.encodeBasicAuth(
095                            ADMIN_USER.getLoggedUser().getUserName(),
096                            ADMIN_USER.getPassword(),
097                            StandardCharsets.UTF_8)))
098        .andExpect(MockMvcResultMatchers.status().isOk());
099
100    // invalid user
101    mockMvc
102        .perform(
103            MockMvcRequestBuilders.get("/admin")
104                .header(
105                    HttpHeaders.AUTHORIZATION,
106                    "Basic "
107                        + HttpHeaders.encodeBasicAuth(
108                            INVALID_USER.getLoggedUser().getUserName(),
109                            INVALID_USER.getPassword(),
110                            StandardCharsets.UTF_8)))
111        .andExpect(MockMvcResultMatchers.status().isUnauthorized());
112  }
113
114  @Test
115  public void securedJwtTest() throws Exception {
116    // user with no permissions
117    mockMvc
118        .perform(
119            MockMvcRequestBuilders.get("/admin")
120                .header(HttpHeaders.AUTHORIZATION, "Bearer " + USER.getJwtToken()))
121        .andExpect(MockMvcResultMatchers.status().isForbidden());
122
123    // user with permissions
124    mockMvc
125        .perform(
126            MockMvcRequestBuilders.get("/admin")
127                .header(HttpHeaders.AUTHORIZATION, "Bearer " + ADMIN_USER.getJwtToken()))
128        .andExpect(MockMvcResultMatchers.status().isOk());
129
130    // invalid user
131    mockMvc
132        .perform(
133            MockMvcRequestBuilders.get("/admin")
134                .header(HttpHeaders.AUTHORIZATION, "Bearer " + INVALID_USER.getJwtToken()))
135        .andExpect(MockMvcResultMatchers.status().isUnauthorized());
136  }
137
138  @Test
139  public void securedGbifAppTest() throws Exception {
140    // user with no permissions
141    mockMvc
142        .perform(
143            MockMvcRequestBuilders.get("/admin")
144                .header(HttpHeaders.AUTHORIZATION, USER.getGbifSchemeHeader())
145                .header(HEADER_GBIF_USER, USER.getLoggedUser().getUserName())
146                .header(HEADER_CONTENT_MD5, "foo"))
147        .andExpect(MockMvcResultMatchers.status().isForbidden());
148
149    // user with permissions
150    mockMvc
151        .perform(
152            MockMvcRequestBuilders.get("/admin")
153                .header(HttpHeaders.AUTHORIZATION, ADMIN_USER.getGbifSchemeHeader())
154                .header(HEADER_GBIF_USER, ADMIN_USER.getLoggedUser().getUserName())
155                .header(HEADER_CONTENT_MD5, "foo"))
156        .andExpect(MockMvcResultMatchers.status().isOk());
157
158    // invalid user
159    mockMvc
160        .perform(
161            MockMvcRequestBuilders.get("/admin")
162                .header(HttpHeaders.AUTHORIZATION, INVALID_USER.getGbifSchemeHeader())
163                .header(HEADER_GBIF_USER, INVALID_USER.getLoggedUser().getUserName())
164                .header(HEADER_CONTENT_MD5, "foo"))
165        .andExpect(MockMvcResultMatchers.status().isUnauthorized());
166  }
167
168  static class ContextInitializer
169      implements ApplicationContextInitializer<ConfigurableApplicationContext> {
170
171    @Override
172    public void initialize(ConfigurableApplicationContext configurableApplicationContext) {
173      TestPropertyValues.of("login.url=" + loginServer.getWireMockServer().baseUrl())
174          .applyTo(configurableApplicationContext.getEnvironment());
175    }
176  }
177}
