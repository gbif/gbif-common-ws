001package org.gbif.ws.util;
002
003import org.junit.Test;
004import org.junit.runner.RunWith;
005import org.junit.runners.Parameterized;
006
007import java.util.Arrays;
008
009import static org.junit.Assert.assertEquals;
010
011@RunWith(value = Parameterized.class)
012public class XSSUtilTest {
013  // decoded request, header, or parameter
014  private String value;
015
016  // value expected after stripXSS is called
017  private String stripXSSValue;
018
019  // true if request contains XSS, or false otherwise
020  private boolean containsXSS;
021
022  public XSSUtilTest(String value, boolean containsXSS, String stripXSSValue) {
023    this.value = value;
024    this.containsXSS = containsXSS;
025    this.stripXSSValue = stripXSSValue;
026  }
027
028  /**
029   * Additional requests to test can be added here.
030   */
031  @Parameterized.Parameters
032  public static Iterable<Object[]> data() {
033    return Arrays.asList(new Object[][] {
034      // contain xss
035      {"http://localhost:8090/ipt/about.do?request_locale=\"'><script>alert(6227)</script>&email=\"'><script>alert(6227)</script>&password=\"'><script>alert(6227)</script>&login-submit=\"'><script>alert(6227)</script>", true,
036              "http://localhost:8090/ipt/about.do?request_locale=\"'>&email=\"'>&password=\"'>&login-submit=\"'>"},
037      {"http://www.gbif.org/dataset/search?q=\"><script>prompt('XSSPOSED')</script>", true, "http://www.gbif.org/dataset/search?q=\">"},
038      {"http://www.gbif.org/index.php?name=<script>window.onload = function() {var link=document.getElementsByTagName(\"a\");link[0].href=\"http://not-real-xssattackexamples.com/\";}</script>", true,
039              "http://www.gbif.org/index.php?name="},
040      {"http://www.gbif.org/dataset/search?q=<iframe>frame it</iframe>", true, "http://www.gbif.org/dataset/search?q="},
041      {"http://www.gbif.org/login.do?email=\"><img src=x onerror=prompt(/XSS/)>", true, "http://www.gbif.org/login.do?email=\">"},
042      {"http://gbif.org/login.do?email=\"><img src=x onmouseover=prompt(/XSS/)>", true, "http://gbif.org/login.do?email=\">"},
043
044      // do not contain xss
045      {"http://gbif.org/login.do?email=sven@gbif.org&password=xyz", false, "http://gbif.org/login.do?email=sven@gbif.org&password=xyz"},
046      {"http://www.gbif.org/ipt", false, null},
047      {"http://api.gbif.org/v1/occurrence/search?year=1800", false, null}});
048  }
049
050  @Test
051  public void testContainsXSS() {
052    assertEquals(containsXSS, XSSUtil.containsXSS(value));
053
054    if(containsXSS){
055      assertEquals(stripXSSValue, XSSUtil.stripXSS(value));
056    }
057  }
058
059}
