001package org.gbif.ws.util;
002
003import java.util.Arrays;
004
005import org.junit.Test;
006import org.junit.runner.RunWith;
007import org.junit.runners.Parameterized;
008
009import static org.junit.Assert.assertEquals;
010
011@RunWith(value = Parameterized.class)
012public class XSSUtilTest {
013  // decoded request, header, or parameter
014  private String value;
015  // true if request contains XSS, or false otherwise
016  private boolean containsXSS;
017
018  public XSSUtilTest(String value, boolean containsXSS) {
019    this.value = value;
020    this.containsXSS = containsXSS;
021  }
022
023  /**
024   * Additional requests to test can be added here.
025   */
026  @Parameterized.Parameters
027  public static Iterable<Object[]> data() {
028    return Arrays.asList(new Object[][] {
029      // contain xss
030      {"http://localhost:8090/ipt/about.do?request_locale=\"'><script>alert(6227)</script>&email=\"'><script>alert(6227)</script>&password=\"'><script>alert(6227)</script>&login-submit=\"'><script>alert(6227)</script>", true},
031      {"http://www.gbif.org/dataset/search?q=\"><script>prompt('XSSPOSED')</script>", true},
032      {"http://www.gbif.org/index.php?name=<script>window.onload = function() {var link=document.getElementsByTagName(\"a\");link[0].href=\"http://not-real-xssattackexamples.com/\";}</script>", true},
033      // do not contain xss
034      {"http://www.gbif.org/ipt", false},
035      {"http://api.gbif.org/v1/occurrence/search?year=1800", false}});
036  }
037
038  @Test
039  public void testContainsXSS() {
040    assertEquals(containsXSS, XSSUtil.containsXSS(value));
041  }
042}
