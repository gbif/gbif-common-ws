001package org.gbif.ws.client.filter;
002
003import org.gbif.ws.server.filter.JsonpResponseFilter;
004import org.gbif.ws.util.ExtraMediaTypes;
005
006import java.util.Collection;
007import javax.ws.rs.core.MediaType;
008import javax.ws.rs.core.MultivaluedMap;
009
010import com.google.common.collect.ImmutableList;
011import com.sun.jersey.api.json.JSONWithPadding;
012import com.sun.jersey.core.util.MultivaluedMapImpl;
013import com.sun.jersey.spi.container.ContainerRequest;
014import com.sun.jersey.spi.container.ContainerResponse;
015import org.junit.Before;
016import org.junit.Test;
017import org.junit.runner.RunWith;
018import org.mockito.Mock;
019import org.mockito.junit.MockitoJUnitRunner;
020
021import static org.junit.Assert.assertEquals;
022import static org.mockito.Matchers.any;
023import static org.mockito.Mockito.never;
024import static org.mockito.Mockito.reset;
025import static org.mockito.Mockito.verify;
026import static org.mockito.Mockito.when;
027
028@RunWith(MockitoJUnitRunner.class)
029public class JsonpResponseFilterTest {
030
031  private JsonpResponseFilter filter = new JsonpResponseFilter();
032  private MultivaluedMap<String, String> params;
033
034  @Before
035  public void setUp() {
036    params = new MultivaluedMapImpl();
037    when(mockRequest.getQueryParameters()).thenReturn(params);
038  }
039
040  @Mock
041  ContainerResponse mockResponse;
042
043  @Mock
044  ContainerRequest mockRequest;
045
046  @Test
047  public void testFilterOnlyJsonAndJavascript() {
048    Collection<MediaType> validMediaTypes =
049      ImmutableList.of(MediaType.APPLICATION_JSON_TYPE, ExtraMediaTypes.APPLICATION_JAVASCRIPT_TYPE);
050
051    when(mockRequest.getQueryParameters()).thenReturn(params);
052    for (MediaType mediaType : validMediaTypes) {
053      when(mockResponse.getMediaType()).thenReturn(mediaType);
054      when(mockResponse.getEntity()).thenReturn("{id:1}");
055      params.putSingle("callback", "foo");
056
057      assertEquals(mockResponse, filter.filter(mockRequest, mockResponse));
058
059      verify(mockResponse).setEntity(any(JSONWithPadding.class));
060      reset(mockResponse);
061    }
062  }
063
064  @Test
065  public void testDontFilterOtherMediatypes() {
066    when(mockResponse.getMediaType()).thenReturn(MediaType.APPLICATION_ATOM_XML_TYPE);
067    filter.filter(mockRequest, mockResponse);
068
069    verify(mockResponse, never()).setEntity(any());
070  }
071
072  @Test
073  public void testNullMediaType() {
074    when(mockResponse.getMediaType()).thenReturn(null);
075    filter.filter(mockRequest, mockResponse);
076
077    verify(mockResponse, never()).setEntity(any());
078  }
079
080  @Test
081  public void testEmptyAndMissingCallback() {
082    when(mockResponse.getMediaType()).thenReturn(MediaType.APPLICATION_JSON_TYPE);
083
084    assertEquals(mockResponse, filter.filter(mockRequest, mockResponse));
085    verify(mockResponse, never()).setEntity(any());
086
087    params.putSingle("callback", "");
088    assertEquals(mockResponse, filter.filter(mockRequest, mockResponse));
089    verify(mockResponse, never()).setEntity(any());
090  }
091
092  @Test
093  public void testEmptyEntity() {
094    when(mockResponse.getMediaType()).thenReturn(MediaType.APPLICATION_JSON_TYPE);
095
096    params.putSingle("callback", "foo");
097    when(mockRequest.getQueryParameters()).thenReturn(params);
098
099    when(mockResponse.getEntity()).thenReturn(null);
100    assertEquals(mockResponse, filter.filter(mockRequest, mockResponse));
101    verify(mockResponse, never()).setEntity(any());
102  }
103}
