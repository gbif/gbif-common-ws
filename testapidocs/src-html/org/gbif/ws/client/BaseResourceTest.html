001package org.gbif.ws.client;
002
003import java.net.URI;
004
005import javax.ws.rs.core.UriBuilder;
006
007import com.google.inject.servlet.GuiceFilter;
008import com.google.inject.servlet.GuiceServletContextListener;
009import com.sun.jersey.api.client.Client;
010import com.sun.jersey.api.client.WebResource;
011import com.sun.jersey.api.client.config.ClientConfig;
012import com.sun.jersey.api.client.config.DefaultClientConfig;
013import com.sun.jersey.api.client.filter.HTTPBasicAuthFilter;
014import com.sun.jersey.api.core.PackagesResourceConfig;
015import com.sun.jersey.api.json.JSONConfiguration;
016import com.sun.jersey.test.framework.JerseyTest;
017import com.sun.jersey.test.framework.WebAppDescriptor;
018import com.sun.jersey.test.framework.spi.container.TestContainerException;
019import com.sun.jersey.test.framework.spi.container.TestContainerFactory;
020import com.sun.jersey.test.framework.spi.container.grizzly.web.GrizzlyWebTestContainerFactory;
021
022public class BaseResourceTest extends JerseyTest {
023  // The system properties that can be set to override the test port
024  private final static String[] SYSTEM_PROPERTY_TEST_PORT = {"jersey.test.port","JERSEY_HTTP_PORT"};
025
026  public static WebAppDescriptor buildWebAppDescriptor(String baseResourcePackage, String contextPath,
027    Class<? extends GuiceServletContextListener> contextListenerClass) throws TestContainerException {
028
029    ClientConfig clientConfig = new DefaultClientConfig();
030    clientConfig.getFeatures().put(JSONConfiguration.FEATURE_POJO_MAPPING, true);
031
032    return new WebAppDescriptor.Builder().contextListenerClass(contextListenerClass).filterClass(GuiceFilter.class)
033      .contextPath(contextPath).servletPath("/").clientConfig(clientConfig)
034      .initParam(PackagesResourceConfig.PROPERTY_PACKAGES, baseResourcePackage).build();
035  }
036
037  protected final String contextPath;
038
039  /**
040   * @throws TestContainerException
041   */
042  public BaseResourceTest(String baseResourcePackage, String contextPath,
043    Class<? extends GuiceServletContextListener> contextListenerClass)
044    throws TestContainerException {
045    super(BaseResourceTest.buildWebAppDescriptor(baseResourcePackage, contextPath, contextListenerClass));
046    this.contextPath = contextPath;
047  }
048
049  public WebResource createExternalWebResource(String baseUrl, String resourceURL) {
050    ClientConfig cc = new DefaultClientConfig();
051    // Custom readers and writers must be added to the client
052    Client client;
053    client = Client.create(cc);
054    WebResource resource = client.resource(baseUrl + resourceURL);
055    return resource;
056  }
057
058  public WebResource createExternalWebResource(String baseUrl, String resourceURL, String userName, String password) {
059    ClientConfig cc = new DefaultClientConfig();
060    // Custom readers and writers must be added to the client
061    Client client;
062    client = Client.create(cc);
063    this.setCredentials(userName, password);
064    WebResource resource = client.resource(baseUrl + resourceURL);
065    return resource;
066  }
067
068  @Override
069  protected TestContainerFactory getTestContainerFactory() {
070    return new GrizzlyWebTestContainerFactory();
071  }
072
073  protected void setCredentials(String userName, String password) {
074    client().addFilter(new HTTPBasicAuthFilter(userName, password));
075  }
076
077  /**
078   * Returns the base URI of the application.
079   * 
080   * @return The base URI of the application
081   */
082  protected static URI getTestBaseURI() {
083    return UriBuilder.fromUri("http://localhost/").port(getTestPort(9998)).build();
084  }
085
086  /**
087   * Reads the system property defined in {@link SYSTEM_PROPERTY_TEST_PORT} to override
088   * the port for Grizzly.  
089   * @param defaultPort default port to use should no properties be provided
090   * @return The port, or the default
091   * @throws TestContainerException Should a property be provided but be an illegal argument
092   */
093  protected static int getTestPort(int defaultPort) throws TestContainerException {
094    for (String property : SYSTEM_PROPERTY_TEST_PORT) {
095      String port = System.getProperty(property);
096      if (port != null) {
097        try {
098          return Integer.parseInt(port);
099        } catch (NumberFormatException e) {
100          throw new TestContainerException(
101            "illegal value[" + port +"] supplied for property[" + property + "] which should defined a port number to use.", e);
102        }
103      }
104    }
105    return defaultPort;
106  }
107}
