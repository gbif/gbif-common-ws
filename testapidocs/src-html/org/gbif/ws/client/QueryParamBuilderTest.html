001package org.gbif.ws.client;
002
003import org.gbif.api.vocabulary.Rank;
004
005import org.junit.Test;
006
007import static org.junit.Assert.assertEquals;
008import static org.junit.Assert.assertNotNull;
009import static org.junit.Assert.assertNull;
010
011public class QueryParamBuilderTest {
012
013  @Test
014  public void testQueryParam() throws Exception {
015    QueryParamBuilder builder = QueryParamBuilder.create();
016    assertNotNull(builder.build());
017    assertEquals(0, builder.build().size());
018
019    builder.queryParam("a", "B").queryParam("b", "C").queryParam("2", "3").queryParam("", "ah").queryParam(null, "ah")
020      .queryParam("ah", "");
021    assertEquals(3, builder.build().size());
022    assertEquals("C", builder.build().getFirst("b"));
023    assertNull(builder.build().get("x"));
024
025    builder = QueryParamBuilder.create("kie", new Object[] {1, 2, 3, 4});
026    assertEquals(1, builder.build().size());
027    assertEquals("1", builder.build().getFirst("kie"));
028    assertEquals(4, builder.build().get("kie").size());
029  }
030
031  @Test
032  public void testKvpBuilder() throws Exception {
033    QueryParamBuilder builder = QueryParamBuilder.create("a", "B", "b", "C", "2", "3", "", "ah", null, "ah", "ah", "");
034    assertEquals(3, builder.build().size());
035    assertEquals("C", builder.build().getFirst("b"));
036    assertEquals("B", builder.build().getFirst("a"));
037    assertNull(builder.build().get("x"));
038
039    builder = QueryParamBuilder.create("rank", Rank.SPECIES);
040    assertEquals(1, builder.build().size());
041    assertEquals("SPECIES", builder.build().getFirst("rank"));
042    assertNull(builder.build().get("x"));
043
044    builder = QueryParamBuilder.create("a", "B", "b", "C");
045    assertEquals(2, builder.build().size());
046    assertEquals("B", builder.build().getFirst("a"));
047    assertEquals("C", builder.build().getFirst("b"));
048  }
049
050  @Test(expected = IllegalArgumentException.class)
051  public void testKvpBuilderException() throws Exception {
052    QueryParamBuilder.create("a", "B", "b", "C", "2");
053  }
054
055}
