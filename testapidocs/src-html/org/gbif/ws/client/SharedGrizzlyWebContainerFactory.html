001package org.gbif.ws.client;
002
003import java.io.IOException;
004import java.net.URI;
005import java.util.EventListener;
006import java.util.List;
007import java.util.Map;
008import java.util.logging.Logger;
009import javax.servlet.Servlet;
010import javax.ws.rs.core.UriBuilder;
011
012import com.sun.grizzly.http.embed.GrizzlyWebServer;
013import com.sun.grizzly.http.servlet.ServletAdapter;
014import com.sun.jersey.api.client.Client;
015import com.sun.jersey.test.framework.AppDescriptor;
016import com.sun.jersey.test.framework.WebAppDescriptor;
017import com.sun.jersey.test.framework.spi.container.TestContainer;
018import com.sun.jersey.test.framework.spi.container.TestContainerException;
019import com.sun.jersey.test.framework.spi.container.TestContainerFactory;
020
021/**
022 * This is a Grizzly web container that will reuse the container across tests rather than creating
023 * a new one each time. This is a patched version of the one that ships with Jersey, with the following
024 * modifications:
025 * a) Repeated calls to the same factory will return the SAME TestContainer
026 * b) start() and stop() on the TestContainer do nothing
027 * Because of these changes one should use this with caution. It is particularly useful when one wants
028 * the same grizzly container to be used for a complete test run, and then teared down at the end of the
029 * test with the JVM. Normally JerseyTest will create a new one per method which can be expensive and also
030 * tearing down grizzly appears to be on a separate thread, thus meaning connection pools are not terminated
031 * immediately (a suspicion that is not thoroughly verified).
032 */
033public class SharedGrizzlyWebContainerFactory implements TestContainerFactory {
034
035  /**
036   * This class has methods for instantiating, starting and stopping the Grizzly Web
037   * Server.
038   */
039  private static class SharedGrizzlyWebTestContainer implements TestContainer {
040
041    private static final Logger LOGGER = Logger.getLogger(SharedGrizzlyWebTestContainer.class.getName());
042
043    final URI baseUri;
044
045    final String contextPath;
046
047    final String servletPath;
048
049    final Class servletClass;
050
051    List<WebAppDescriptor.FilterDescriptor> filters = null;
052
053    final List<Class<? extends EventListener>> eventListeners;
054
055    final Map<String, String> initParams;
056
057    final Map<String, String> contextParams;
058
059    private GrizzlyWebServer webServer;
060
061    /**
062     * Creates an instance of {@link GrizzlyWebTestContainer}
063     *
064     * @param baseUri URI of the application
065     * @param ad      An instance of {@link WebAppDescriptor}
066     */
067    private SharedGrizzlyWebTestContainer(URI baseUri, WebAppDescriptor ad) {
068      this.baseUri = UriBuilder.fromUri(baseUri).path(ad.getContextPath()).path(ad.getServletPath()).build();
069
070      LOGGER.info("Creating Grizzly Web Container configured at the base URI " + this.baseUri);
071      this.contextPath = ad.getContextPath();
072      this.servletPath = ad.getServletPath();
073      this.servletClass = ad.getServletClass();
074      this.filters = ad.getFilters();
075      this.initParams = ad.getInitParams();
076      this.contextParams = ad.getContextParams();
077      this.eventListeners = ad.getListeners();
078
079      instantiateGrizzlyWebServer();
080
081    }
082
083    @Override
084    public URI getBaseUri() {
085      return baseUri;
086    }
087
088    @Override
089    public Client getClient() {
090      return null;
091    }
092
093    /**
094     * Instantiates the Grizzly Web Server
095     */
096    private void instantiateGrizzlyWebServer() {
097      webServer = new GrizzlyWebServer(baseUri.getPort());
098      ServletAdapter sa = new ServletAdapter();
099      Servlet servletInstance;
100      if (servletClass != null) {
101        try {
102          servletInstance = (Servlet) servletClass.newInstance();
103        } catch (InstantiationException ex) {
104          throw new TestContainerException(ex);
105        } catch (IllegalAccessException ex) {
106          throw new TestContainerException(ex);
107        }
108        sa.setServletInstance(servletInstance);
109      }
110
111      for (Class<? extends EventListener> eventListener : eventListeners) {
112        sa.addServletListener(eventListener.getName());
113      }
114
115      // Filter support
116      if (filters != null) {
117        try {
118          for (WebAppDescriptor.FilterDescriptor d : this.filters) {
119            sa.addFilter(d.getFilterClass().newInstance(), d.getFilterName(), d.getInitParams());
120          }
121        } catch (InstantiationException ex) {
122          throw new TestContainerException(ex);
123        } catch (IllegalAccessException ex) {
124          throw new TestContainerException(ex);
125        }
126      }
127
128      for (String contextParamName : contextParams.keySet()) {
129        sa.addContextParameter(contextParamName, contextParams.get(contextParamName));
130      }
131
132      for (String initParamName : initParams.keySet()) {
133        sa.addInitParameter(initParamName, initParams.get(initParamName));
134      }
135
136      if (contextPath != null && contextPath.length() > 0) {
137        if (!contextPath.startsWith("/")) {
138          sa.setContextPath("/" + contextPath);
139        } else {
140          sa.setContextPath(contextPath);
141        }
142      }
143
144      if (servletPath != null && servletPath.length() > 0) {
145        if (!servletPath.startsWith("/")) {
146          sa.setServletPath("/" + servletPath);
147        } else {
148          sa.setServletPath(servletPath);
149        }
150      }
151
152      String[] mapping = null;
153      webServer.addGrizzlyAdapter(sa, mapping);
154      try {
155        LOGGER.info("Starting the Grizzly Web Container...");
156        webServer.start();
157      } catch (IOException e) {
158        throw new TestContainerException(e);
159      }
160    }
161
162    @Override
163    public void start() {
164      // Does nothing
165    }
166
167    @Override
168    public void stop() {
169      // Does nothing
170    }
171
172  }
173
174  // The singleton object
175  private static SharedGrizzlyWebTestContainer instance;
176
177  private static synchronized SharedGrizzlyWebTestContainer getInstance(URI baseUri, WebAppDescriptor ad) {
178    if (instance == null) {
179      instance = new SharedGrizzlyWebTestContainer(baseUri, ad);
180    }
181    return instance;
182  }
183
184  @Override
185  public TestContainer create(URI baseUri, AppDescriptor ad) {
186    if (!(ad instanceof WebAppDescriptor)) {
187      throw new IllegalArgumentException("The application descriptor must be an instance of WebAppDescriptor");
188    }
189
190    return getInstance(baseUri, (WebAppDescriptor) ad);
191  }
192
193  @Override
194  public Class<WebAppDescriptor> supports() {
195    return WebAppDescriptor.class;
196  }
197}
