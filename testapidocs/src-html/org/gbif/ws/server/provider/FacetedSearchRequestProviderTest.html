013package org.gbif.ws.server.provider;
014
015import org.gbif.api.model.common.search.FacetedSearchRequest;
016import org.gbif.ws.server.provider.SearchRequestProviderTest.TestSearchParameter;
017
018import javax.ws.rs.core.Context;
019import javax.ws.rs.core.MultivaluedMap;
020
021import com.sun.jersey.api.core.HttpContext;
022import com.sun.jersey.api.core.HttpRequestContext;
023import com.sun.jersey.core.spi.component.ComponentContext;
024import com.sun.jersey.core.spi.component.ComponentScope;
025import com.sun.jersey.core.util.MultivaluedMapImpl;
026import com.sun.jersey.spi.inject.Injectable;
027import org.junit.Test;
028import org.mockito.Mockito;
029
030import static org.gbif.ws.util.WebserviceParameter.PARAM_FACET;
031import static org.gbif.ws.util.WebserviceParameter.PARAM_FACET_MULTISELECT;
032
033import static org.junit.Assert.assertEquals;
034import static org.junit.Assert.assertNotNull;
035import static org.junit.Assert.assertNull;
036import static org.junit.Assert.assertTrue;
037
038
039public class FacetedSearchRequestProviderTest {
040
041  private final FacetedSearchRequestProvider provider =
042    new FacetedSearchRequestProvider(FacetedSearchRequest.class, TestSearchParameter.class);
043
044  @Test
045  public void testGetInjectable() {
046    ComponentContext mockComponentContext = Mockito.mock(ComponentContext.class);
047    Context mockContext = Mockito.mock(Context.class);
048
049    Injectable<FacetedSearchRequest> injectable =
050      provider.getInjectable(mockComponentContext, mockContext, FacetedSearchRequest.class);
051    assertNotNull(injectable);
052
053    injectable = provider.getInjectable(mockComponentContext, mockContext, String.class);
054    assertNull(injectable);
055  }
056
057  @Test
058  public void testGetValue() {
059    // mock context and request
060    HttpContext ctx = Mockito.mock(HttpContext.class);
061    HttpRequestContext req = Mockito.mock(HttpRequestContext.class);
062    Mockito.when(ctx.getRequest()).thenReturn(req);
063
064    MultivaluedMap<String, String> params = new MultivaluedMapImpl();
065    Mockito.when(req.getQueryParameters()).thenReturn(params);
066
067    // test with real facets
068    params.clear();
069    params.add(PARAM_FACET, TestSearchParameter.FIRSTNAME.name());
070    params.add(PARAM_FACET, TestSearchParameter.LASTNAME.name());
071    params.add(PARAM_FACET_MULTISELECT, Boolean.TRUE.toString());
072    FacetedSearchRequest<TestSearchParameter> x = (FacetedSearchRequest<TestSearchParameter>) provider.getValue(ctx);
073    assertTrue(x.getFacets().contains(TestSearchParameter.FIRSTNAME));
074    assertTrue(x.getFacets().contains(TestSearchParameter.LASTNAME));
075    assertTrue(x.isMultiSelectFacets());
076
077    // test with real facets, case insensitive
078    params.clear();
079    params.add(PARAM_FACET, "firstname");
080    params.add(PARAM_FACET, "LastName");
081    x = (FacetedSearchRequest<TestSearchParameter>) provider.getValue(ctx);
082    assertTrue(x.getFacets().contains(TestSearchParameter.FIRSTNAME));
083    assertTrue(x.getFacets().contains(TestSearchParameter.LASTNAME));
084
085    // test with wrong facet names
086    params.clear();
087    params.add(PARAM_FACET, "noname");
088    params.add(PARAM_FACET, "bisrtname");
089    x = (FacetedSearchRequest<TestSearchParameter>) provider.getValue(ctx);
090    assertTrue(x.getFacets().isEmpty());
091
092  }
093
094  @Test
095  public void testScope() {
096    assertEquals(ComponentScope.PerRequest, provider.getScope());
097  }
098}
