001package org.gbif.ws.server.provider;
002
003
004import com.sun.jersey.api.core.HttpContext;
005import com.sun.jersey.api.core.HttpRequestContext;
006import com.sun.jersey.core.spi.component.ComponentContext;
007import com.sun.jersey.core.spi.component.ComponentScope;
008import com.sun.jersey.core.util.MultivaluedMapImpl;
009import com.sun.jersey.spi.inject.Injectable;
010import org.gbif.api.model.common.paging.Pageable;
011import org.junit.Test;
012
013import javax.ws.rs.core.Context;
014import javax.ws.rs.core.MultivaluedMap;
015
016import static org.gbif.api.model.common.paging.PagingConstants.*;
017import static org.junit.Assert.*;
018import static org.mockito.Mockito.mock;
019import static org.mockito.Mockito.when;
020
021public class PageableProviderTest {
022
023  private final PageableProvider provider = new PageableProvider();
024
025  @Test
026  public void testGetInjectable() {
027    ComponentContext mockComponentContext = mock(ComponentContext.class);
028    Context mockContext = mock(Context.class);
029
030    Injectable<Pageable> injectable = provider.getInjectable(mockComponentContext, mockContext, Pageable.class);
031    assertNotNull(injectable);
032
033    injectable = provider.getInjectable(mockComponentContext, mockContext, String.class);
034    assertNull(injectable);
035  }
036
037  @Test
038  public void testGetValue() {
039    // mock context and request
040    HttpContext ctx = mock(HttpContext.class);
041    HttpRequestContext req = mock(HttpRequestContext.class);
042    when(ctx.getRequest()).thenReturn(req);
043
044    MultivaluedMap<String, String> params = new MultivaluedMapImpl();
045    when(req.getQueryParameters()).thenReturn(params);
046
047    // test with no params
048    assertEquals(DEFAULT_PARAM_OFFSET, provider.getValue(ctx).getOffset());
049    assertEquals(DEFAULT_PARAM_LIMIT, provider.getValue(ctx).getLimit());
050
051    // test with params
052    params.add(PARAM_OFFSET, "200");
053    params.add(PARAM_LIMIT, "10");
054    assertEquals(200, provider.getValue(ctx).getOffset());
055    assertEquals(10, provider.getValue(ctx).getLimit());
056
057    // test with bad content
058    params = new MultivaluedMapImpl();
059    when(req.getQueryParameters()).thenReturn(params);
060    params.add(PARAM_OFFSET, "NOT VALID");
061    params.add(PARAM_LIMIT, "-20");
062    assertEquals(DEFAULT_PARAM_OFFSET, provider.getValue(ctx).getOffset());
063    assertEquals(DEFAULT_PARAM_LIMIT, provider.getValue(ctx).getLimit());
064
065    // try to use insane limits
066    params.putSingle(PARAM_LIMIT, "20000");
067    assertEquals(DEFAULT_PARAM_OFFSET, provider.getValue(ctx).getOffset());
068    assertEquals(PageableProvider.LIMIT_CAP, provider.getValue(ctx).getLimit());
069  }
070
071  @Test
072  public void testScope() {
073    assertEquals(ComponentScope.PerRequest, provider.getScope());
074  }
075}
