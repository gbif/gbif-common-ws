001package org.gbif.ws.server.provider;

003import org.gbif.api.model.checklistbank.search.NameUsageSearchParameter;
004import org.gbif.api.model.checklistbank.search.NameUsageSearchRequest;

006import javax.ws.rs.core.Context;
007import javax.ws.rs.core.MultivaluedMap;
008
009import com.sun.jersey.api.core.HttpContext;
010import com.sun.jersey.api.core.HttpRequestContext;
011import com.sun.jersey.core.spi.component.ComponentContext;
012import com.sun.jersey.core.spi.component.ComponentScope;
013import com.sun.jersey.core.util.MultivaluedMapImpl;
014import com.sun.jersey.spi.inject.Injectable;
015import junit.framework.TestCase;
016import org.junit.Test;
017import org.mockito.Mockito;
018
019import static org.gbif.ws.util.WebserviceParameter.PARAM_FACET;
020import static org.gbif.ws.util.WebserviceParameter.PARAM_HIGHLIGHT;
021import static org.gbif.ws.util.WebserviceParameter.PARAM_HIGHLIGHT_CONTEXT;
022import static org.gbif.ws.util.WebserviceParameter.PARAM_HIGHLIGHT_FIELD;
023import static org.gbif.ws.util.WebserviceParameter.PARAM_QUERY_FIELD;
024
025/**
026 *
027 */
028public class NameUsageSearchRequestProviderTest extends TestCase {
029
030  private final NameUsageSearchRequestProvider provider = new NameUsageSearchRequestProvider();
031
032  @Test
033  public void testGetInjectable() {
034    ComponentContext mockComponentContext = Mockito.mock(ComponentContext.class);
035    Context mockContext = Mockito.mock(Context.class);
036
037    Injectable<NameUsageSearchRequest> injectable = provider.getInjectable(mockComponentContext, mockContext, NameUsageSearchRequest.class);
038    assertNotNull(injectable);
039
040    injectable = provider.getInjectable(mockComponentContext, mockContext, String.class);
041    assertNull(injectable);
042  }
043
044  @Test
045  public void testGetValue() {
046    // mock context and request
047    HttpContext ctx = Mockito.mock(HttpContext.class);
048    HttpRequestContext req = Mockito.mock(HttpRequestContext.class);
049    Mockito.when(ctx.getRequest()).thenReturn(req);
050
051    MultivaluedMap<String, String> params = new MultivaluedMapImpl();
052    Mockito.when(req.getQueryParameters()).thenReturn(params);
053
054    // test with real facets
055    params.clear();
056    params.add(PARAM_FACET, NameUsageSearchParameter.HABITAT.name());
057    params.add(PARAM_FACET, NameUsageSearchParameter.DATASET_KEY.name());
058    params.add(PARAM_HIGHLIGHT, Boolean.TRUE.toString());
059    params.add(PARAM_HIGHLIGHT_CONTEXT, String.valueOf(250));
060    params.add(PARAM_HIGHLIGHT_FIELD, NameUsageSearchRequest.QueryField.DESCRIPTION.name());
061    params.add(PARAM_HIGHLIGHT_FIELD, NameUsageSearchRequest.QueryField.VERNACULAR.name());
062    params.add(PARAM_QUERY_FIELD, NameUsageSearchRequest.QueryField.VERNACULAR.name());
063    params.add(PARAM_QUERY_FIELD, NameUsageSearchRequest.QueryField.SCIENTIFIC.name());
064
065    NameUsageSearchRequest x = provider.getValue(ctx);
066    assertTrue(x.getFacets().contains(NameUsageSearchParameter.HABITAT));
067    assertTrue(x.getFacets().contains(NameUsageSearchParameter.DATASET_KEY));
068    assertEquals(2, x.getFacets().size());
069
070    assertTrue(x.isHighlight());
071    assertEquals((Integer)250, x.getHighlightContext());
072
073    assertEquals(2, x.getHighlightFields().size());
074    assertTrue(x.getHighlightFields().contains(NameUsageSearchRequest.QueryField.DESCRIPTION));
075    assertTrue(x.getHighlightFields().contains(NameUsageSearchRequest.QueryField.VERNACULAR));
076
077    assertEquals(2, x.getQueryFields().size());
078    assertTrue(x.getQueryFields().contains(NameUsageSearchRequest.QueryField.VERNACULAR));
079    assertTrue(x.getQueryFields().contains(NameUsageSearchRequest.QueryField.SCIENTIFIC));
080  }
081
082  @Test
083  public void testScope() {
084    assertEquals(ComponentScope.PerRequest, provider.getScope());
085  }
086}
