046  public enum TestSearchParameter implements SearchParameter {
047    FIRSTNAME(String.class),
048    LASTNAME(String.class),
049    DATASET_KEY(UUID.class),
050    COUNTRY(Country.class);
051
052    private final Class<?> type;
053
054    TestSearchParameter(Class<?> type) {
055      this.type = type;
056    }
057
058    public Class<?> type() {
059      return type;
060    }
061  }
062
063  private final SearchRequestProvider provider =
064    new SearchRequestProvider(SearchRequest.class, TestSearchParameter.class);
065
066  /**
067   * Tests that empty parameters are excluded by the SearchRequestProvider.
068   */
069  @Test
070  public void testEmptyParameter() {
071    // mock context and request
072    HttpContext ctx = Mockito.mock(HttpContext.class);
073    HttpRequestContext req = Mockito.mock(HttpRequestContext.class);
074    Mockito.when(ctx.getRequest()).thenReturn(req);
075    MultivaluedMap<String, String> params = new MultivaluedMapImpl();
076    Mockito.when(req.getQueryParameters()).thenReturn(params);
077    params.add(TestSearchParameter.FIRSTNAME.name(), "");
078    params.add(TestSearchParameter.FIRSTNAME.name(), null);
079    assertTrue(provider.getValue(ctx).getParameters().get(TestSearchParameter.FIRSTNAME).isEmpty());
080  }
081
082  @Test
083  public void testTypedParameterValidation() {
084    // mock context and request
085    HttpContext ctx = Mockito.mock(HttpContext.class);
086    HttpRequestContext req = Mockito.mock(HttpRequestContext.class);
087    Mockito.when(ctx.getRequest()).thenReturn(req);
088    MultivaluedMap<String, String> params = new MultivaluedMapImpl();
089    Mockito.when(req.getQueryParameters()).thenReturn(params);
090    params.add(TestSearchParameter.COUNTRY.name(), "FR");
091    final UUID uuid = UUID.randomUUID();
092    params.add(TestSearchParameter.DATASET_KEY.name(), uuid.toString());
093
094    SearchRequest<TestSearchParameter> sr = provider.getValue(ctx);
095
096    assertEquals(1, sr.getParameters().get(TestSearchParameter.COUNTRY).size());
097    assertEquals(Country.FRANCE.getIso2LetterCode(), sr.getParameters().get(TestSearchParameter.COUNTRY).iterator()
098      .next());
099
100    assertEquals(1, sr.getParameters().get(TestSearchParameter.DATASET_KEY).size());
101    assertEquals(uuid.toString(), sr.getParameters().get(TestSearchParameter.DATASET_KEY).iterator().next());
102  }

132  @Test
133  public void testGetValue() {
134    // mock context and request
135    HttpContext ctx = Mockito.mock(HttpContext.class);
136    HttpRequestContext req = Mockito.mock(HttpRequestContext.class);
137    Mockito.when(ctx.getRequest()).thenReturn(req);
138
139    MultivaluedMap<String, String> params = new MultivaluedMapImpl();
140    Mockito.when(req.getQueryParameters()).thenReturn(params);
141
142    // test with no params
143    assertEquals(DEFAULT_PARAM_OFFSET, provider.getValue(ctx).getOffset());
144    assertEquals(DEFAULT_PARAM_LIMIT, provider.getValue(ctx).getLimit());
145
146    // test with params
147    params.add(PARAM_OFFSET, "200");
148    params.add(PARAM_LIMIT, "10");
149    params.add(TestSearchParameter.FIRSTNAME.name(), "Carla");
150    assertEquals(200, provider.getValue(ctx).getOffset());
151    assertEquals(10, provider.getValue(ctx).getLimit());
152    assertEquals("Carla", provider.getValue(ctx).getParameters().get(TestSearchParameter.FIRSTNAME).iterator().next());
153    assertTrue(provider.getValue(ctx).getParameters().get(TestSearchParameter.LASTNAME).isEmpty());
154
155    // test case insensitive parameter names
156    params.clear();
157    params.add("firstName", "Carla");
158    assertEquals("Carla", provider.getValue(ctx).getParameters().get(TestSearchParameter.FIRSTNAME).iterator().next());
159    assertTrue(provider.getValue(ctx).getParameters().get(TestSearchParameter.LASTNAME).isEmpty());
160
161    // test with wrong parameter names
162    params.clear();
163    params.add("coolname", "Carla");
164    assertTrue(provider.getValue(ctx).getParameters().isEmpty());
165    assertTrue(provider.getValue(ctx).getParameters().get(TestSearchParameter.FIRSTNAME).isEmpty());
166    assertTrue(provider.getValue(ctx).getParameters().get(TestSearchParameter.LASTNAME).isEmpty());
167
168    // test with wrong paging params
169    params.clear();
170    params.add(PARAM_OFFSET, "twenty");
171    params.add(PARAM_LIMIT, "-10");
172    params.add(PARAM_FACET, "f1");
173    params.add(PARAM_FACET, "f2");
174    assertEquals(DEFAULT_PARAM_OFFSET, provider.getValue(ctx).getOffset());
175    assertEquals(DEFAULT_PARAM_LIMIT, provider.getValue(ctx).getLimit());
176  }
