006import org.gbif.ws.server.GbifHttpServletRequestWrapper;
007import org.gbif.ws.util.SecurityConstants;
008import org.junit.Before;
009import org.junit.Test;
010import org.junit.runner.RunWith;
011import org.mockito.Mock;
012import org.mockito.junit.MockitoJUnitRunner;
013import org.springframework.http.HttpHeaders;
014import org.springframework.security.core.authority.SimpleGrantedAuthority;
015import org.springframework.security.core.context.SecurityContext;

019import javax.servlet.FilterChain;

025
026import static org.junit.Assert.assertEquals;
027import static org.junit.Assert.assertNotNull;
028import static org.junit.Assert.assertNull;
029import static org.mockito.ArgumentMatchers.any;
030import static org.mockito.Mockito.atLeastOnce;

032import static org.mockito.Mockito.verify;
033import static org.mockito.Mockito.when;
034
035/**
036 * Unit tests related to {@link AppIdentityFilter}.
037 */
038@RunWith(MockitoJUnitRunner.class)
039public class AppIdentityFilterTest {
040
041  //initialize in setUp()
042  private AppIdentityFilter appIdentityFilter;
043
044  @Mock
045  private GbifAuthService authServiceMock;
046
047  private SecurityContext context;
048
049  private String content;
050
051  @Before
052  public void setUp() {
053    content = "content";
054    context = new SecurityContextImpl();
055    SecurityContextHolder.setContext(context);
056    AppkeysConfigurationProperties appkeysConfiguration = new AppkeysConfigurationProperties();
057    appkeysConfiguration.setWhitelist(Collections.singletonList("appkey"));
058    appIdentityFilter = new AppIdentityFilter(authServiceMock, appkeysConfiguration);
059  }
060
061  /**
062   * AppIdentityFilter expect the appkey as username.
063   * If the user (header 'x-gbif-user') does not match appkey then the APP role will not be provided.
064   */
065  @Test
066  public void testRandomUsername() throws Exception {
067    // GIVEN
068    HttpServletRequest mockRequest = mock(HttpServletRequest.class);
069    HttpServletResponse mockResponse = mock(HttpServletResponse.class);
070    FilterChain chain = mock(FilterChain.class);
071
072    when(mockRequest.getHeader(HttpHeaders.AUTHORIZATION)).thenReturn("GBIF appkey:blabla");
073    when(mockRequest.getHeader(SecurityConstants.HEADER_GBIF_USER)).thenReturn("myuser"); // user does not match appkey
074    when(mockRequest.getInputStream()).thenReturn(new DelegatingServletInputStream(new ByteArrayInputStream(content.getBytes())));
075    when(authServiceMock.isValidRequest(any(GbifHttpServletRequestWrapper.class))).thenReturn(true);
076
077    // WHEN
078    appIdentityFilter.doFilter(mockRequest, mockResponse, chain);
079
080    // THEN
081    assertNull(context.getAuthentication());
082    verify(mockRequest).getHeader(HttpHeaders.AUTHORIZATION);
083    verify(mockRequest).getHeader(SecurityConstants.HEADER_GBIF_USER);
084    verify(mockRequest, atLeastOnce()).getInputStream();
085    verify(authServiceMock).isValidRequest(any(GbifHttpServletRequestWrapper.class));
086  }
087
088  /**
089   * Try with the appkey which matches the user header but not present in the white list.
090   * So the APP role will not be provided.
091   */
092  @Test
093  public void testAppkeyNotInWhiteList() throws Exception {
094    // GIVEN
095    HttpServletRequest mockRequest = mock(HttpServletRequest.class);
096    HttpServletResponse mockResponse = mock(HttpServletResponse.class);
097    FilterChain chain = mock(FilterChain.class);
098
099    when(mockRequest.getHeader(HttpHeaders.AUTHORIZATION)).thenReturn("GBIF myuser:blabla");
100    when(mockRequest.getHeader(SecurityConstants.HEADER_GBIF_USER)).thenReturn("myuser"); // user does not match appkey
101    when(mockRequest.getInputStream()).thenReturn(new DelegatingServletInputStream(new ByteArrayInputStream(content.getBytes())));
102    when(authServiceMock.isValidRequest(any(GbifHttpServletRequestWrapper.class))).thenReturn(true);
103
104    // WHEN
105    appIdentityFilter.doFilter(mockRequest, mockResponse, chain);
106
107    // THEN
108    assertNull(context.getAuthentication());
109    verify(mockRequest).getHeader(HttpHeaders.AUTHORIZATION);
110    verify(mockRequest).getHeader(SecurityConstants.HEADER_GBIF_USER);
111    verify(mockRequest, atLeastOnce()).getInputStream();
112    verify(authServiceMock).isValidRequest(any(GbifHttpServletRequestWrapper.class));
113  }
114
115  /**
116   * Try with the appkey which matches the user header and presents in the white list.
117   * The user should be successfully authenticated with the APP role.
118   */
119  @Test
120  public void testRightRequest() throws Exception {
121    // GIVEN
122    HttpServletRequest mockRequest = mock(HttpServletRequest.class);
123    HttpServletResponse mockResponse = mock(HttpServletResponse.class);
124    FilterChain chain = mock(FilterChain.class);
125
126    when(mockRequest.getHeader(HttpHeaders.AUTHORIZATION)).thenReturn("GBIF appkey:blabla");
127    when(mockRequest.getHeader(SecurityConstants.HEADER_GBIF_USER)).thenReturn("appkey");
128    when(mockRequest.getInputStream()).thenReturn(new DelegatingServletInputStream(new ByteArrayInputStream(content.getBytes())));
129    when(authServiceMock.isValidRequest(any(GbifHttpServletRequestWrapper.class))).thenReturn(true);
130
131    // WHEN
132    appIdentityFilter.doFilter(mockRequest, mockResponse, chain);
133
134    // THEN
135    assertNotNull(context.getAuthentication());
136    assertEquals("appkey", context.getAuthentication().getName());
137    assertNotNull(context.getAuthentication().getAuthorities());
138    assertEquals("APP", ((List<SimpleGrantedAuthority>) context.getAuthentication().getAuthorities()).get(0).getAuthority());
139    verify(mockRequest).getHeader(HttpHeaders.AUTHORIZATION);
140    verify(mockRequest).getHeader(SecurityConstants.HEADER_GBIF_USER);
141    verify(mockRequest, atLeastOnce()).getInputStream();
142    verify(authServiceMock).isValidRequest(any(GbifHttpServletRequestWrapper.class));
143  }
