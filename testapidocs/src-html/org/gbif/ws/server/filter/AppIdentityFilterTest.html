050/**
051 * Unit tests related to {@link AppIdentityFilter}.
052 */
053@ExtendWith(MockitoExtension.class)
054public class AppIdentityFilterTest {
055
056  // initialize in setUp()
057  private AppIdentityFilter appIdentityFilter;
058
059  @Mock private GbifAuthService authServiceMock;
060
061  private SecurityContext context;
062
063  private String content;
064
065  @BeforeEach
066  public void setUp() {
067    content = "content";
068    context = new SecurityContextImpl();
069    SecurityContextHolder.setContext(context);
070    AppkeysConfigurationProperties appkeysConfiguration = new AppkeysConfigurationProperties();
071    appkeysConfiguration.setWhitelist(Collections.singletonList("appkey"));
072    appIdentityFilter = new AppIdentityFilter(authServiceMock, appkeysConfiguration);
073  }
074
075  /**
076   * AppIdentityFilter expect the appkey as username.
077   * If the user (header 'x-gbif-user') does not match appkey then the APP role will not be provided.
078   */
079  @Test
080  public void testRandomUsername() throws Exception {
081    // GIVEN
082    HttpServletRequest mockRequest = mock(HttpServletRequest.class);
083    HttpServletResponse mockResponse = mock(HttpServletResponse.class);
084    FilterChain chain = mock(FilterChain.class);
085
086    when(mockRequest.getHeader(HttpHeaders.AUTHORIZATION)).thenReturn("GBIF appkey:blabla");
087    when(mockRequest.getHeader(SecurityConstants.HEADER_GBIF_USER))
088        .thenReturn("myuser"); // user does not match appkey
089    when(mockRequest.getInputStream())
090        .thenReturn(new DelegatingServletInputStream(new ByteArrayInputStream(content.getBytes())));
091    when(authServiceMock.isValidRequest(any(GbifHttpServletRequestWrapper.class))).thenReturn(true);
092
093    // WHEN
094    appIdentityFilter.doFilter(mockRequest, mockResponse, chain);
095
096    // THEN
097    assertNull(context.getAuthentication());
098    verify(mockRequest).getHeader(HttpHeaders.AUTHORIZATION);
099    verify(mockRequest).getHeader(SecurityConstants.HEADER_GBIF_USER);
100    verify(mockRequest, atLeastOnce()).getInputStream();
101    verify(authServiceMock).isValidRequest(any(GbifHttpServletRequestWrapper.class));
102  }
103
104  /**
105   * Try with the appkey which matches the user header but not present in the white list.
106   * So the APP role will not be provided.
107   */
108  @Test
109  public void testAppkeyNotInWhiteList() throws Exception {

132
133  /**
134   * Try with the appkey which matches the user header and presents in the white list.
135   * The user should be successfully authenticated with the APP role.
136   */
137  @Test
138  public void testRightRequest() throws Exception {
139    // GIVEN
140    HttpServletRequest mockRequest = mock(HttpServletRequest.class);
141    HttpServletResponse mockResponse = mock(HttpServletResponse.class);
142    FilterChain chain = mock(FilterChain.class);
143
144    when(mockRequest.getHeader(HttpHeaders.AUTHORIZATION)).thenReturn("GBIF appkey:blabla");
145    when(mockRequest.getHeader(SecurityConstants.HEADER_GBIF_USER)).thenReturn("appkey");
146    when(mockRequest.getInputStream())
147        .thenReturn(new DelegatingServletInputStream(new ByteArrayInputStream(content.getBytes())));
148    when(authServiceMock.isValidRequest(any(GbifHttpServletRequestWrapper.class))).thenReturn(true);
149
150    // WHEN
151    appIdentityFilter.doFilter(mockRequest, mockResponse, chain);
152
153    // THEN
154    assertNotNull(context.getAuthentication());
155    assertEquals("appkey", context.getAuthentication().getName());
156    assertNotNull(context.getAuthentication().getAuthorities());
157    assertEquals(
158        "APP",
159        ((List<SimpleGrantedAuthority>) context.getAuthentication().getAuthorities())
160            .get(0)
161            .getAuthority());
162    verify(mockRequest).getHeader(HttpHeaders.AUTHORIZATION);
163    verify(mockRequest).getHeader(SecurityConstants.HEADER_GBIF_USER);
164    verify(mockRequest, atLeastOnce()).getInputStream();
165    verify(authServiceMock).isValidRequest(any(GbifHttpServletRequestWrapper.class));
166  }
