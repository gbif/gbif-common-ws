036@RunWith(MockitoJUnitRunner.class)
037public class AuthFilterTest {
038
039  @Mock
040  UserService userService;
041  @Mock
042  ContainerRequest mockRequest;
043  ArgumentCaptor<SecurityContext> secCtxCaptor = ArgumentCaptor.forClass(SecurityContext.class);
044
045  AuthFilter filter;
046  static String adminAuthGbif;
047  static String heinzAuthBasic;
048  static String legacyBasic;
049
050  static User admin = new User();
051  static User heinz = new User();
052
053  static {
054    admin.setUserName("admin");
055    admin.setEmail("admin@mailinator.com");
056    admin.getRoles().add(UserRole.ADMIN);
057    admin.getRoles().add(UserRole.USER);
058
059    heinz.setUserName("heinz");
060    heinz.setEmail("heinz@mailinator.com");
061    heinz.getRoles().add(UserRole.USER);
062
063    try {
064      adminAuthGbif = "GBIF " + new String(Base64.encode("1234567890:admin"), "UTF8");
065      heinzAuthBasic = "Basic " + new String(Base64.encode("heinz:HEINZ"), "UTF8");
066      legacyBasic = "Basic " + new String(Base64.encode(UUID.randomUUID().toString() + ":password"), "UTF8");
067    } catch (UnsupportedEncodingException e) {
068      // TODO: Handle exception
069    }
070  }
071
072  @Before
073  public void initMocks() {
074    when(userService.get(Matchers.<String>eq("admin"))).thenReturn(admin);
075    when(userService.authenticate(Matchers.<String>eq("admin"), Matchers.<String>any())).thenReturn(admin);
076    when(userService.get(Matchers.<String>eq("heinz"))).thenReturn(heinz);
077    when(userService.authenticate(Matchers.<String>eq("heinz"), Matchers.<String>any())).thenReturn(heinz);
078    // setup basic POST request
079    when(mockRequest.getQueryParameters()).thenReturn(new MultivaluedMapImpl());
080    when(mockRequest.getMethod()).thenReturn("POST");
081    setMockAuth(null);
082
083    GbifAuthService authService = GbifAuthService.multiKeyAuthService(GbifAuthServiceTest.buildAppKeyMap());
084    filter = new AuthFilter(userService, authService);
085  }
086
087  @Test
088  public void testFilterWithAnonymous() throws Exception {
089    assertPrincipalName(null);
090
091    setMockAuth("");
092    assertPrincipalName(null);
093
094    setMockAuth("Digest ");
095    assertPrincipalName(null);
096  }
097
098  @Test
099  public void testFilterWithLegacyAccount() throws Exception {
100    setMockAuth(legacyBasic);
101    assertPrincipalName(null);
102  }
103
104  @Test
105  public void testFilterWithBasicHeinz() throws Exception {
106    setMockAuth(heinzAuthBasic);
107    assertPrincipalName("heinz");
108  }
109

117
118  @Test
119  public void testFilterWithGET() throws Exception {
120    when(mockRequest.getMethod()).thenReturn("GET");
121    assertPrincipalName(null);
122  }
123
124  private void assertPrincipalName(String username) {
125    ContainerRequest req = filter.filter(mockRequest);
126    if (username == null) {
127      verify(req, atLeastOnce()).setSecurityContext(secCtxCaptor.capture());
128      assertNull(secCtxCaptor.getValue().getUserPrincipal());
129      assertFalse(secCtxCaptor.getValue().isUserInRole("ADMIN"));
130    } else {
131      verify(req, atLeastOnce()).setSecurityContext(secCtxCaptor.capture());
132      assertEquals(username, secCtxCaptor.getValue().getUserPrincipal().getName());
133      assertTrue(secCtxCaptor.getValue().isUserInRole("ADMIN") || secCtxCaptor.getValue().isUserInRole("USER"));
134    }
135  }
136
137  /**
138   * @param expectedType if null expect to be authorized
139   */
140  private void assertResponseStatus(Response.Status expectedType) {
141    boolean webexcept = false;
142    try {
143      ContainerRequest req = filter.filter(mockRequest);
144    } catch (WebApplicationException e) {
145      webexcept = true;
146      assertWebApplicationExceptionOfType(e, expectedType);
147    }
148    if (!webexcept && expectedType != null) {
149      fail("Expected WebApplicationException of type " + expectedType);
150    }
151  }
152
153  private void assertWebApplicationExceptionOfType(WebApplicationException e, Response.Status expectedType) {
154    assertTrue(expectedType.getStatusCode() == e.getResponse().getStatus());
155  }
156
157  private void setMockAuth(String authentication) {
158    when(mockRequest.getHeaderValue(Matchers.<String>eq(ContainerRequest.AUTHORIZATION))).thenReturn(authentication);
159  }
