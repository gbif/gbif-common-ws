001package org.gbif.ws.server.interceptor;
002
003import static org.junit.Assert.assertNull;
004
005import org.gbif.api.model.registry.Citation;
006import org.gbif.api.model.registry.Dataset;
007import org.junit.Test;
008
009public class StringTrimInterceptorTest {
010
011  private static final StringTrimInterceptor TRIMMER = new StringTrimInterceptor();
012
013  @SuppressWarnings("ConstantConditions")
014  @Test
015  public void test() {
016    Dataset dataset = new Dataset();
017    dataset.setTitle("   ");
018    TRIMMER.trimStringsOf(dataset);
019    assertNull("Dataset title should be null", dataset.getTitle());
020
021    Citation citation = new Citation();
022    citation.setText("");
023    dataset.setCitation(citation);
024    TRIMMER.trimStringsOf(dataset);
025    assertNull("Citation text should be null", dataset.getCitation().getText());
026  }
027}
